### 智能推荐算法

```python
# 推荐算法伪代码
def generate_recommendations(current_doc, user_profile):
    recommendations = []
    
    # 1. 前置知识推荐
    prerequisites = get_prerequisites(current_doc)
    for prereq in prerequisites:
        if prereq not in user_profile.mastered:
            recommendations.append({
                "type": "prerequisite",
                "doc": prereq,
                "priority": "high"
            })
    
    # 2. 后续学习推荐
    successors = get_successors(current_doc)
    ready = [s for s in successors 
             if all(p in user_profile.mastered for p in get_prerequisites(s))]
    for succ in ready[:3]:
        recommendations.append({
            "type": "successor",
            "doc": succ,
            "priority": "medium"
        })
    
    # 3. 相关主题推荐
    related = get_related_by_content(current_doc)
    for rel in related[:3]:
        recommendations.append({
            "type": "related",
            "doc": rel,
            "priority": "low"
        })
    
    return recommendations
```
