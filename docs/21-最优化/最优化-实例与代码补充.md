# 最优化实例与代码实现

## 实例1：生产计划优化（线性规划）

### 问题描述
某工厂生产A、B两种产品：
- 产品A：利润300元/件，需要机器2小时，人工3小时
- 产品B：利润500元/件，需要机器4小时，人工2小时

资源限制：
- 机器总时间：100小时
- 人工总时间：90小时

求：最大化利润的生产计划

### 数学模型
$$\max \quad 300x_1 + 500x_2$$
$$\text{s.t.} \quad 2x_1 + 4x_2 \leq 100$$
$$\quad\quad 3x_1 + 2x_2 \leq 90$$
$$\quad\quad x_1, x_2 \geq 0$$

### Python实现（PuLP库）

```python
import pulp

# 创建问题
prob = pulp.LpProblem("Production_Planning", pulp.LpMaximize)

# 决策变量
x1 = pulp.LpVariable("Product_A", lowBound=0, cat='Continuous')
x2 = pulp.LpVariable("Product_B", lowBound=0, cat='Continuous')

# 目标函数
prob += 300*x1 + 500*x2, "Total_Profit"

# 约束条件
prob += 2*x1 + 4*x2 <= 100, "Machine_Hours"
prob += 3*x1 + 2*x2 <= 90, "Labor_Hours"

# 求解
prob.solve()

# 输出结果
print(f"最优解:")
print(f"  产品A产量: {pulp.value(x1):.2f} 件")
print(f"  产品B产量: {pulp.value(x2):.2f} 件")
print(f"  最大利润: {pulp.value(prob.objective):.2f} 元")

# 对偶变量（影子价格）
print(f"\n资源影子价格:")
for name, c in prob.constraints.items():
    print(f"  {name}: {c.pi:.2f}")
```

### 运行结果
```
最优解:
  产品A产量: 10.00 件
  产品B产量: 20.00 件
  最大利润: 13000.00 元

资源影子价格:
  Machine_Hours: 87.50
  Labor_Hours: 37.50
```

### 敏感性分析

```python
# 参数敏感性分析
import numpy as np
import matplotlib.pyplot as plt

profits_A = np.linspace(200, 400, 100)
optimal_x1 = []
optimal_x2 = []

for profit_A in profits_A:
    prob = pulp.LpProblem("Sensitivity", pulp.LpMaximize)
    x1 = pulp.LpVariable("x1", lowBound=0)
    x2 = pulp.LpVariable("x2", lowBound=0)
    prob += profit_A*x1 + 500*x2
    prob += 2*x1 + 4*x2 <= 100
    prob += 3*x1 + 2*x2 <= 90
    prob.solve()
    optimal_x1.append(pulp.value(x1))
    optimal_x2.append(pulp.value(x2))

plt.plot(profits_A, optimal_x1, label='Product A')
plt.plot(profits_A, optimal_x2, label='Product B')
plt.xlabel('Profit of Product A')
plt.ylabel('Optimal Production')
plt.legend()
plt.title('Sensitivity Analysis')
plt.savefig('sensitivity_analysis.png')
```

## 实例2：投资组合优化（二次规划）

### 问题描述
投资者有3种资产可选：
- 资产1：预期收益10%，风险（标准差）15%
- 资产2：预期收益12%，风险（标准差）18%
- 资产3：预期收益8%，风险（标准差）10%

相关系数矩阵：
$$\rho = \begin{pmatrix} 1 & 0.3 & 0.1 \\ 0.3 & 1 & 0.2 \\ 0.1 & 0.2 & 1 \end{pmatrix}$$

目标：在预期收益≥10%的条件下，最小化投资组合风险

### Python实现（CVXPY库）

```python
import cvxpy as cp
import numpy as np

# 参数设置
expected_returns = np.array([0.10, 0.12, 0.08])
std_devs = np.array([0.15, 0.18, 0.10])
corr_matrix = np.array([[1.0, 0.3, 0.1],
                        [0.3, 1.0, 0.2],
                        [0.1, 0.2, 1.0]])

# 计算协方差矩阵
cov_matrix = np.outer(std_devs, std_devs) * corr_matrix

# 决策变量
weights = cp.Variable(3)

# 目标函数：最小化方差
portfolio_variance = cp.quad_form(weights, cov_matrix)
objective = cp.Minimize(portfolio_variance)

# 约束条件
constraints = [
    cp.sum(weights) == 1,  # 权重和为1
    weights >= 0,  # 不允许卖空
    expected_returns @ weights >= 0.10  # 预期收益约束
]

# 求解
prob = cp.Problem(objective, constraints)
result = prob.solve()

print(f"最优投资组合:")
for i, w in enumerate(weights.value):
    print(f"  资产{i+1}: {w:.4f} ({w*100:.2f}%)")

portfolio_return = expected_returns @ weights.value
portfolio_risk = np.sqrt(portfolio_variance.value)
print(f"\n组合预期收益: {portfolio_return:.4f} ({portfolio_return*100:.2f}%)")
print(f"组合风险(标准差): {portfolio_risk:.4f} ({portfolio_risk*100:.2f}%)")
```

## 实例3：梯度下降优化

### 目标函数
$$f(x, y) = x^2 + 2y^2 + 2\sin(2\pi x)\sin(2\pi y)$$

### Python实现

```python
import numpy as np
import matplotlib.pyplot as plt
from mpl_toolkits.mplot3d import Axes3D

def f(x, y):
    """目标函数"""
    return x**2 + 2*y**2 + 2*np.sin(2*np.pi*x)*np.sin(2*np.pi*y)

def grad_f(x, y):
    """梯度"""
    df_dx = 2*x + 4*np.pi*np.cos(2*np.pi*x)*np.sin(2*np.pi*y)
    df_dy = 4*y + 4*np.pi*np.sin(2*np.pi*x)*np.cos(2*np.pi*y)
    return np.array([df_dx, df_dy])

def gradient_descent(start, learning_rate, n_iterations):
    """梯度下降"""
    path = [start]
    x = np.array(start)
    
    for i in range(n_iterations):
        gradient = grad_f(x[0], x[1])
        x = x - learning_rate * gradient
        path.append(x.copy())
        
        # 收敛检查
        if np.linalg.norm(gradient) < 1e-6:
            print(f"收敛于迭代 {i}")
            break
    
    return np.array(path)

# 运行优化
start_point = np.array([0.5, 0.5])
learning_rate = 0.01
path = gradient_descent(start_point, learning_rate, 1000)

print(f"起点: {start_point}")
print(f"终点: {path[-1]}")
print(f"最小值: {f(path[-1][0], path[-1][1]):.6f}")

# 可视化
x = np.linspace(-1, 1, 100)
y = np.linspace(-1, 1, 100)
X, Y = np.meshgrid(x, y)
Z = f(X, Y)

plt.figure(figsize=(10, 8))
contour = plt.contour(X, Y, Z, levels=20)
plt.clabel(contour, inline=True, fontsize=8)
plt.plot(path[:, 0], path[:, 1], 'r-', linewidth=2, label='Gradient Descent Path')
plt.plot(path[0, 0], path[0, 1], 'go', markersize=10, label='Start')
plt.plot(path[-1, 0], path[-1, 1], 'r*', markersize=15, label='End')
plt.xlabel('x')
plt.ylabel('y')
plt.legend()
plt.title('Gradient Descent Optimization')
plt.colorbar(contour)
plt.savefig('gradient_descent.png')
```

### 习题1
求解：
$$\min \quad x^2 + y^2$$
$$\text{s.t.} \quad x + y = 1$$

**解答**：使用拉格朗日乘子法

$L = x^2 + y^2 - \lambda(x + y - 1)$

$\frac{\partial L}{\partial x} = 2x - \lambda = 0$
$\frac{\partial L}{\partial y} = 2y - \lambda = 0$
$\frac{\partial L}{\partial \lambda} = -(x + y - 1) = 0$

由前两式：$x = y$，代入约束：$2x = 1$，$x = y = 0.5$

最优值：$0.5^2 + 0.5^2 = 0.5$

### 习题2
证明：凸函数的局部最小值是全局最小值

**证明**：设 $f$ 凸，$x^*$ 是局部最小值。假设存在 $y$ 使 $f(y) < f(x^*)$。

由凸性：$f(\lambda y + (1-\lambda)x^*) \leq \lambda f(y) + (1-\lambda)f(x^*) < f(x^*)$

当 $\lambda \to 0$，$\lambda y + (1-\lambda)x^*$ 在 $x^*$ 任意邻域内，矛盾。
