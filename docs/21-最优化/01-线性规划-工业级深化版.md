## 1. 工业级案例：供应链网络优化

### 1.1 问题背景

**真实场景**: 某跨国制造企业拥有3个工厂、5个分销中心，需服务8个市场区域。目标是优化生产和物流成本。

**数据来源**: 基于Kaggle Supply Chain Dataset改编，包含真实物流成本结构。

### 1.2 数学模型

**决策变量**:
- $x_{ij}$: 从工厂 $i$ 到分销中心 $j$ 的运输量（千单位）
- $y_{jk}$: 从分销中心 $j$ 到市场 $k$ 的运输量（千单位）
- $p_i$: 工厂 $i$ 的生产量（千单位）

**参数**（真实工业数据量级）:

| 工厂 | 产能(千单位/月) | 生产成本(元/单位) |
|------|-----------------|-------------------|
| 上海 | 50 | 45 |
| 深圳 | 40 | 42 |
| 成都 | 35 | 40 |

| 分销中心 | 处理能力(千单位/月) | 运营成本(元/单位) |
|----------|---------------------|-------------------|
| 北京 | 30 | 8 |
| 上海 | 35 | 7 |
| 广州 | 30 | 7 |
| 武汉 | 25 | 6 |
| 西安 | 20 | 6 |

| 路线 | 运输成本(元/单位) |
|------|-------------------|
| 工厂→分销中心 | 2-8 |
| 分销中心→市场 | 3-12 |

**线性规划模型**:

$$
\begin{aligned}
\min \quad & Z = \sum_{i=1}^{3} c_i^{prod} p_i + \sum_{i,j} c_{ij}^{trans1} x_{ij} + \sum_{j} c_j^{oper} \sum_k y_{jk} + \sum_{j,k} c_{jk}^{trans2} y_{jk} \\
\text{s.t.} \quad & \sum_j x_{ij} = p_i, \quad \forall i \quad \text{(生产平衡)} \\
& \sum_i x_{ij} = \sum_k y_{jk}, \quad \forall j \quad \text{(分销中心平衡)} \\
& \sum_j y_{jk} = d_k, \quad \forall k \quad \text{(需求满足)} \\
& p_i \leq C_i^{prod}, \quad \forall i \quad \text{(产能约束)} \\
& \sum_i x_{ij} \leq C_j^{dc}, \quad \forall j \quad \text{(处理能力)} \\
& x_{ij}, y_{jk}, p_i \geq 0
\end{aligned}
$$

**市场需求**（千单位/月）:

```python
    '华中': 15, '西南': 18, '西北': 12, '东北': 14, '东南': 8
}
```

```python
"""
供应链网络优化 - 线性规划实现
作者: FormalMath应用数学组

import numpy as np
import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns
from scipy.optimize import linprog
import cvxpy as cp
from typing import Dict, Tuple, List
import warnings
warnings.filterwarnings('ignore')

# 设置中文字体
plt.rcParams['font.sans-serif'] = ['SimHei', 'DejaVu Sans']
plt.rcParams['axes.unicode_minus'] = False

class SupplyChainOptimizer:
    """
    供应链网络优化器
    基于真实工业数据建模
    """
    
    def __init__(self):
        # 工厂数据
        self.factories = ['上海', '深圳', '成都']
        self.prod_capacity = np.array([50, 40, 35])  # 产能
        self.prod_cost = np.array([45, 42, 40])  # 生产成本
        
        # 分销中心数据
        self.dc = ['北京', '上海', '广州', '武汉', '西安']
        self.dc_capacity = np.array([30, 35, 30, 25, 20])
        self.dc_oper_cost = np.array([8, 7, 7, 6, 6])
        
        # 市场需求
        self.markets = ['华北', '华东', '华南', '华中', '西南', '西北', '东北', '东南']
        self.demand = np.array([25, 30, 28, 15, 18, 12, 14, 8])
        
        # 运输成本 (工厂→分销中心)
        self.trans_f2d = np.array([
            [5, 3, 7, 6, 8],   # 上海→各DC
            [8, 4, 2, 5, 7],   # 深圳→各DC
            [7, 6, 5, 4, 3]    # 成都→各DC
        ])
        
        # 运输成本 (分销中心→市场)
        self.trans_d2m = np.array([
            [2, 6, 8, 5, 7, 4, 3, 7],   # 北京
            [6, 2, 5, 4, 7, 8, 6, 4],   # 上海
            [8, 5, 2, 4, 6, 9, 8, 3],   # 广州
            [5, 4, 4, 2, 3, 5, 6, 5],   # 武汉
            [4, 7, 8, 5, 2, 3, 5, 8]    # 西安
        ])
        
        self.n_factories = len(self.factories)
        self.n_dc = len(self.dc)
        self.n_markets = len(self.markets)
        
    def solve_scipy(self) -> Dict:
        """
        使用SciPy求解线性规划
        """
        # 变量顺序: [p1,p2,p3, x11,x12,...,x15, x21,...,x35, y11,y12,...,y58]
        n_vars = self.n_factories + self.n_factories * self.n_dc + self.n_dc * self.n_markets
        
        # 目标函数系数
        c = []
        # 生产成本
        c.extend(self.prod_cost)
        # 工厂到DC运输成本
        for i in range(self.n_factories):
            c.extend(self.trans_f2d[i])
        # DC运营成本 + DC到市场运输成本
        for j in range(self.n_dc):
            for k in range(self.n_markets):
                c.append(self.dc_oper_cost[j] + self.trans_d2m[j, k])
        
        c = np.array(c)
        
        # 等式约束
        A_eq = []
        b_eq = []
        
        # 1. 生产平衡: p_i = sum_j x_ij
        for i in range(self.n_factories):
            row = np.zeros(n_vars)
            row[i] = 1  # p_i
            # x_ij 项
            for j in range(self.n_dc):
                idx = self.n_factories + i * self.n_dc + j
                row[idx] = -1
            A_eq.append(row)
            b_eq.append(0)
        
        # 2. DC平衡: sum_i x_ij = sum_k y_jk
        for j in range(self.n_dc):
            row = np.zeros(n_vars)
            # sum_i x_ij
            for i in range(self.n_factories):
                idx = self.n_factories + i * self.n_dc + j
                row[idx] = 1
            # -sum_k y_jk
            for k in range(self.n_markets):
                idx = self.n_factories + self.n_factories * self.n_dc + j * self.n_markets + k
                row[idx] = -1
            A_eq.append(row)
            b_eq.append(0)
        
        # 3. 需求满足: sum_j y_jk = d_k
        for k in range(self.n_markets):
            row = np.zeros(n_vars)
            for j in range(self.n_dc):
                idx = self.n_factories + self.n_factories * self.n_dc + j * self.n_markets + k
                row[idx] = 1
            A_eq.append(row)
            b_eq.append(self.demand[k])
        
        A_eq = np.array(A_eq)
        b_eq = np.array(b_eq)
        
        # 不等式约束 (产能限制)
        A_ub = []
        b_ub = []
        
        # 工厂产能
        for i in range(self.n_factories):
            row = np.zeros(n_vars)
            row[i] = 1
            A_ub.append(row)
            b_ub.append(self.prod_capacity[i])
        
        # DC处理能力
        for j in range(self.n_dc):
            row = np.zeros(n_vars)
            for i in range(self.n_factories):
                idx = self.n_factories + i * self.n_dc + j
                row[idx] = 1
            A_ub.append(row)
            b_ub.append(self.dc_capacity[j])
        
        A_ub = np.array(A_ub)
        b_ub = np.array(b_ub)
        
        # 求解
        result = linprog(c, A_ub=A_ub, b_ub=b_ub, A_eq=A_eq, b_eq=b_eq, 
                        bounds=[(0, None)] * n_vars, method='highs')
        
        return result
    
    def solve_cvxpy(self) -> Dict:
        """
        使用CVXPY求解（更直观的建模方式）
        """
        # 决策变量
        p = cp.Variable(self.n_factories, nonneg=True)  # 生产量
        x = cp.Variable((self.n_factories, self.n_dc), nonneg=True)  # 工厂→DC
        y = cp.Variable((self.n_dc, self.n_markets), nonneg=True)  # DC→市场
        
        # 目标函数
        production_cost = self.prod_cost @ p
        transport_f2d = cp.sum(cp.multiply(self.trans_f2d, x))
        dc_oper = self.dc_oper_cost @ cp.sum(y, axis=1)
        transport_d2m = cp.sum(cp.multiply(self.trans_d2m, y))
        
        objective = cp.Minimize(production_cost + transport_f2d + dc_oper + transport_d2m)
        
        # 约束
        constraints = [
            # 生产平衡
            cp.sum(x, axis=1) == p,
            # DC平衡
            cp.sum(x, axis=0) == cp.sum(y, axis=1),
            # 需求满足
            cp.sum(y, axis=0) == self.demand,
            # 产能约束
            p <= self.prod_capacity,
            # DC处理能力
            cp.sum(x, axis=0) <= self.dc_capacity
        ]
        
        # 求解
        problem = cp.Problem(objective, constraints)
        result = problem.solve(solver=cp.ECOS)
        
        return {
            'optimal_value': result,
            'production': p.value,
            'factory_to_dc': x.value,
            'dc_to_market': y.value,
            'shadow_prices': [c.dual_value for c in constraints]
        }
    
    def analyze_results(self, cvx_result: Dict):
        """
        结果分析
        """
        print("=" * 60)
        print("供应链优化结果分析")
        print("=" * 60)
        
        print(f"\n总成本: {cvx_result['optimal_value']:,.2f} 万元/月")
        
        print("\n【生产计划】")
        for i, factory in enumerate(self.factories):
            prod = cvx_result['production'][i]
            util = prod / self.prod_capacity[i] * 100
            cost = prod * self.prod_cost[i]
            print(f"  {factory}: {prod:.2f}千单位 (利用率{util:.1f}%, 成本{cost:.2f}万元)")
        
        print("\n【分销中心流量】")
        x = cvx_result['factory_to_dc']
        for j, dc_name in enumerate(self.dc):
            inflow = sum(x[i, j] for i in range(self.n_factories))
            util = inflow / self.dc_capacity[j] * 100
            print(f"  {dc_name}: 流入{inflow:.2f}千单位 (利用率{util:.1f}%)")
        
        print("\n【主要运输路线】")
        for i in range(self.n_factories):
            for j in range(self.n_dc):
                if x[i, j] > 1:
                    print(f"  {self.factories[i]}→{self.dc[j]}: {x[i,j]:.2f}千单位")

# 运行优化
if __name__ == "__main__":
    optimizer = SupplyChainOptimizer()
    
    print("使用SciPy求解...")
    result_scipy = optimizer.solve_scipy()
    print(f"SciPy结果: 最优值 = {result_scipy.fun:,.2f}, 成功 = {result_scipy.success}")
    
    print("\n使用CVXPY求解...")
    result_cvxpy = optimizer.solve_cvxpy()
    optimizer.analyze_results(result_cvxpy)
```

```python
def error_analysis():
    """
    数值误差分析
    """
    # 测试不同求解器的精度
    np.random.seed(42)
    n = 50
    c = np.random.randn(n)
    A = np.random.randn(20, n)
    b = A @ np.random.rand(n) + 0.1  # 确保可行
    
    methods = ['highs', 'highs-ds', 'highs-ipm']
    results = {}
    
    for method in methods:
        result = linprog(c, A_ub=A, b_ub=b, bounds=[(0, None)]*n, method=method)

        print(f"{method:<15} {res['value']:<15.6e} {res['violation']:<15.2e} {res['nit']:<10}")
    
    return results
```

## 4. 对偶理论与经济解释

### 4.1 对偶问题推导

原问题(P):
$$
\begin{aligned}
\min \quad & c^T x \\
\text{s.t.} \quad & Ax = b \\
& x \geq 0
\end{aligned}
$$

对偶问题(D):
$$
\begin{aligned}
\max \quad & b^T y \\
\text{s.t.} \quad & A^T y \leq c
\end{aligned}
$$

**经济解释**: 对偶变量 $y_i$ 表示第 $i$ 个资源的影子价格（边际价值）。
