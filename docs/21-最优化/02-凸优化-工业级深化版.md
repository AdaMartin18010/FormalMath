### 1.3 均值-方差优化模型

**Markowitz均值-方差模型**:

$$
\begin{aligned}
\min_{x} \quad & x^T \Sigma x \quad \text{(组合方差)} \\
\text{s.t.} \quad & \mu^T x \geq R_{target} \quad \text{(收益目标)} \\
& \mathbf{1}^T x = 1 \quad \text{(预算约束)} \\
& x \geq 0 \quad \text{(禁止卖空)} \\
& x_i \leq u_i \quad \text{(个股权重上限)}
\end{aligned}
$$

**扩展约束**:
- 行业分散约束
- 换手率约束
- 跟踪误差约束

```python
class PortfolioOptimizer:
    """
    投资组合优化器
    实现多种凸优化模型
    """
    
    def __init__(self, data: PortfolioData):
        self.data = data
        self.n_assets = len(data.TICKERS)
        self.mu = data.mean_returns
        self.Sigma = data.cov_matrix
        
    def efficient_frontier(self, n_points=50):
        """
        计算有效前沿
        """
        x = cp.Variable(self.n_assets)
        
        # 预期收益范围
        mu_min = self.mu.min()
        mu_max = self.mu.max() * 1.2
        target_returns = np.linspace(mu_min, mu_max, n_points)
        
        risks = []
        returns = []
        portfolios = []
        
        for target in target_returns:
            # 最小化风险，约束收益
            objective = cp.Minimize(cp.quad_form(x, self.Sigma))
            constraints = [
                self.mu @ x >= target,
                cp.sum(x) == 1,
                x >= 0,
                x <= 0.3  # 单资产上限30%
            ]
            
            prob = cp.Problem(objective, constraints)
            prob.solve()
            
            if prob.status == 'optimal':
                risks.append(np.sqrt(x.value @ self.Sigma @ x.value))
                returns.append(self.mu @ x.value)
                portfolios.append(x.value.copy())
        
        return np.array(risks), np.array(returns), portfolios
    
    def maximize_sharpe_ratio(self, risk_free_rate=0.04):
        """
        最大化夏普比率（转化为凸问题）
        """
        y = cp.Variable(self.n_assets)
        kappa = cp.Variable()
        
        # 最大化夏普比率 = 最大化 (mu - rf)^T x / sqrt(x^T Sigma x)
        # 等价于：最小化 y^T Sigma y，约束 (mu - rf)^T y = 1，然后 x = y / sum(y)
        
        objective = cp.Minimize(cp.quad_form(y, self.Sigma))
        constraints = [
            (self.mu - risk_free_rate) @ y == 1,
            cp.sum(y) == kappa,
            y >= 0,
            kappa >= 0
        ]
        
        prob = cp.Problem(objective, constraints)
        prob.solve()
        
        if prob.status == 'optimal':
            x_opt = y.value / np.sum(y.value)
            sharpe = (self.mu @ x_opt - risk_free_rate) / np.sqrt(x_opt @ self.Sigma @ x_opt)
            return x_opt, sharpe, np.sqrt(x_opt @ self.Sigma @ x_opt), self.mu @ x_opt
        return None, None, None, None
    
    def risk_parity(self):
        """
        风险平价组合
        各资产对组合风险的贡献相等
        """
        x = cp.Variable(self.n_assets)
        
        # 使用对数障碍函数实现风险平价
        objective = cp.Minimize(cp.quad_form(x, self.Sigma) - 
                               2 * cp.sum(cp.log(x)) / self.n_assets)
        constraints = [cp.sum(x) == 1, x >= 0.01]
        
        prob = cp.Problem(objective, constraints)
        prob.solve()
        
        if prob.status == 'optimal':
            x_opt = x.value / np.sum(x.value)
            # 计算风险贡献
            portfolio_var = x_opt @ self.Sigma @ x_opt
            marginal_risk = self.Sigma @ x_opt
            risk_contrib = x_opt * marginal_risk / portfolio_var
            return x_opt, risk_contrib
        return None, None
    
    def robust_optimization(self, uncertainty_level=0.1):
        """
        鲁棒优化（考虑参数估计不确定性）
        """
        x = cp.Variable(self.n_assets)
        
        # 使用椭球不确定性集合
        # min max_{mu in U} -mu^T x + lambda * x^T Sigma x
        
        lambda_param = cp.Parameter(nonneg=True)
        lambda_param.value = 1.0
        
        # 保守估计: mu - uncertainty * ||x||_2
        objective = cp.Minimize(
            -self.mu @ x + uncertainty_level * cp.norm(x, 2) + 
            lambda_param * cp.quad_form(x, self.Sigma)
        )
        constraints = [cp.sum(x) == 1, x >= 0, x <= 0.3]
        
        prob = cp.Problem(objective, constraints)
        prob.solve()
        
        return x.value if prob.status == 'optimal' else None

# 运行优化
optimizer = PortfolioOptimizer(data)

print("=" * 60)
```

```python
    """
    蒙特卡洛回测：检验组合在样本外数据上的表现
    """
    # 获取最优组合权重
    x_opt, _, _, _ = optimizer.maximize_sharpe_ratio()
    
    # 等权重组合（基准）
    x_eq = np.ones(optimizer.n_assets) / optimizer.n_assets
    
    # 生成样本外情景（参数扰动）
    np.random.seed(42)
    outof_sample_returns = []
    
    for _ in range(n_scenarios):
        # 扰动预期收益和协方差
        mu_perturbed = optimizer.mu + np.random.randn(optimizer.n_assets) * 0.02
        Sigma_perturbed = optimizer.Sigma * (1 + np.random.randn() * 0.1)
        
        ret_opt = mu_perturbed @ x_opt
        risk_opt = np.sqrt(x_opt @ Sigma_perturbed @ x_opt)
```

### 3.1 梯度下降法

```python
class GradientDescent:
    """
    梯度下降算法实现
    包含多种步长选择策略
    """
    
    def __init__(self, f, grad_f, method='backtracking'):
        self.f = f
        self.grad_f = grad_f
        self.method = method
        self.history = {'x': [], 'f': [], 'grad_norm': []}
    
    def backtracking_line_search(self, x, d, alpha=1.0, rho=0.5, c=1e-4):
        """
        回溯线搜索
        """
        fx = self.f(x)
        grad_fx = self.grad_f(x)
        
        while self.f(x + alpha * d) > fx + c * alpha * np.dot(grad_fx, d):
            alpha *= rho
            if alpha < 1e-10:
                break
        return alpha
    
    def solve(self, x0, max_iter=1000, tol=1e-6):
        """
        求解优化问题
        """
        x = x0.copy()
        
        for k in range(max_iter):
            grad = self.grad_f(x)
            grad_norm = np.linalg.norm(grad)
            
            self.history['x'].append(x.copy())
            self.history['f'].append(self.f(x))
            self.history['grad_norm'].append(grad_norm)
            
            if grad_norm < tol:
                print(f"收敛于迭代 {k}")
                break
            
            # 下降方向
            d = -grad
            
            # 步长选择
            if self.method == 'fixed':
                alpha = 0.01
            elif self.method == 'backtracking':
                alpha = self.backtracking_line_search(x, d)
            elif self.method == 'exact':
                # 精确线搜索（二次函数）
                alpha = np.dot(grad, grad) / np.dot(grad, self.hess_f(x) @ grad) if hasattr(self, 'hess_f') else 0.01
            
            x = x + alpha * d
        
        return x, self.f(x), k
```

### 3.2 牛顿法

```python
class NewtonMethod:
    """
    牛顿法实现
    """
    
    def __init__(self, f, grad_f, hess_f):
        self.f = f
        self.grad_f = grad_f
        self.hess_f = hess_f
        self.history = {'x': [], 'f': [], 'grad_norm': []}
    
    def solve(self, x0, max_iter=100, tol=1e-6):
        """
        求解
        """
        x = x0.copy()
        
        for k in range(max_iter):
            grad = self.grad_f(x)
            hess = self.hess_f(x)
            grad_norm = np.linalg.norm(grad)
            
            self.history['x'].append(x.copy())
            self.history['f'].append(self.f(x))
            self.history['grad_norm'].append(grad_norm)
            
            if grad_norm < tol:
                print(f"牛顿法收敛于迭代 {k}")
                break
            
            # 牛顿方向: d = -H^{-1} g
            try:
                d = -np.linalg.solve(hess, grad)
            except np.linalg.LinAlgError:
                # Hessian奇异，使用梯度方向
                d = -grad
            
            # 线搜索
            alpha = 1.0
            fx = self.f(x)
            while self.f(x + alpha * d) > fx + 1e-4 * alpha * np.dot(grad, d):
                alpha *= 0.5
                if alpha < 1e-10:
                    break
            
            x = x + alpha * d
        
        return x, self.f(x), k

# 测试牛顿法
def test_newton_method():
    """
    测试牛顿法
    """
    Q = np.array([[4, 1], [1, 3]])
    b = np.array([1, 2])
    
    f = lambda x: 0.5 * x @ Q @ x - b @ x
    grad_f = lambda x: Q @ x - b
    hess_f = lambda x: Q
    
    newton = NewtonMethod(f, grad_f, hess_f)
    x0 = np.array([0.0, 0.0])
    x_opt, f_opt, iterations = newton.solve(x0)
    
    print(f"\n牛顿法结果:")
    print(f"  最优解: {x_opt}")
    print(f"  最优值: {f_opt:.6f}")
    print(f"  迭代次数: {iterations}")

test_newton_method()
```
