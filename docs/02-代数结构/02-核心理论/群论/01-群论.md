                strength_metrics['logical_coherence'],

    def add_historical_event(self, period: str, event: str, figure: str, contribution: str):
        """æ·»åŠ å†å²äº‹ä»¶"""

        """åˆ›å»ºå‘å±•å…³ç³»å›¾"""
        # æ·»åŠ æ—¶æœŸèŠ‚ç‚¹

            if period in self.timeline:

                    event = event_data['event']

        return self.development_graph
