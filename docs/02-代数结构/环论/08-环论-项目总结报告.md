    """构造代数簇"""
    variety = set()
    for point in ring.elements:
        if all(polynomial.evaluate(point) == 0 for polynomial in ideal):
            variety.add(point)
    return variety

def localization(ring, multiplicative_set):
    """环的局部化"""
    fractions = []
    for a in ring.elements:
        for s in multiplicative_set:
            fractions.append((a, s))
    return fractions
