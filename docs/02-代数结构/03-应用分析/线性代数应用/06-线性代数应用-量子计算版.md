## 一、量子比特与量子态





### 1.1 理论基础





**定义 1.1** (量子比特)


量子比特是量子计算的基本单位，可以表示为二维复向量空间中的单位向量：


$$|\psi\rangle = \alpha|0\rangle + \beta|1\rangle$$





其中 $\alpha, \beta \in \mathbb{C}$ 满足 $|\alpha|^2 + |\beta|^2 = 1$。





**定理 1.1** (量子态的线性叠加)


量子态满足线性叠加原理：如果 $|\psi_1\rangle$ 和 $|\psi_2\rangle$ 是量子态，那么它们的线性组合 $a|\psi_1\rangle + b|\psi_2\rangle$ 也是量子态（如果归一化）。





### 1.2 数学表示





```python


import numpy as np


from typing import List, Tuple, Optional


import matplotlib.pyplot as plt


from mpl_toolkits.mplot3d import Axes3D





class QuantumBit:


    """量子比特类 - 基于线性代数的实现"""





    def __init__(self, alpha: complex = 1.0, beta: complex = 0.0):


        """


        初始化量子比特





        Args:


            alpha: |0⟩态的振幅


            beta: |1⟩态的振幅


        """


        self.alpha = alpha


        self.beta = beta


        self._normalize()





    def _normalize(self):


        """归一化量子态"""


        norm = np.sqrt(abs(self.alpha)**2 + abs(self.beta)**2)


        if norm > 1e-10:


            self.alpha /= norm


            self.beta /= norm





    def get_state_vector(self) -> np.ndarray:


        """获取状态向量"""


        return np.array([self.alpha, self.beta])





    def get_density_matrix(self) -> np.ndarray:


        """获取密度矩阵"""


        state = self.get_state_vector()


        return np.outer(state, state.conj())





    def measure(self) -> int:


        """


        测量量子比特





        Returns:


            测量结果 (0 或 1)


        """


        prob_0 = abs(self.alpha)**2


        return 0 if np.random.random() < prob_0 else 1





    def __str__(self) -> str:


        return f"|ψ⟩ = {self.alpha:.3f}|0⟩ + {self.beta:.3f}|1⟩"





def quantum_bit_examples():


    """量子比特示例"""


    # 基态 |0⟩


    q0 = QuantumBit(1, 0)


    print(f"基态 |0⟩: {q0}")





    # 基态 |1⟩


    q1 = QuantumBit(0, 1)


    print(f"基态 |1⟩: {q1}")





    # 叠加态 (|0⟩ + |1⟩)/√2


    q_plus = QuantumBit(1/np.sqrt(2), 1/np.sqrt(2))


    print(f"叠加态 |+⟩: {q_plus}")





    # 叠加态 (|0⟩ - |1⟩)/√2


    q_minus = QuantumBit(1/np.sqrt(2), -1/np.sqrt(2))


    print(f"叠加态 |-⟩: {q_minus}")





    return q0, q1, q_plus, q_minus


```





### 1.3 多量子比特系统





**定义 1.2** (多量子比特系统)


n个量子比特的系统可以表示为 $2^n$ 维复向量空间中的单位向量：


$$|\psi\rangle = \sum_{i=0}^{2^n-1} \alpha_i |i\rangle$$





其中 $\sum_{i=0}^{2^n-1} |\alpha_i|^2 = 1$。





```python


class MultiQubitSystem:


    """多量子比特系统"""





    def __init__(self, n_qubits: int):


        self.n_qubits = n_qubits


        self.dimension = 2**n_qubits


        self.state_vector = np.zeros(self.dimension, dtype=complex)


        self.state_vector[0] = 1.0  # 初始化为 |0...0⟩





    def set_state(self, state_vector: np.ndarray):


        """设置状态向量"""


        if len(state_vector) != self.dimension:


            raise ValueError(f"状态向量维度必须为 {self.dimension}")





        # 归一化


        norm = np.sqrt(np.sum(np.abs(state_vector)**2))


        self.state_vector = state_vector / norm





    def get_density_matrix(self) -> np.ndarray:


        """获取密度矩阵"""


        return np.outer(self.state_vector, self.state_vector.conj())





    def measure_all(self) -> int:


        """


        测量所有量子比特





        Returns:


            测量结果的二进制表示


        """


        probabilities = np.abs(self.state_vector)**2


        result = np.random.choice(self.dimension, p=probabilities)


        return result


```





## 二、量子门与线性变换





### 2.1 单量子比特门





**定义 2.1** (量子门)


量子门是作用在量子比特上的酉变换，可以表示为酉矩阵。





**定理 2.1** (量子门的性质)


量子门必须是酉的：$U^\dagger U = UU^\dagger = I$。





```python


class QuantumGate:


    """量子门基类"""





    def __init__(self, matrix: np.ndarray):


        """


        初始化量子门





        Args:


            matrix: 门的矩阵表示


        """


        self.matrix = matrix


        self._verify_unitary()





    def _verify_unitary(self):


        """验证门的酉性"""


        product = self.matrix @ self.matrix.conj().T


        identity = np.eye(self.matrix.shape[0])


        if not np.allclose(product, identity, atol=1e-10):


            raise ValueError("量子门必须是酉的")





    def apply(self, qubit: QuantumBit) -> QuantumBit:


        """应用门到量子比特"""


        state = qubit.get_state_vector()


        new_state = self.matrix @ state


        return QuantumBit(new_state[0], new_state[1])





class PauliGates:


    """Pauli门"""





    @staticmethod


    def X() -> QuantumGate:


        """X门 (NOT门)"""


        matrix = np.array([[0, 1], [1, 0]], dtype=complex)


        return QuantumGate(matrix)





    @staticmethod


    def Y() -> QuantumGate:


        """Y门"""


        matrix = np.array([[0, -1j], [1j, 0]], dtype=complex)


        return QuantumGate(matrix)





    @staticmethod


    def Z() -> QuantumGate:


        """Z门"""


        matrix = np.array([[1, 0], [0, -1]], dtype=complex)


        return QuantumGate(matrix)





class HadamardGate:


    """Hadamard门"""





    @staticmethod


    def H() -> QuantumGate:


        """H门"""


        matrix = (1/np.sqrt(2)) * np.array([[1, 1], [1, -1]], dtype=complex)


        return QuantumGate(matrix)





class PhaseGates:


    """相位门"""





    @staticmethod


    def S() -> QuantumGate:


        """S门 (π/2相位门)"""


        matrix = np.array([[1, 0], [0, 1j]], dtype=complex)


        return QuantumGate(matrix)





    @staticmethod


    def T() -> QuantumGate:


        """T门 (π/4相位门)"""


        matrix = np.array([[1, 0], [0, np.exp(1j * np.pi / 4)]], dtype=complex)


        return QuantumGate(matrix)





    @staticmethod


    def R(phi: float) -> QuantumGate:


        """R门 (任意相位门)"""


        matrix = np.array([[1, 0], [0, np.exp(1j * phi)]], dtype=complex)


        return QuantumGate(matrix)





def quantum_gate_examples():


    """量子门示例"""


    # 创建量子比特


    q = QuantumBit(1, 0)  # |0⟩态


    print(f"初始态: {q}")





    # 应用X门


    x_gate = PauliGates.X()


    q_x = x_gate.apply(q)


    print(f"应用X门后: {q_x}")





    # 应用H门


    h_gate = HadamardGate.H()


    q_h = h_gate.apply(q)


    print(f"应用H门后: {q_h}")





    # 应用Z门


    z_gate = PauliGates.Z()


    q_z = z_gate.apply(q_h)


    print(f"应用Z门后: {q_z}")





    return q, q_x, q_h, q_z


```

## 三、量子算法





### 3.1 Deutsch算法





**定义 3.1** (Deutsch问题)


给定一个函数 $f: \{0,1\} \rightarrow \{0,1\}$，判断 $f$ 是常数函数还是平衡函数。





**算法 3.1** (Deutsch算法)





1. 准备两个量子比特：$|0\rangle|1\rangle$


2. 应用H门到两个量子比特：$\frac{1}{2}(|0\rangle + |1\rangle)(|0\rangle - |1\rangle)$


3. 应用量子预言机 $U_f$


4. 应用H门到第一个量子比特


5. 测量第一个量子比特





```python


def deutsch_algorithm(f: callable) -> bool:


    """


    Deutsch算法实现





    Args:


        f: 函数 f: {0,1} -> {0,1}





    Returns:


        True if f is constant, False if f is balanced


    """


    # 准备初始态 |0⟩|1⟩


    q1 = QuantumBit(1, 0)  # |0⟩


    q2 = QuantumBit(0, 1)  # |1⟩





    # 应用H门到两个量子比特


    h_gate = HadamardGate.H()


    q1 = h_gate.apply(q1)


    q2 = h_gate.apply(q2)





    # 应用量子预言机 U_f


    # 这里简化实现，实际需要构造U_f


    if f(0) == f(1):


        # 常数函数


        q1 = h_gate.apply(q1)


        return q1.measure() == 0


    else:


        # 平衡函数


        q1 = h_gate.apply(q1)


        return q1.measure() == 1





def deutsch_algorithm_example():


    """Deutsch算法示例"""


    # 常数函数


    def constant_0(x):


        return 0

        return 1





    # 平衡函数


    def balanced_identity(x):


        return x





    def balanced_not(x):


        return 1 - x





    print("Deutsch算法测试:")


    print(f"常数函数 f(x)=0: {deutsch_algorithm(constant_0)}")


    print(f"常数函数 f(x)=1: {deutsch_algorithm(constant_1)}")


    print(f"平衡函数 f(x)=x: {deutsch_algorithm(balanced_identity)}")


    print(f"平衡函数 f(x)=1-x: {deutsch_algorithm(balanced_not)}")


```





### 3.2 Grover算法





**定义 3.2** (搜索问题)

1. 初始化n个量子比特为均匀叠加态

## 四、量子纠缠与Bell态





### 4.1 Bell态





**定义 4.1** (Bell态)


Bell态是两量子比特系统的最大纠缠态：


$$|\Phi^+\rangle = \frac{1}{\sqrt{2}}(|00\rangle + |11\rangle)$$


$$|\Phi^-\rangle = \frac{1}{\sqrt{2}}(|00\rangle - |11\rangle)$$


$$|\Psi^+\rangle = \frac{1}{\sqrt{2}}(|01\rangle + |10\rangle)$$


$$|\Psi^-\rangle = \frac{1}{\sqrt{2}}(|01\rangle - |10\rangle)$$





```python


class BellStates:


    """Bell态"""





    @staticmethod


    def phi_plus() -> MultiQubitSystem:


        """|Φ⁺⟩ = (|00⟩ + |11⟩)/√2"""


        system = MultiQubitSystem(2)


        state = np.array([1, 0, 0, 1], dtype=complex) / np.sqrt(2)


        system.set_state(state)


        return system





    @staticmethod


    def phi_minus() -> MultiQubitSystem:


        """|Φ⁻⟩ = (|00⟩ - |11⟩)/√2"""


        system = MultiQubitSystem(2)


        state = np.array([1, 0, 0, -1], dtype=complex) / np.sqrt(2)


        system.set_state(state)


        return system





    @staticmethod


    def psi_plus() -> MultiQubitSystem:


        """|Ψ⁺⟩ = (|01⟩ + |10⟩)/√2"""


        system = MultiQubitSystem(2)


        state = np.array([0, 1, 1, 0], dtype=complex) / np.sqrt(2)


        system.set_state(state)


        return system





    @staticmethod


    def psi_minus() -> MultiQubitSystem:


        """|Ψ⁻⟩ = (|01⟩ - |10⟩)/√2"""


        system = MultiQubitSystem(2)


        state = np.array([0, 1, -1, 0], dtype=complex) / np.sqrt(2)


        system.set_state(state)


        return system





def bell_state_examples():


    """Bell态示例"""


    print("Bell态示例:")





    phi_plus = BellStates.phi_plus()


    print(f"|Φ⁺⟩: {phi_plus.state_vector}")





    phi_minus = BellStates.phi_minus()


    print(f"|Φ⁻⟩: {phi_minus.state_vector}")





    psi_plus = BellStates.psi_plus()


    print(f"|Ψ⁺⟩: {psi_plus.state_vector}")





    psi_minus = BellStates.psi_minus()


    print(f"|Ψ⁻⟩: {psi_minus.state_vector}")





    return phi_plus, phi_minus, psi_plus, psi_minus
```
