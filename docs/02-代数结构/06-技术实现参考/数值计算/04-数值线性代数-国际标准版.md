### 1.1 LUеҲҶи§Ј

**е®ҡд№ү 1.1** (LUеҲҶи§Ј)
и®ҫ $A$ жҳҜ $n \times n$ зҹ©йҳөпјҢеҰӮжһңеӯҳеңЁдёӢдёүи§’зҹ©йҳө $L$ е’ҢдёҠдёүи§’зҹ©йҳө $U$ дҪҝеҫ—пјҡ
$$A = LU$$
еҲҷз§°иҝҷжҳҜ $A$ зҡ„LUеҲҶи§ЈгҖӮ

**з®—жі•е®һзҺ°**:

```python
import numpy as np
from typing import Tuple, Optional

def lu_decomposition(A: np.ndarray, pivot: bool = True) -> Tuple[np.ndarray, np.ndarray, Optional[np.ndarray]]:
    """
    LUеҲҶи§Јз®—жі•

    Args:
        A: иҫ“е…Ҙзҹ©йҳө
        pivot: жҳҜеҗҰдҪҝз”ЁйҖүдё»е…ғ

    Returns:
        L: дёӢдёүи§’зҹ©йҳө
        U: дёҠдёүи§’зҹ©йҳө
        P: зҪ®жҚўзҹ©йҳөпјҲеҰӮжһңpivot=Trueпјү
    """
    n = A.shape[0]
    A_copy = A.copy().astype(float)

    if pivot:
        P = np.eye(n)
        L = np.eye(n)

        for k in range(n-1):
            # йҖүдё»е…ғ
            pivot_row = k + np.argmax(np.abs(A_copy[k:, k]))
            if pivot_row != k:
                A_copy[[k, pivot_row]] = A_copy[[pivot_row, k]]
                P[[k, pivot_row]] = P[[pivot_row, k]]
                if k > 0:
                    L[[k, pivot_row], :k] = L[[pivot_row, k], :k]

            # ж¶Ҳе…ғ
            for i in range(k+1, n):
                L[i, k] = A_copy[i, k] / A_copy[k, k]
                A_copy[i, k:] -= L[i, k] * A_copy[k, k:]

        U = np.triu(A_copy)
        return L, U, P
    else:
        L = np.eye(n)

        for k in range(n-1):
            for i in range(k+1, n):
                L[i, k] = A_copy[i, k] / A_copy[k, k]
                A_copy[i, k:] -= L[i, k] * A_copy[k, k:]

        U = np.triu(A_copy)
        return L, U, None

def solve_lu(L: np.ndarray, U: np.ndarray, b: np.ndarray, P: Optional[np.ndarray] = None) -> np.ndarray:
    """
    дҪҝз”ЁLUеҲҶи§ЈжұӮи§ЈзәҝжҖ§ж–№зЁӢз»„ Ax = b
    """
    n = L.shape[0]

    # еүҚеҗ‘д»Је…Ҙ Ly = Pb
    if P is not None:
        y = np.linalg.solve(L, P @ b)
    else:
        y = np.linalg.solve(L, b)

    # еҗҺеҗ‘д»Је…Ҙ Ux = y
    x = np.linalg.solve(U, y)

    return x

```

### 1.2 QRеҲҶи§Ј

**е®ҡд№ү 1.2** (QRеҲҶи§Ј)
и®ҫ $A$ жҳҜ $m \times n$ зҹ©йҳөпјҢеҰӮжһңеӯҳеңЁжӯЈдәӨзҹ©йҳө $Q$ е’ҢдёҠдёүи§’зҹ©йҳө $R$ дҪҝеҫ—пјҡ
$$A = QR$$
еҲҷз§°иҝҷжҳҜ $A$ зҡ„QRеҲҶи§ЈгҖӮ

**Gram-SchmidtжӯЈдәӨеҢ–**:

```python
def gram_schmidt_qr(A: np.ndarray) -> Tuple[np.ndarray, np.ndarray]:
    """
    дҪҝз”ЁGram-SchmidtжӯЈдәӨеҢ–зҡ„QRеҲҶи§Ј
    """
    m, n = A.shape
    Q = np.zeros((m, n))
    R = np.zeros((n, n))

    for j in range(n):
        v = A[:, j].copy()

        # еҮҸеҺ»еүҚйқўжүҖжңүеҗ‘йҮҸзҡ„жҠ•еҪұ
        for i in range(j):
            R[i, j] = np.dot(Q[:, i], A[:, j])
            v -= R[i, j] * Q[:, i]

        # еҪ’дёҖеҢ–
        R[j, j] = np.linalg.norm(v)
        if R[j, j] > 1e-12:
            Q[:, j] = v / R[j, j]
        else:
            Q[:, j] = v

    return Q, R

def householder_qr(A: np.ndarray) -> Tuple[np.ndarray, np.ndarray]:
    """
    дҪҝз”ЁHouseholderеҸҳжҚўзҡ„QRеҲҶи§ЈпјҲжӣҙзЁіе®ҡпјү
    """
    m, n = A.shape
    A_copy = A.copy().astype(float)
    Q = np.eye(m)

    for k in range(min(m-1, n)):
        # жһ„йҖ Householderеҗ‘йҮҸ
        x = A_copy[k:, k]
        e1 = np.zeros_like(x)
        e1[0] = 1

        u = x - np.linalg.norm(x) * e1
        if np.linalg.norm(u) > 1e-12:
            u = u / np.linalg.norm(u)
        else:
            u = np.zeros_like(u)

        # Householderзҹ©йҳө H = I - 2uu^T
        H = np.eye(m-k) - 2 * np.outer(u, u)

        # жӣҙж–°Aе’ҢQ
        A_copy[k:, k:] = H @ A_copy[k:, k:]
        Q[k:, :] = H @ Q[k:, :]

    R = np.triu(A_copy[:n, :])
    Q = Q.T

    return Q, R

```

### 1.3 еҘҮејӮеҖјеҲҶи§Ј (SVD)

**е®ҡд№ү 1.3** (еҘҮејӮеҖјеҲҶи§Ј)
и®ҫ $A$ жҳҜ $m \times n$ зҹ©йҳөпјҢеӯҳеңЁжӯЈдәӨзҹ©йҳө $U$гҖҒеҜ№и§’зҹ©йҳө $\Sigma$ е’ҢжӯЈдәӨзҹ©йҳө $V^T$ дҪҝеҫ—пјҡ
$$A = U\Sigma V^T$$

**з®—жі•е®һзҺ°**:

```python
def power_iteration(A: np.ndarray, max_iter: int = 100, tol: float = 1e-10) -> Tuple[float, np.ndarray]:
    """
    е№Ӯиҝӯд»Јжі•и®Ўз®—жңҖеӨ§зү№еҫҒеҖје’Ңзү№еҫҒеҗ‘йҮҸ
    """
    n = A.shape[0]
    x = np.random.randn(n)
    x = x / np.linalg.norm(x)

    for _ in range(max_iter):
        x_new = A @ x
        x_new = x_new / np.linalg.norm(x_new)

        if np.linalg.norm(x_new - x) < tol:
            break
        x = x_new

    eigenvalue = np.dot(x, A @ x)
    return eigenvalue, x

def svd_power_method(A: np.ndarray, k: int = None) -> Tuple[np.ndarray, np.ndarray, np.ndarray]:
    """
    дҪҝз”Ёе№Ӯиҝӯд»Јжі•зҡ„SVDеҲҶи§ЈпјҲз®ҖеҢ–зүҲпјү
    """
    if k is None:
        k = min(A.shape)

    m, n = A.shape
    U = np.zeros((m, k))
    S = np.zeros(k)
    Vt = np.zeros((k, n))

    # и®Ўз®—A^T Aзҡ„жңҖеӨ§зү№еҫҒеҖје’Ңзү№еҫҒеҗ‘йҮҸ
    ATA = A.T @ A
    for i in range(k):
        eigenvalue, eigenvector = power_iteration(ATA)
        S[i] = np.sqrt(eigenvalue)
        Vt[i, :] = eigenvector
        U[:, i] = (A @ eigenvector) / S[i]

    return U, S, Vt

def svd_truncated(A: np.ndarray, k: int) -> Tuple[np.ndarray, np.ndarray, np.ndarray]:
    """
    жҲӘж–ӯSVDеҲҶи§Ј
    """
    U, s, Vt = np.linalg.svd(A, full_matrices=False)
    return U[:, :k], s[:k], Vt[:k, :]

```

## дәҢгҖҒзү№еҫҒеҖји®Ўз®—

### 2.1 е№Ӯиҝӯд»Јжі•

**з®—жі•жҸҸиҝ°**:
еҜ№дәҺзҹ©йҳө $A$пјҢе№Ӯиҝӯд»Јжі•йҖҡиҝҮиҝӯд»Ј $x_{k+1} = \frac{Ax_k}{\|Ax_k\|}$ жқҘи®Ўз®—жңҖеӨ§зү№еҫҒеҖјгҖӮ

```python
def inverse_power_iteration(A: np.ndarray, sigma: float = 0, max_iter: int = 100) -> Tuple[complex, np.ndarray]:
    """
    еҸҚе№Ӯиҝӯд»Јжі•и®Ўз®—жңҖжҺҘиҝ‘sigmaзҡ„зү№еҫҒеҖје’Ңзү№еҫҒеҗ‘йҮҸ
    """
    n = A.shape[0]
    I = np.eye(n)
    B = A - sigma * I

    x = np.random.randn(n)
    x = x / np.linalg.norm(x)

    for _ in range(max_iter):
        # и§ЈзәҝжҖ§ж–№зЁӢз»„ Bx_new = x
        x_new = np.linalg.solve(B, x)
        x_new = x_new / np.linalg.norm(x_new)

        if np.linalg.norm(x_new - x) < 1e-10:
            break
        x = x_new

    eigenvalue = sigma + 1 / np.dot(x, np.linalg.solve(B, x))
    return eigenvalue, x

```

### 2.2 QRз®—жі•

**QRз®—жі•**жҳҜи®Ўз®—жүҖжңүзү№еҫҒеҖјзҡ„ж ҮеҮҶж–№жі•пјҡ

```python
def qr_algorithm(A: np.ndarray, max_iter: int = 100, tol: float = 1e-10) -> Tuple[np.ndarray, np.ndarray]:
    """
    QRз®—жі•и®Ўз®—жүҖжңүзү№еҫҒеҖј
    """
    n = A.shape[0]
    A_copy = A.copy().astype(float)
    Q_total = np.eye(n)

    for iteration in range(max_iter):
        # QRеҲҶи§Ј
        Q, R = np.linalg.qr(A_copy)

        # жӣҙж–°A
        A_copy = R @ Q
        Q_total = Q_total @ Q

        # жЈҖжҹҘж”¶ж•ӣжҖ§
        if np.max(np.abs(np.tril(A_copy, -1))) < tol:
            break

    eigenvalues = np.diag(A_copy)
    eigenvectors = Q_total

    return eigenvalues, eigenvectors

```

### 3.1 зӣҙжҺҘж–№жі•

**й«ҳж–Ҝж¶ҲеҺ»жі•**:

```python
def gaussian_elimination(A: np.ndarray, b: np.ndarray) -> np.ndarray:
    """
    й«ҳж–Ҝж¶ҲеҺ»жі•жұӮи§ЈзәҝжҖ§ж–№зЁӢз»„
    """
    n = A.shape[0]
    A_aug = np.column_stack([A, b])

    # еүҚеҗ‘ж¶Ҳе…ғ
    for k in range(n-1):
        for i in range(k+1, n):
            factor = A_aug[i, k] / A_aug[k, k]
            A_aug[i, k:] -= factor * A_aug[k, k:]

    # еҗҺеҗ‘д»Је…Ҙ
    x = np.zeros(n)
    for i in range(n-1, -1, -1):
        x[i] = (A_aug[i, -1] - np.dot(A_aug[i, i+1:n], x[i+1:])) / A_aug[i, i]

    return x

```

### 3.2 иҝӯд»Јж–№жі•

**йӣ…еҸҜжҜ”иҝӯд»Јжі•**:

```python
def jacobi_iteration(A: np.ndarray, b: np.ndarray, x0: np.ndarray = None,
                    max_iter: int = 1000, tol: float = 1e-10) -> np.ndarray:
    """
    йӣ…еҸҜжҜ”иҝӯд»Јжі•жұӮи§ЈзәҝжҖ§ж–№зЁӢз»„
    """
    n = A.shape[0]
    if x0 is None:
        x0 = np.zeros(n)

    x = x0.copy()

    for iteration in range(max_iter):
        x_new = np.zeros(n)

        for i in range(n):
            sum_ax = 0
            for j in range(n):
                if i != j:
                    sum_ax += A[i, j] * x[j]
            x_new[i] = (b[i] - sum_ax) / A[i, i]

        if np.linalg.norm(x_new - x) < tol:
            break
        x = x_new

    return x

```

**е…ұиҪӯжўҜеәҰжі•**:

```python
def conjugate_gradient(A: np.ndarray, b: np.ndarray, x0: np.ndarray = None,
                      max_iter: int = 1000, tol: float = 1e-10) -> np.ndarray:
    """
    е…ұиҪӯжўҜеәҰжі•жұӮи§ЈзәҝжҖ§ж–№зЁӢз»„пјҲйҖӮз”ЁдәҺеҜ№з§°жӯЈе®ҡзҹ©йҳөпјү
    """
    n = A.shape[0]
    if x0 is None:
        x0 = np.zeros(n)

    x = x0.copy()
    r = b - A @ x
    p = r.copy()

    for iteration in range(max_iter):
        Ap = A @ p
        alpha = np.dot(r, r) / np.dot(p, Ap)
        x = x + alpha * p
        r_new = r - alpha * Ap

        if np.linalg.norm(r_new) < tol:
            break

        beta = np.dot(r_new, r_new) / np.dot(r, r)
        p = r_new + beta * p
        r = r_new

    return x

```

## еӣӣгҖҒж•°еҖјзЁіе®ҡжҖ§еҲҶжһҗ

### 4.1 жқЎд»¶ж•°

**е®ҡд№ү 4.1** (зҹ©йҳөжқЎд»¶ж•°)
зҹ©йҳө $A$ зҡ„жқЎд»¶ж•°е®ҡд№үдёәпјҡ
$$\kappa(A) = \|A\| \cdot \|A^{-1}\|$$

```python
def condition_number(A: np.ndarray, norm_type: str = "2") -> float:
    """
    и®Ўз®—зҹ©йҳөжқЎд»¶ж•°
    """
    if norm_type == "2":
        return np.linalg.cond(A, 2)
    elif norm_type == "1":
        return np.linalg.cond(A, 1)
    elif norm_type == "inf":
        return np.linalg.cond(A, np.inf)
    else:
        raise ValueError("дёҚж”ҜжҢҒзҡ„иҢғж•°зұ»еһӢ")

def backward_error_analysis(A: np.ndarray, b: np.ndarray, x_computed: np.ndarray) -> float:
    """
    еҗҺеҗ‘иҜҜе·®еҲҶжһҗ
    """
    residual = b - A @ x_computed
    backward_error = np.linalg.norm(residual) / np.linalg.norm(b)
    return backward_error
```

```python
import multiprocessing as mp
from concurrent.futures import ProcessPoolExecutor

def parallel_matrix_multiply(A: np.ndarray, B: np.ndarray, n_jobs: int = -1) -> np.ndarray:
    """
    е№¶иЎҢзҹ©йҳөд№ҳжі•
    """
    if n_jobs == -1:
        n_jobs = mp.cpu_count()

    m, n = A.shape
    n, p = B.shape

    def multiply_block(args):
        i_start, i_end = args
        return A[i_start:i_end, :] @ B

    # еҲҶеқ—
    block_size = m // n_jobs
    blocks = [(i * block_size, min((i + 1) * block_size, m)) for i in range(n_jobs)]

    with ProcessPoolExecutor(max_workers=n_jobs) as executor:
        results = list(executor.map(multiply_block, blocks))

    return np.vstack(results)

```

```python
def memory_efficient_qr(A: np.ndarray) -> Tuple[np.ndarray, np.ndarray]:
    """
    еҶ…еӯҳй«ҳж•Ҳзҡ„QRеҲҶи§Ј
    """
    m, n = A.shape
    A_copy = A.copy()
    Q = np.eye(m)

    for k in range(min(m-1, n)):
        # е°ұең°и®Ўз®—HouseholderеҸҳжҚў
        x = A_copy[k:, k]
        norm_x = np.linalg.norm(x)

        if norm_x > 1e-12:
            if x[0] >= 0:
                x[0] += norm_x
            else:
                x[0] -= norm_x

            beta = 2.0 / np.dot(x, x)

            # жӣҙж–°A
            for j in range(k, n):
                v_dot_a = np.dot(x, A_copy[k:, j])
                A_copy[k:, j] -= beta * v_dot_a * x

            # жӣҙж–°Q
            for j in range(m):
                v_dot_q = np.dot(x, Q[k:, j])
                Q[k:, j] -= beta * v_dot_q * x

    R = np.triu(A_copy[:n, :])
    return Q.T, R

```

```python
def high_precision_svd(A: np.ndarray, precision: str = "double") -> Tuple[np.ndarray, np.ndarray, np.ndarray]:
```
