-- Axiom of Extensionality

  simp [PowerSet],
  exact mem_powerset_iff
end

-- Axiom of Infinity
axiom infinity : 
  ∃ x, ∅ ∈ x ∧ ∀ y, y ∈ x → y ∪ {y} ∈ x

-- Inductive set definition

  ∅ ∈ x ∧ ∀ y, y ∈ x → y ∪ {y} ∈ x
