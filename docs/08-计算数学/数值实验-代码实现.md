# 计算数学数值实验与代码实现

## 实验1：数值积分误差分析

### 问题

计算 $\int_0^1 e^x dx$，比较不同数值积分方法的精度

### Python实现

```python
import numpy as np
import matplotlib.pyplot as plt
from scipy import integrate

# 精确值
exact_value = np.e - 1
print(f"精确值: {exact_value:.10f}")

# 被积函数
def f(x):
    return np.exp(x)

# 不同方法
methods = {
    '左矩形法': lambda n: sum(f(np.linspace(0, 1, n, endpoint=False))) / n,
    '右矩形法': lambda n: sum(f(np.linspace(0, 1, n, endpoint=False)[1:])) / n,
    '梯形法': lambda n: integrate.trapezoid(f(np.linspace(0, 1, n)),
                                            np.linspace(0, 1, n)),
    'Simpson法': lambda n: integrate.simpson(f(np.linspace(0, 1, n)),
                                              np.linspace(0, 1, n))
}

# 误差分析
n_values = [2**i for i in range(2, 12)]
plt.figure(figsize=(12, 8))

for method_name, method_func in methods.items():
    errors = []
    for n in n_values:
        approx = method_func(n)
        error = abs(approx - exact_value)
        errors.append(error)

    plt.loglog(n_values, errors, 'o-', label=method_name, linewidth=2, markersize=8)

plt.xlabel('Number of subintervals (n)', fontsize=12)
plt.ylabel('Absolute Error', fontsize=12)
plt.title('Numerical Integration Error Analysis', fontsize=14)
plt.legend(fontsize=10)
plt.grid(True, which="both", ls="--", alpha=0.5)
plt.savefig('integration_error.png', dpi=150, bbox_inches='tight')
plt.show()

print("\n结论：Simpson法精度最高，收敛速度最快")
```

## 实验2：线性方程组求解

### 问题

求解 $Ax = b$，其中 $A$ 是大型稀疏矩阵

```python
import numpy as np
from scipy.sparse import diags
from scipy.sparse.linalg import cg, gmres, spsolve
import time

# 构造三对角矩阵（一维泊松方程离散化）
n = 1000
diagonals = [np.ones(n-1), -2*np.ones(n), np.ones(n-1)]
A = diags(diagonals, [-1, 0, 1], format='csr')
b = np.random.rand(n)

print(f"矩阵大小: {n} x {n}")
print(f"非零元素: {A.nnz}")
print(f"稀疏度: {A.nnz / (n*n) * 100:.4f}%")

# 直接求解
start = time.time()
x_direct = spsolve(A, b)
time_direct = time.time() - start
print(f"\n直接求解时间: {time_direct:.4f}秒")

# 共轭梯度法
start = time.time()
x_cg, info = cg(A, b, tol=1e-10)
time_cg = time.time() - start
print(f"CG方法时间: {time_cg:.4f}秒, 迭代次数: {info}")

# GMRES方法
start = time.time()
x_gmres, info = gmres(A, b, tol=1e-10)
time_gmres = time.time() - start
print(f"GMRES时间: {time_gmres:.4f}秒, 迭代次数: {info}")

# 误差比较
print(f"\n误差分析:")
print(f"直接求解残差: {np.linalg.norm(A @ x_direct - b):.2e}")
print(f"CG残差: {np.linalg.norm(A @ x_cg - b):.2e}")
print(f"GMRES残差: {np.linalg.norm(A @ x_gmres - b):.2e}")
```

## 实验3：ODE求解对比

### 问题

求解初值问题：$y' = -10y$，$y(0) = 1$，精确解 $y(t) = e^{-10t}$

```python
import numpy as np
from scipy.integrate import solve_ivp
import matplotlib.pyplot as plt

def ode(t, y):
    return -10 * y

# 精确解
def exact(t):
    return np.exp(-10 * t)

# 时间区间
t_span = [0, 2]
t_eval = np.linspace(0, 2, 200)
y0 = [1]

# 不同方法求解
methods = ['RK45', 'RK23', 'Radau', 'BDF']
plt.figure(figsize=(14, 10))

for i, method in enumerate(methods, 1):
    sol = solve_ivp(ode, t_span, y0, method=method, t_eval=t_eval, dense_output=True)

    plt.subplot(2, 2, i)
    plt.plot(sol.t, sol.y[0], 'b-', label=f'{method} (numerical)', linewidth=2)
    plt.plot(t_eval, exact(t_eval), 'r--', label='Exact', linewidth=1.5)
    plt.xlabel('t')
    plt.ylabel('y')
    plt.title(f'{method} Method')
    plt.legend()
    plt.grid(True)

    # 计算误差
    error = np.abs(sol.y[0] - exact(sol.t))
    max_error = np.max(error)
    print(f"{method}: Max Error = {max_error:.2e}, NFE = {sol.nfev}")

plt.tight_layout()
plt.savefig('ode_comparison.png', dpi=150)
plt.show()
```

## 实验4：FFT性能分析

```python
import numpy as np
import time
import matplotlib.pyplot as plt

# 测试不同大小的FFT
sizes = [2**i for i in range(5, 15)]
times = []

for n in sizes:
    x = np.random.rand(n)

    # 计时
    start = time.time()
    for _ in range(100):  # 重复100次取平均
        np.fft.fft(x)
    elapsed = (time.time() - start) / 100
    times.append(elapsed)

    print(f"n = {n:5d}: {elapsed*1000:.3f} ms")

# 理论复杂度 O(n log n)
theoretical = [t * (sizes[0] * np.log2(sizes[0])) / (n * np.log2(n))
               for n, t in zip(sizes, times)]

plt.figure(figsize=(10, 6))
plt.loglog(sizes, times, 'bo-', label='Measured', linewidth=2, markersize=8)
plt.loglog(sizes, theoretical, 'r--', label='O(n log n) Reference', linewidth=2)
plt.xlabel('Problem Size (n)', fontsize=12)
plt.ylabel('Time (seconds)', fontsize=12)
plt.title('FFT Performance Analysis', fontsize=14)
plt.legend(fontsize=11)
plt.grid(True, which="both", ls="--", alpha=0.5)
plt.savefig('fft_performance.png', dpi=150)
plt.show()
```

### 习题1

实现Newton插值法，并比较与Lagrange插值的数值稳定性

**解答要点**：

```python
def newton_interpolation(x, y, x_eval):
    n = len(x)
    # 差商表
    divided_diff = np.zeros((n, n))
    divided_diff[:, 0] = y

    for j in range(1, n):
        for i in range(n - j):
            divided_diff[i, j] = (divided_diff[i+1, j-1] - divided_diff[i, j-1]) / (x[i+j] - x[i])

    # 计算插值
    result = divided_diff[0, 0]
    product = 1
    for i in range(1, n):
        product *= (x_eval - x[i-1])
        result += divided_diff[0, i] * product

    return result
```
