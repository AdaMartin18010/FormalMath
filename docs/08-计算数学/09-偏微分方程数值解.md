## 2. 有限差分法

### 2.1 椭圆方程

#### 2D Poisson方程

$$-\nabla^2 u = f \text{ in } \Omega, \quad u = g \text{ on } \partial\Omega$$

#### 五点差分格式

在均匀网格$h = \Delta x = \Delta y$上：

$$\frac{-u_{i+1,j} - u_{i-1,j} - u_{i,j+1} - u_{i,j-1} + 4u_{i,j}}{h^2} = f_{i,j}$$

```python
import numpy as np
import matplotlib.pyplot as plt
from scipy.sparse import diags, kron, eye
from scipy.sparse.linalg import spsolve

def poisson_2d_fd(n, f_func, g_func):
    """
    用有限差分求解2D Poisson方程
    -∇²u = f in [0,1]×[0,1]
    u = g on boundary
    """
    h = 1.0 / (n + 1)
    x = np.linspace(0, 1, n + 2)
    y = np.linspace(0, 1, n + 2)
    X, Y = np.meshgrid(x, y)
    
    # 内部网格点
    N = n * n
    
    # 构造离散Laplace算子 (Kron积构造块三对角)
    # 一维离散Laplace
    T = diags([1, -2, 1], [-1, 0, 1], shape=(n, n))
    I = eye(n)
    
    # 2D: I ⊗ T + T ⊗ I
    A = (kron(I, T) + kron(T, I)) / h**2
    
    # 右端项
    F = f_func(X[1:-1, 1:-1], Y[1:-1, 1:-1]).flatten()
    
    # 处理边界条件
    # 左边
    F[:n] += g_func(0, y[1:-1]) / h**2
    # 右边
    F[-n:] += g_func(1, y[1:-1]) / h**2
    # 底边
    F[::n] += g_func(x[1:-1], 0) / h**2
    # 顶边
    F[n-1::n] += g_func(x[1:-1], 1) / h**2
    
    # 求解
    U_interior = spsolve(-A, F)
    
    # 重构完整解
    U = np.zeros((n + 2, n + 2))
    U[1:-1, 1:-1] = U_interior.reshape((n, n))
    U[0, :] = g_func(x, 0)
    U[-1, :] = g_func(x, 1)
    U[:, 0] = g_func(0, y)
    U[:, -1] = g_func(1, y)
    
    return X, Y, U

# 测试: u(x,y) = sin(πx)sin(πy), f = 2π²sin(πx)sin(πy)
f_func = lambda x, y: 2 * np.pi**2 * np.sin(np.pi * x) * np.sin(np.pi * y)
g_func = lambda x, y: np.sin(np.pi * x) * np.sin(np.pi * y)

n = 50
X, Y, U = poisson_2d_fd(n, f_func, g_func)
U_exact = g_func(X, Y)

print(f"最大误差: {np.max(np.abs(U - U_exact)):.6e}")

# 可视化
fig, axes = plt.subplots(1, 3, figsize=(15, 4))

im1 = axes[0].contourf(X, Y, U, levels=20, cmap='viridis')
axes[0].set_title('数值解')
plt.colorbar(im1, ax=axes[0])

im2 = axes[1].contourf(X, Y, U_exact, levels=20, cmap='viridis')
axes[1].set_title('精确解')
plt.colorbar(im2, ax=axes[1])

im3 = axes[2].contourf(X, Y, np.abs(U - U_exact), levels=20, cmap='hot')
axes[2].set_title('误差')
plt.colorbar(im3, ax=axes[2])

plt.tight_layout()
plt.show()
```

### 2.2 抛物方程

#### 热方程

$$u_t = u_{xx}, \quad u(x,0) = u_0(x), \quad u(0,t) = u(1,t) = 0$$

#### 显式格式（前向Euler）

$$\frac{u_i^{n+1} - u_i^n}{\Delta t} = \frac{u_{i+1}^n - 2u_i^n + u_{i-1}^n}{h^2}$$

$$u_i^{n+1} = u_i^n + \lambda(u_{i+1}^n - 2u_i^n + u_{i-1}^n), \quad \lambda = \frac{\Delta t}{h^2}$$

**稳定性条件**：$\lambda \leq 1/2$

```python
def heat_explicit(u0, T, dt, h):
    """显式格式求解热方程"""
    N = len(u0)
    lam = dt / h**2
    
    if lam > 0.5:
        print(f"警告: λ={lam:.2f} > 0.5，可能不稳定")
    
    n_steps = int(T / dt)
    u = u0.copy()
    
    for n in range(n_steps):
        u_new = u.copy()
        u_new[1:-1] = u[1:-1] + lam * (u[2:] - 2*u[1:-1] + u[:-2])
        u = u_new
    
    return u

# 测试

h = 1.0 / (N - 1)
x = np.linspace(0, 1, N)
u0 = np.sin(np.pi * x)

# 显式（稳定）
dt_stable = 0.4 * h**2
u_stable = heat_explicit(u0, T=0.1, dt=dt_stable, h=h)

# 显式（不稳定）
dt_unstable = 0.6 * h**2
u_unstable = heat_explicit(u0, T=0.1, dt=dt_unstable, h=h)

# 精确解
u_exact = np.exp(-np.pi**2 * 0.1) * u0

plt.figure(figsize=(10, 4))
plt.plot(x, u0, 'k--', label='初始条件')
plt.plot(x, u_stable, 'b-o', label=f'显式 λ=0.4')
plt.plot(x, u_unstable, 'r-x', label=f'显式 λ=0.6 (不稳定)')
plt.plot(x, u_exact, 'g-', label='精确解')
plt.legend()

```

#### 隐式格式（后向Euler）

$$\frac{u_i^{n+1} - u_i^n}{\Delta t} = \frac{u_{i+1}^{n+1} - 2u_i^{n+1} + u_{i-1}^{n+1}}{h^2}$$

**无条件稳定**，但每步需解线性系统。

```python
def heat_implicit(u0, T, dt, h):
    """隐式格式求解热方程"""
    from scipy.sparse import diags
    from scipy.sparse.linalg import spsolve
    
    N = len(u0)
    lam = dt / h**2
    n_steps = int(T / dt)
    
    # 构造三对角矩阵 (I + λA)
    A = diags([-lam, 1 + 2*lam, -lam], [-1, 0, 1], shape=(N-2, N-2))
    
    u = u0.copy()
    
    for n in range(n_steps):
        # 右端项
        b = u[1:-1].copy()
        # 边界条件
        b[0] += lam * u[0]
        b[-1] += lam * u[-1]
        
        u[1:-1] = spsolve(A, b)
    
    return u

# 比较显式和隐式
dt = 0.01
T = 0.5
u_exp = heat_explicit(u0, T, dt, h)
u_imp = heat_implicit(u0, T, dt, h)
u_exact = np.exp(-np.pi**2 * T) * u0

print(f"显式误差: {np.linalg.norm(u_exp - u_exact):.6e}")
print(f"隐式误差: {np.linalg.norm(u_imp - u_exact):.6e}")
```

#### Crank-Nicolson格式

$$\frac{u_i^{n+1} - u_i^n}{\Delta t} = \frac{1}{2}\left[\frac{u_{i+1}^{n+1} - 2u_i^{n+1} + u_{i-1}^{n+1}}{h^2} + \frac{u_{i+1}^n - 2u_i^n + u_{i-1}^n}{h^2}\right]$$

**无条件稳定**，$O(\Delta t^2 + h^2)$精度。

```python
def heat_crank_nicolson(u0, T, dt, h):
    """Crank-Nicolson格式"""
    from scipy.sparse import diags
    from scipy.sparse.linalg import spsolve
    
    N = len(u0)
    lam = dt / (2 * h**2)
    n_steps = int(T / dt)
    
    # 左端: I + λA
    L = diags([-lam, 1 + 2*lam, -lam], [-1, 0, 1], shape=(N-2, N-2))
    # 右端: I - λA
    R = diags([lam, 1 - 2*lam, lam], [-1, 0, 1], shape=(N-2, N-2))
    
    u = u0.copy()
    
    for n in range(n_steps):
        b = R @ u[1:-1]
        b[0] += lam * (u[0] + u[0])
        b[-1] += lam * (u[-1] + u[-1])
        
        u[1:-1] = spsolve(L, b)
    
    return u
```

### 2.3 双曲方程

#### 波动方程

$$u_{tt} = c^2 u_{xx}$$

#### 显式蛙跳格式

$$\frac{u_i^{n+1} - 2u_i^n + u_i^{n-1}}{\Delta t^2} = c^2 \frac{u_{i+1}^n - 2u_i^n + u_{i-1}^n}{h^2}$$

**稳定性条件**：$c\Delta t / h \leq 1$ (CFL条件)

```python
def wave_equation(u0, v0, T, dt, h, c=1.0):
    """
    求解波动方程
    u0: 初始位移
    v0: 初始速度
    """
    N = len(u0)
    r = c * dt / h
    
    if r > 1:
        print(f"警告: CFL数={r:.2f} > 1，不稳定")
    
    n_steps = int(T / dt)
    
    # 存储两个时间层
    u_prev = u0.copy()  # u^{n-1}
    u_curr = u0 + dt * v0  # u^n (用Taylor展开)
    
    for n in range(1, n_steps):
        u_next = u_curr.copy()
        u_next[1:-1] = (2 * u_curr[1:-1] - u_prev[1:-1] + 
                       r**2 * (u_curr[2:] - 2*u_curr[1:-1] + u_curr[:-2]))
        
        u_prev = u_curr.copy()
        u_curr = u_next.copy()
    
    return u_curr

# 测试：弦振动
N = 100
h = 1.0 / (N - 1)
x = np.linspace(0, 1, N)
u0 = np.sin(np.pi * x)  # 初始波形
v0 = np.zeros(N)  # 零初始速度

# 动画帧
fig, axes = plt.subplots(2, 3, figsize=(15, 8))
axes = axes.flatten()

for i, T in enumerate([0, 0.25, 0.5, 0.75, 1.0, 1.25]):
    u = wave_equation(u0, v0, T, dt=0.001, h=h, c=1.0)
    u_exact = np.sin(np.pi * x) * np.cos(np.pi * T)
    
    axes[i].plot(x, u, 'b-', label='数值')
    axes[i].plot(x, u_exact, 'r--', label='精确')
    axes[i].set_title(f't = {T}')
    axes[i].set_ylim(-1.5, 1.5)
    if i == 0:
        axes[i].legend()

plt.tight_layout()
plt.show()
```

#### 对流方程与迎风格式

$$u_t + c u_x = 0$$

**迎风格式**（$c > 0$）：
$$\frac{u_i^{n+1} - u_i^n}{\Delta t} + c \frac{u_i^n - u_{i-1}^n}{h} = 0$$

```python
def upwind_scheme(u0, T, dt, h, c=1.0):
    """迎风格式求解对流方程"""
    N = len(u0)
    lam = c * dt / h
    n_steps = int(T / dt)
    
    u = u0.copy()
    
    for n in range(n_steps):
        u_new = u.copy()
        if c > 0:
            u_new[1:] = u[1:] - lam * (u[1:] - u[:-1])
        else:
            u_new[:-1] = u[:-1] - lam * (u[1:] - u[:-1])
        u = u_new
    
    return u

# 测试：方波传播
N = 200
h = 1.0 / N
x = np.linspace(0, 2, N)
u0 = np.where((x > 0.4) & (x < 0.6), 1.0, 0.0)

dt = 0.8 * h  # CFL = 0.8

plt.figure(figsize=(12, 4))
for i, T in enumerate([0, 0.3, 0.6, 0.9]):
    plt.subplot(1, 4, i + 1)
    u = upwind_scheme(u0, T, dt, h, c=1.0)
    plt.plot(x, u, 'b-')
    plt.plot(x, np.roll(u0, int(T/h)), 'r--', alpha=0.5)
    plt.title(f't = {T}')
    plt.ylim(-0.2, 1.2)

plt.tight_layout()
plt.show()
```

### 2.4 稳定性分析

#### von Neumann分析

设$u_j^n = \rho^n e^{ikjh}$，代入差分格式求增长因子$\rho$。

**例：显式热方程**

增长因子：$\rho = 1 + 2\lambda(\cos kh - 1) = 1 - 4\lambda\sin^2(kh/2)$

稳定性要求$|\rho| \leq 1$对所有$k$成立，即$\lambda \leq 1/2$。

## 3. 有限元法

### 3.1 变分形式

#### 弱形式

对于Poisson方程$-\nabla^2 u = f$，乘以测试函数$v$并积分：

$$\int_\Omega \nabla u \cdot \nabla v \, dx = \int_\Omega fv \, dx$$

**Sobolev空间**：$H_0^1(\Omega) = \{v \in L^2 : \nabla v \in L^2, v|_{\partial\Omega} = 0\}$

**变分问题**：找$u \in H_0^1(\Omega)$使得
$$a(u, v) = F(v), \quad \forall v \in H_0^1(\Omega)$$

其中$a(u,v) = \int_\Omega \nabla u \cdot \nabla v$，$F(v) = \int_\Omega fv$。

### 3.2 离散化

#### Galerkin方法

取有限维子空间$V_h \subset H_0^1$，找$u_h \in V_h$使得
$$a(u_h, v_h) = F(v_h), \quad \forall v_h \in V_h$$

#### 分片线性基函数（1D）

在节点$x_i$处，帽函数：
$$\phi_i(x) = \begin{cases} \frac{x - x_{i-1}}{h} & x_{i-1} \leq x \leq x_i \\ \frac{x_{i+1} - x}{h} & x_i \leq x \leq x_{i+1} \\ 0 & \text{otherwise} \end{cases}$$

```python
def fem_1d_poisson(n, f_func):
    """
    1D Poisson方程有限元求解
    -u'' = f, u(0) = u(1) = 0
    使用分片线性基函数
    """
    h = 1.0 / (n + 1)
    x = np.linspace(0, 1, n + 2)
    
    # 刚度矩阵（三对角）
    # a(φ_i, φ_j) = ∫ φ_i' φ_j' = tridiag(-1/h, 2/h, -1/h)
    K = diags([-1/h, 2/h, -1/h], [-1, 0, 1], shape=(n, n))
    
    # 载荷向量（用梯形法则近似）
    F = np.zeros(n)
    for i in range(n):
        xi = x[i+1]
        F[i] = h * f_func(xi)  # 简化处理
    
    # 求解
    U = spsolve(K, F)
    
    # 完整解（含边界）
    u_full = np.zeros(n + 2)
    u_full[1:-1] = U
    
    return x, u_full

# 测试: -u'' = π²sin(πx), u = sin(πx)
n = 50
f_func = lambda x: np.pi**2 * np.sin(np.pi * x)
x, u_fem = fem_1d_poisson(n, f_func)
u_exact = np.sin(np.pi * x)

plt.figure(figsize=(10, 4))
plt.plot(x, u_fem, 'b-o', label='FEM')
plt.plot(x, u_exact, 'r-', label='精确解')
plt.legend()
plt.title('1D Poisson方程有限元解')
plt.show()

print(f"L2误差: {np.sqrt(np.mean((u_fem - u_exact)**2)):.6e}")
```

### 3.3 误差估计

#### Céa引理

$$\|u - u_h\|_{H^1} \leq C \inf_{v_h \in V_h} \|u - v_h\|_{H^1}$$

对于分片线性元：
$$\|u - u_h\|_{H^1} \leq Ch\|u''\|_{L^2}, \quad \|u - u_h\|_{L^2} \leq Ch^2\|u''\|_{L^2}$$

```python
def convergence_study():
    """收敛性研究"""
    f_func = lambda x: np.pi**2 * np.sin(np.pi * x)
    u_exact_func = lambda x: np.sin(np.pi * x)
    
    n_values = [4, 8, 16, 32, 64, 128]
    h_values = []
    errors_l2 = []
    errors_h1 = []
    
    for n in n_values:
        h = 1.0 / (n + 1)
        x, u_fem = fem_1d_poisson(n, f_func)
        u_exact = u_exact_func(x)
        
        # L2误差
        err_l2 = np.sqrt(np.mean((u_fem - u_exact)**2))
        
        # H1半范数（导数误差）
        du_fem = np.gradient(u_fem, h)
        du_exact = np.pi * np.cos(np.pi * x)
        err_h1 = np.sqrt(np.mean((du_fem - du_exact)**2))
        
        h_values.append(h)
        errors_l2.append(err_l2)
        errors_h1.append(err_h1)
    
    plt.figure(figsize=(10, 5))
    plt.loglog(h_values, errors_l2, 'bo-', label='L2误差 O(h²)')
    plt.loglog(h_values, errors_h1, 'rs-', label='H1误差 O(h)')
    plt.loglog(h_values, np.array(h_values)**2, 'b--', alpha=0.5, label='O(h²)')
    plt.loglog(h_values, np.array(h_values), 'r--', alpha=0.5, label='O(h)')
    plt.xlabel('h')
    plt.ylabel('误差')
    plt.legend()
    plt.title('有限元收敛性')
    plt.grid(True)
    plt.show()
    
    # 计算收敛阶
    order_l2 = np.polyfit(np.log(h_values), np.log(errors_l2), 1)[0]
    order_h1 = np.polyfit(np.log(h_values), np.log(errors_h1), 1)[0]
    print(f"L2收敛阶: {order_l2:.2f}")
    print(f"H1收敛阶: {order_h1:.2f}")

convergence_study()
```

### 3.4 实现细节

#### 2D有限元（简要）

```python
from scipy.sparse import csr_matrix

def assemble_stiffness_2d(nodes, elements):
    """
    组装2D刚度矩阵（三角形单元）
    """
    N = len(nodes)
    K = csr_matrix((N, N))
    
    for elem in elements:
        # 三角形三个顶点
        x = nodes[elem, 0]
        y = nodes[elem, 1]
        
        # 计算面积
        area = 0.5 * abs((x[1]-x[0])*(y[2]-y[0]) - (x[2]-x[0])*(y[1]-y[0]))
        
        # 梯度矩阵
        B = np.array([[y[1]-y[2], y[2]-y[0], y[0]-y[1]],
                      [x[2]-x[1], x[0]-x[2], x[1]-x[0]]]) / (2*area)
        
        # 单元刚度矩阵
        K_elem = area * B.T @ B
        
        # 组装到全局矩阵
        for i in range(3):
            for j in range(3):
                K[elem[i], elem[j]] += K_elem[i, j]
    
    return K
```

## 4. 多重网格方法

### 4.1 基本思想

- **光滑**：高频误差快速衰减
- **粗网格修正**：低频误差在粗网格上变为高频
- **递归**：V-cycle或W-cycle

### 4.2 两网格算法

```python
def jacobi_smoother(A, b, x0, num_iterations, omega=0.8):
    """加权Jacobi光滑子"""
    x = x0.copy()
    D = A.diagonal()
    
    for _ in range(num_iterations):
        x = x + omega * (b - A @ x) / D
    
    return x

def restriction_matrix(n_fine):
    """构造限制矩阵（完全加权）"""
    n_coarse = (n_fine + 1) // 2 - 1
    R = csr_matrix((n_coarse, n_fine))
    
    for i in range(n_coarse):
        j = 2 * i + 1
        R[i, j-1] = 1/4
        R[i, j] = 1/2
        R[i, j+1] = 1/4
    
    return R

def two_grid_cycle(A_fine, b, x0, R, P, nu1=2, nu2=2):
    """两网格V-cycle"""
    n_fine = len(b)
    n_coarse = R.shape[0]
    
    # 前光滑
    x = jacobi_smoother(A_fine, b, x0, nu1)
    
    # 残量
    r_fine = b - A_fine @ x
    
    # 限制到粗网格
    r_coarse = R @ r_fine
    
    # 粗网格求解
    A_coarse = R @ A_fine @ P
    e_coarse = spsolve(A_coarse, r_coarse)
    
    # 延拓并修正
    x = x + P @ e_coarse
    
    # 后光滑
    x = jacobi_smoother(A_fine, b, x, nu2)
    
    return x

# 完整多重网格求解器
def multigrid_solver(n_levels, f_func, tol=1e-10, max_cycles=100):
    """
    多重网格求解1D Poisson方程
    """
    # 最细网格
    n = 2**(n_levels + 1) - 1
    h = 1.0 / (n + 1)
    x = np.linspace(0, 1, n + 2)[1:-1]
    
    # 构造矩阵
    A = diags([-1/h**2, 2/h**2, -1/h**2], [-1, 0, 1], shape=(n, n))
    b = f_func(x)
    
    # 构造多重网格层次
    grids = []
    for level in range(n_levels):
        n_level = (n + 1) // (2**level) - 1
        R = restriction_matrix(n_level)
        P = 2 * R.T  # 延拓是限制的转置的两倍
        grids.append({'R': R, 'P': P})
    
    # V-cycle递归
    def v_cycle(A, b, x, level):
        if level == n_levels - 1:
            # 最粗层直接求解
            return spsolve(A, b)
        
        R = grids[level]['R']
        P = grids[level]['P']
        
        # 前光滑
        x = jacobi_smoother(A, b, x, 2)
        
        # 粗网格修正
        r = b - A @ x
        r_coarse = R @ r
        A_coarse = R @ A @ P
        e_coarse = v_cycle(A_coarse, r_coarse, np.zeros(len(r_coarse)), level + 1)
        x = x + P @ e_coarse
        
        # 后光滑
        x = jacobi_smoother(A, b, x, 2)
        
        return x
    
    # 迭代
    x = np.zeros(n)
    for cycle in range(max_cycles):
        x_new = v_cycle(A, b, x, 0)
        residual = np.linalg.norm(b - A @ x_new)
        
        if residual < tol:
            print(f"收敛于{cycle+1}个V-cycle")
            break
        
        x = x_new
    
    return np.concatenate([[0], x, [1]])

# 测试
f_func = lambda x: np.pi**2 * np.sin(np.pi * x)
u_mg = multigrid_solver(n_levels=6, f_func=f_func)

x_full = np.linspace(0, 1, len(u_mg))
u_exact = np.sin(np.pi * x_full)
print(f"多重网格误差: {np.max(np.abs(u_mg - u_exact)):.6e}")
```

```python
def lid_driven_cavity(n, Re=100, num_iterations=1000):
    """
    顶盖驱动方腔流（涡量-流函数形式）
    """
    h = 1.0 / n
    
    # 初始化
    psi = np.zeros((n+1, n+1))  # 流函数
    omega = np.zeros((n+1, n+1))  # 涡量
    
    for it in range(num_iterations):
        # 求解流函数方程: ∇²ψ = -ω
        psi_old = psi.copy()
        for i in range(1, n):
            for j in range(1, n):
                psi[i, j] = 0.25 * (psi[i+1, j] + psi[i-1, j] + 
                                    psi[i, j+1] + psi[i, j-1] + h**2 * omega[i, j])
        
        # 边界条件
        psi[0, :] = psi[n, :] = psi[:, 0] = psi[:, n] = 0
        
        # 更新涡量边界（用流函数计算）
        omega[1:-1, 0] = -2 * psi[1:-1, 1] / h**2  # 左壁
        omega[1:-1, n] = -2 * psi[1:-1, n-1] / h**2  # 右壁
        omega[0, 1:-1] = -2 * psi[1, 1:-1] / h**2  # 底壁
        omega[n, 1:-1] = -2 * (psi[n-1, 1:-1] + h) / h**2  # 顶壁（驱动）
        
        # 求解涡量输运方程（简化处理）
        omega_old = omega.copy()
        for i in range(1, n):
            for j in range(1, n):
                # 对流项（中心差分，简化）
                u = (psi[i, j+1] - psi[i, j-1]) / (2 * h)
                v = -(psi[i+1, j] - psi[i-1, j]) / (2 * h)
                
                conv_x = u * (omega[i, j+1] - omega[i, j-1]) / (2 * h) if u > 0 else 0
                conv_y = v * (omega[i+1, j] - omega[i-1, j]) / (2 * h) if v > 0 else 0
                
                # 扩散项
                diff = (omega[i+1, j] + omega[i-1, j] + omega[i, j+1] + omega[i, j-1] - 4*omega[i, j]) / h**2
                
                omega[i, j] = omega_old[i, j] + 0.01 * (diff / Re - conv_x - conv_y)
        
        if it % 100 == 0:
            err = np.max(np.abs(psi - psi_old))
            print(f"迭代 {it}, 误差: {err:.6e}")
    
    return psi, omega
```

```python
def heat_conduction_2d_transient(nx, ny, nt, alpha=1.0):
    """
    2D非稳态热传导
    u_t = α(u_xx + u_yy)
    """
    hx, hy = 1.0 / nx, 1.0 / ny
    dt = 0.25 * min(hx, hy)**2 / alpha  # 稳定性限制
    
    x = np.linspace(0, 1, nx + 1)
    y = np.linspace(0, 1, ny + 1)
    X, Y = np.meshgrid(x, y)
    
    # 初始条件（高斯分布）
    u = np.exp(-((X - 0.5)**2 + (Y - 0.5)**2) / 0.1)
    
    # 边界条件
    u[0, :] = u[-1, :] = u[:, 0] = u[:, -1] = 0
    
    # 存储用于动画
    snapshots = [u.copy()]
    
    for n in range(nt):
        u_new = u.copy()
        u_new[1:-1, 1:-1] = (u[1:-1, 1:-1] + 
                            alpha * dt / hx**2 * (u[2:, 1:-1] - 2*u[1:-1, 1:-1] + u[:-2, 1:-1]) +
                            alpha * dt / hy**2 * (u[1:-1, 2:] - 2*u[1:-1, 1:-1] + u[1:-1, :-2]))
        
        # 保持边界条件
        u_new[0, :] = u_new[-1, :] = u_new[:, 0] = u_new[:, -1] = 0
        
        u = u_new
        
        if n % (nt // 10) == 0:
            snapshots.append(u.copy())
    
    # 可视化
    fig, axes = plt.subplots(2, len(snapshots)//2, figsize=(15, 6))
    axes = axes.flatten()
    
    for i, snap in enumerate(snapshots):
        im = axes[i].contourf(X, Y, snap, levels=20, cmap='hot')
        axes[i].set_title(f't = {i * dt * (nt // 10):.3f}')
        plt.colorbar(im, ax=axes[i])
    
    plt.tight_layout()
    plt.show()
    
    return u

# 运行
```
