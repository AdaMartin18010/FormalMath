## 1. 概述

### 1.1 优化问题分类

| 类型 | 形式 | 方法 |
|-----|------|------|
| **无约束** | $\min_x f(x)$ | 梯度下降、牛顿法 |
| **等式约束** | $\min f(x)$ s.t. $h(x) = 0$ | 拉格朗日乘子 |
| **不等式约束** | $\min f(x)$ s.t. $g(x) \leq 0$ | KKT条件、内点法 |
| **凸优化** | $f$凸，可行域凸 | 多项式时间算法 |
| **非凸/全局** | 多局部极小 | 随机搜索、启发式 |

### 1.2 收敛性概念

- **局部收敛**：从足够接近最优解的点出发收敛
- **全局收敛**：从任意初始点收敛
- **Q-线性**：$\|x_{k+1} - x^*\| \leq C \|x_k - x^*\|$
- **Q-超线性**：$\|x_{k+1} - x^*\| / \|x_k - x^*\| \to 0$
- **Q-二次**：$\|x_{k+1} - x^*\| \leq C \|x_k - x^*\|^2$

## 2. 无约束优化

### 2.1 梯度下降法

#### 基本形式

$$x_{k+1} = x_k - \alpha_k \nabla f(x_k)$$

#### 步长选择

| 策略 | 公式 | 特点 |
|-----|------|------|
| 固定步长 | $\alpha_k = \alpha$ | 简单，需调参 |
| 精确线搜索 | $\alpha_k = \arg\min_\alpha f(x_k - \alpha \nabla f(x_k))$ | 最优但计算昂贵 |
| Armijo规则 | $f(x_k - \alpha \nabla f(x_k)) \leq f(x_k) - c_1 \alpha \|\nabla f(x_k)\|^2$ | 实用 |

#### 收敛性分析

**定理 2.1**：若$f$是$L$-光滑凸函数，则固定步长$\alpha < 2/L$的梯度下降满足：
$$f(x_k) - f^* \leq \frac{2\|x_0 - x^*\|^2}{\alpha k}$$

对于强凸函数（参数$\mu$）：
$$f(x_k) - f^* \leq \left(1 - \frac{\mu}{L}\right)^k (f(x_0) - f^*)$$

```python
import numpy as np
import matplotlib.pyplot as plt

def gradient_descent(f, grad_f, x0, alpha=0.01, tol=1e-6, max_iter=1000):
    """
    基本梯度下降
    
    参数:
        f: 目标函数
        grad_f: 梯度函数
        x0: 初始点
        alpha: 步长
    """
    x = x0.copy()
    history = [x.copy()]
    f_values = [f(x)]
    
    for k in range(max_iter):
        gradient = grad_f(x)
        x_new = x - alpha * gradient
        
        history.append(x_new.copy())
        f_values.append(f(x_new))
        
        if np.linalg.norm(x_new - x) < tol:
            print(f"收敛于{k+1}次迭代")
            break
        
        x = x_new
    
    return x, np.array(history), f_values

def gradient_descent_armijo(f, grad_f, x0, alpha_init=1.0, c=0.1, rho=0.5, 
                            tol=1e-6, max_iter=1000):
    """带Armijo线搜索的梯度下降"""
    x = x0.copy()
    history = [x.copy()]
    f_values = [f(x)]
    
    for k in range(max_iter):
        gradient = grad_f(x)
        grad_norm_sq = np.dot(gradient, gradient)
        
        # Armijo线搜索
        alpha = alpha_init
        while f(x - alpha * gradient) > f(x) - c * alpha * grad_norm_sq:
            alpha *= rho
        
        x_new = x - alpha * gradient
        
        history.append(x_new.copy())
        f_values.append(f(x_new))
        
        if np.linalg.norm(gradient) < tol:
            print(f"收敛于{k+1}次迭代，步长={alpha:.4f}")
            break
        
        x = x_new
    
    return x, np.array(history), f_values
```

```python
def momentum_gradient_descent(f, grad_f, x0, alpha=0.01, beta=0.9, 
                               max_iter=1000, tol=1e-6):
    """带动量的梯度下降"""
    x = x0.copy()
    v = np.zeros_like(x)
    
    for k in range(max_iter):
        gradient = grad_f(x)
        v = beta * v + gradient
        x_new = x - alpha * v
        
        if np.linalg.norm(x_new - x) < tol:
            return x_new, k + 1
        
        x = x_new
    
    return x, max_iter

def nesterov_accelerated_gradient(f, grad_f, x0, alpha=0.01, beta=0.9, 
                                   max_iter=1000, tol=1e-6):
    """Nesterov加速梯度下降"""
    x = x0.copy()
    y = x0.copy()
    
    for k in range(max_iter):
        x_new = y - alpha * grad_f(y)
        y = x_new + beta * (x_new - x)
        
        if np.linalg.norm(x_new - x) < tol:
            return x_new, k + 1
        
        x = x_new
    
    return x, max_iter

def adam_optimizer(f, grad_f, x0, alpha=0.001, beta1=0.9, beta2=0.999, 
                   eps=1e-8, max_iter=1000, tol=1e-6):
    """Adam优化器"""
    x = x0.copy()
    m = np.zeros_like(x)
    v = np.zeros_like(x)
    
    for k in range(1, max_iter + 1):
        gradient = grad_f(x)
        
        m = beta1 * m + (1 - beta1) * gradient
        v = beta2 * v + (1 - beta2) * gradient**2
        
        m_hat = m / (1 - beta1**k)
        v_hat = v / (1 - beta2**k)
        
        x_new = x - alpha * m_hat / (np.sqrt(v_hat) + eps)
        
        if np.linalg.norm(x_new - x) < tol:
            return x_new, k
        
        x = x_new
    
    return x, max_iter

# 比较各种方法
methods = {

    'Momentum': lambda: momentum_gradient_descent(rosenbrock, rosenbrock_grad, x0),
    'Nesterov': lambda: nesterov_accelerated_gradient(rosenbrock, rosenbrock_grad, x0),
```

### 2.2 牛顿法

#### 基本形式

$$x_{k+1} = x_k - [\nabla^2 f(x_k)]^{-1} \nabla f(x_k)$$

**局部Q-二次收敛**。

```python
def newton_method(f, grad_f, hess_f, x0, alpha=1.0, tol=1e-6, max_iter=100):
    """
    牛顿法
    
    参数:
        hess_f: Hessian矩阵函数
        alpha: 步长（可结合线搜索）
    """
    x = x0.copy()
    history = [x.copy()]
    
    for k in range(max_iter):
        gradient = grad_f(x)
        hessian = hess_f(x)
        
        # 求解线性系统
        try:
            d = np.linalg.solve(hessian, -gradient)
        except np.linalg.LinAlgError:
            # Hessian奇异，使用梯度方向
            d = -gradient
        
        x_new = x + alpha * d
        history.append(x_new.copy())
        
        if np.linalg.norm(d) < tol:
            print(f"牛顿法收敛于{k+1}次迭代")
            return x_new, np.array(history)
        
        x = x_new
    
    return x, np.array(history)

# Rosenbrock Hessian
rosenbrock_hess = lambda x: np.array([
    [2 + 1200*x[0]**2 - 400*x[1], -400*x[0]],
    [-400*x[0], 200]
])

x0 = np.array([-1.0, 2.0])
x_opt, history_newton = newton_method(rosenbrock, rosenbrock_grad, rosenbrock_hess, x0)
print(f"牛顿法结果: {x_opt}")
```

#### 收敛性定理

**定理 2.2**：若$f$三阶连续可微，$\nabla^2 f(x^*)$正定，则从足够接近$x^*$的点出发，牛顿法Q-二次收敛。

### 2.3 拟牛顿法

#### BFGS算法

避免计算Hessian，用低秩更新近似逆Hessian。

```python
def bfgs(f, grad_f, x0, tol=1e-6, max_iter=1000):
    """BFGS拟牛顿法"""
    n = len(x0)
    x = x0.copy()
    H = np.eye(n)  # 逆Hessian近似
    
    for k in range(max_iter):
        gradient = grad_f(x)
        
        if np.linalg.norm(gradient) < tol:
            print(f"BFGS收敛于{k+1}次迭代")
            return x
        
        # 搜索方向
        d = -H @ gradient
        
        # 线搜索
        alpha = 1.0
        c = 0.1
        rho = 0.5
        while f(x + alpha * d) > f(x) + c * alpha * np.dot(gradient, d):
            alpha *= rho
        
        s = alpha * d
        x_new = x + s
        
        y = grad_f(x_new) - gradient
        
        # BFGS更新
        rho_k = 1.0 / np.dot(y, s)
        if rho_k > 0:  # 保证正定性
            I = np.eye(n)
            H = (I - rho_k * np.outer(s, y)) @ H @ (I - rho_k * np.outer(y, s)) + rho_k * np.outer(s, s)
        
        x = x_new
    
    return x

x_opt_bfgs = bfgs(rosenbrock, rosenbrock_grad, x0)
print(f"BFGS结果: {x_opt_bfgs}")
```

### 2.4 共轭梯度法

用于大规模问题的无矩阵优化。

```python
def nonlinear_conjugate_gradient(f, grad_f, x0, method='polak-ribiere', 
                                  tol=1e-6, max_iter=1000):
    """非线性共轭梯度法"""
    x = x0.copy()
    gradient = grad_f(x)
    d = -gradient.copy()
    
    for k in range(max_iter):
        # 线搜索
        alpha = 1.0
        c = 0.1
        while f(x + alpha * d) > f(x) + c * alpha * np.dot(gradient, d):
            alpha *= 0.5
        
        x_new = x + alpha * d
        gradient_new = grad_f(x_new)
        
        if np.linalg.norm(gradient_new) < tol:
            print(f"CG收敛于{k+1}次迭代")
            return x_new
        
        # 计算beta
        if method == 'fletcher-reeves':
            beta = np.dot(gradient_new, gradient_new) / np.dot(gradient, gradient)
        elif method == 'polak-ribiere':
            beta = max(0, np.dot(gradient_new, gradient_new - gradient) / np.dot(gradient, gradient))
        else:
            beta = 0
        
        d = -gradient_new + beta * d
        gradient = gradient_new
        x = x_new
    
    return x
```

## 3. 约束优化

### 3.1 罚函数法

#### 外点罚函数

将约束问题转化为无约束：
$$P(x, \mu) = f(x) + \frac{\mu}{2} \sum_i [g_i(x)]_+^2 + \frac{\mu}{2} \sum_j h_j(x)^2$$

```python
def exterior_penalty_method(f, grad_f, g, h, x0, mu_init=1.0, gamma=10.0, 
                            tol=1e-6, max_iter=100):
    """外点罚函数法"""
    def penalty_objective(x, mu):
        penalty = 0
        if g is not None:
            for gi in g(x):
                penalty += max(0, gi)**2
        if h is not None:
            for hj in h(x):
                penalty += hj**2
        return f(x) + (mu / 2) * penalty
    
    x = x0.copy()
    mu = mu_init
    
    for k in range(max_iter):
        # 最小化罚函数
        result = minimize(lambda x: penalty_objective(x, mu), x, method='BFGS')
        x = result.x
        
        # 检查约束违反
        violation = 0
        if g is not None:
            violation += sum(max(0, gi)**2 for gi in g(x))
        if h is not None:
            violation += sum(hj**2 for hj in h(x))
        
        if violation < tol:
            print(f"罚函数法收敛于{k+1}次迭代")
            return x
        
        mu *= gamma
    
    return x

from scipy.optimize import minimize

# 测试: min x^2 + y^2 s.t. x + y = 1
f = lambda x: x[0]**2 + x[1]**2
g = lambda x: []  # 无不等式约束
h = lambda x: [x[0] + x[1] - 1]  # 等式约束

x0 = np.array([0.0, 0.0])
x_opt = exterior_penalty_method(f, None, g, h, x0)
print(f"约束优化解: {x_opt}")
print(f"约束验证: {x_opt[0] + x_opt[1]}")
```

### 3.2 内点法

用于不等式约束，从可行域内部逼近。

```python
def barrier_method(f, grad_f, hess_f, g, grad_g, x0, mu_init=1.0, gamma=0.1, 
                   tol=1e-6, max_iter=100):
    """障碍函数内点法（简化版）"""
    def barrier_objective(x, mu):
        if np.any(g(x) >= 0):
            return np.inf
        barrier = -mu * sum(np.log(-gi) for gi in g(x))
        return f(x) + barrier
    
    x = x0.copy()
    mu = mu_init
    
    for k in range(max_iter):
        # 牛顿步
        result = minimize(lambda x: barrier_objective(x, mu), x, 
                         method='Nelder-Mead', options={'maxiter': 1000})
        x = result.x
        
        if mu < tol:
            print(f"内点法收敛于{k+1}次迭代")
            return x
        
        mu *= gamma
    
    return x

# 测试: min x^2 + y^2 s.t. x >= 1, y >= 1
```

## 4. 全局优化

### 4.1 遗传算法

```python
class GeneticAlgorithm:
    """遗传算法实现"""
    
    def __init__(self, pop_size=100, n_genes=2, bounds=(-5, 5),
                 crossover_rate=0.8, mutation_rate=0.1, elitism=0.1):
        self.pop_size = pop_size
        self.n_genes = n_genes
        self.bounds = bounds
        self.crossover_rate = crossover_rate
        self.mutation_rate = mutation_rate
        self.elitism = elitism
        
    def initialize(self):
        """初始化种群"""
        return np.random.uniform(self.bounds[0], self.bounds[1], 
                                (self.pop_size, self.n_genes))
    
    def evaluate(self, population, fitness_func):
        """评估适应度"""
        return np.array([fitness_func(ind) for ind in population])
    
    def select(self, population, fitness):
        """轮盘赌选择"""
        fitness = fitness - fitness.min() + 1e-10  # 确保正值
        probs = fitness / fitness.sum()
        indices = np.random.choice(len(population), size=len(population), 
                                  p=probs, replace=True)
        return population[indices]
    
    def crossover(self, parent1, parent2):
        """单点交叉"""
        if np.random.random() < self.crossover_rate:
            point = np.random.randint(1, self.n_genes)
            child1 = np.concatenate([parent1[:point], parent2[point:]])
            child2 = np.concatenate([parent2[:point], parent1[point:]])
            return child1, child2
        return parent1.copy(), parent2.copy()
    
    def mutate(self, individual):
        """高斯变异"""
        for i in range(self.n_genes):
            if np.random.random() < self.mutation_rate:
                individual[i] += np.random.normal(0, 0.1)
                individual[i] = np.clip(individual[i], self.bounds[0], self.bounds[1])
        return individual
    
    def optimize(self, fitness_func, max_generations=100):
        """优化主循环"""
        population = self.initialize()
        best_fitness_history = []
        
        for gen in range(max_generations):
            fitness = self.evaluate(population, fitness_func)
            
            # 记录最优
            best_idx = np.argmax(fitness)
            best_fitness_history.append(fitness[best_idx])
            
            # 精英保留
            n_elite = int(self.elitism * self.pop_size)
            elite_indices = np.argsort(fitness)[-n_elite:]
            elites = population[elite_indices].copy()
            
            # 选择
            selected = self.select(population, fitness)
            
            # 交叉和变异
            new_population = []
            for i in range(0, self.pop_size - n_elite, 2):
                parent1, parent2 = selected[i], selected[i+1]
                child1, child2 = self.crossover(parent1, parent2)
                new_population.extend([self.mutate(child1), self.mutate(child2)])
            
            # 加入精英
            new_population = new_population[:self.pop_size - n_elite]
            population = np.vstack([new_population, elites])
            
            if gen % 10 == 0:
                print(f"Gen {gen}: Best fitness = {best_fitness_history[-1]:.6f}")
        
        # 返回最优解
        fitness = self.evaluate(population, fitness_func)
        best_idx = np.argmax(fitness)
        return population[best_idx], best_fitness_history

# 测试：最大化 Rastrigin函数（多峰）
def rastrigin(x):
    A = 10
    return -(A * len(x) + sum(xi**2 - A * np.cos(2 * np.pi * xi) for xi in x))

ga = GeneticAlgorithm(pop_size=200, n_genes=2, bounds=(-5.12, 5.12))

print(f"\n最优解: {best_solution}")
print(f"最优值: {rastrigin(best_solution):.6f}")

plt.xlabel('Generation')
plt.ylabel('Best Fitness')
plt.title('GA收敛曲线')
```

### 4.2 模拟退火

```python
def simulated_annealing(f, x0, bounds, T_init=100.0, T_min=1e-6, alpha=0.95, 
                        max_iter=10000):
    """
    模拟退火算法
    
    参数:
        f: 目标函数（要最小化）
        x0: 初始解
        bounds: [(min, max), ...] 每个维度的边界
        T_init: 初始温度
        alpha: 冷却系数
    """
    x = x0.copy()
    x_best = x.copy()
    f_current = f(x)
    f_best = f_current
    T = T_init
    
    history = [f_current]
    
    for i in range(max_iter):
        if T < T_min:
            break
        
        # 生成邻域解
        x_new = x + np.random.normal(0, T/10, len(x))
        x_new = np.clip(x_new, [b[0] for b in bounds], [b[1] for b in bounds])
        
        f_new = f(x_new)
        delta = f_new - f_current
        
        # Metropolis准则
        if delta < 0 or np.random.random() < np.exp(-delta / T):
            x = x_new
            f_current = f_new
            
            if f_new < f_best:
                x_best = x_new.copy()
                f_best = f_new
        
        history.append(f_current)
        T *= alpha
        
        if i % 1000 == 0:
            print(f"Iter {i}: T={T:.4f}, f={f_current:.6f}, best={f_best:.6f}")
    
    return x_best, f_best, history

# 测试：Rosenbrock（转换为最小化）
x0 = np.array([-1.0, 2.0])
bounds = [(-2, 2), (-1, 3)]
x_opt, f_opt, history = simulated_annealing(rosenbrock, x0, bounds)
print(f"\n模拟退火解: {x_opt}")
print(f"最优值: {f_opt:.10f}")
```

### 4.3 粒子群优化

```python
def particle_swarm_optimization(f, n_particles=30, n_dims=2, bounds=(-5, 5),
                                 w=0.7, c1=1.5, c2=1.5, max_iter=100):
    """
    粒子群优化 (PSO)
    
    参数:
        w: 惯性权重
        c1, c2: 认知和社会系数
    """
    # 初始化
    particles = np.random.uniform(bounds[0], bounds[1], (n_particles, n_dims))
    velocities = np.random.uniform(-1, 1, (n_particles, n_dims))
    
    # 个人最优
    pbest = particles.copy()
    pbest_fitness = np.array([f(p) for p in particles])
    
    # 全局最优
    gbest_idx = np.argmax(pbest_fitness)
    gbest = pbest[gbest_idx].copy()
    
    history = [pbest_fitness.max()]
    
    for i in range(max_iter):
        for j in range(n_particles):
            # 更新速度
            r1, r2 = np.random.rand(2)
            velocities[j] = (w * velocities[j] + 
                            c1 * r1 * (pbest[j] - particles[j]) +
                            c2 * r2 * (gbest - particles[j]))
            
            # 更新位置
            particles[j] += velocities[j]
            particles[j] = np.clip(particles[j], bounds[0], bounds[1])
            
            # 更新个人最优
            fitness = f(particles[j])
            if fitness > pbest_fitness[j]:
                pbest[j] = particles[j].copy()
                pbest_fitness[j] = fitness
                
                # 更新全局最优
                if fitness > pbest_fitness[gbest_idx]:
                    gbest = particles[j].copy()
                    gbest_idx = j
        
        history.append(pbest_fitness.max())
        
        if i % 10 == 0:
            print(f"Iter {i}: Best = {history[-1]:.6f}")
    
    return gbest, history

# 测试
best_pso, history_pso = particle_swarm_optimization(rastrigin, n_particles=50, 
                                                     max_iter=200)
print(f"\nPSO最优解: {best_pso}")
print(f"最优值: {rastrigin(best_pso):.6f}")
```

### 5.1 随机梯度下降

```python
def sgd(f, grad_f, X, y, w0, batch_size=32, alpha=0.01, n_epochs=100):
    """
    随机梯度下降
    
    参数:
        X: 数据矩阵 (n_samples, n_features)
        y: 标签 (n_samples,)
        grad_f: 梯度函数，接受 (X_batch, y_batch, w)
    """
    n_samples = len(y)
    w = w0.copy()
    
    for epoch in range(n_epochs):
        # 打乱数据
        indices = np.random.permutation(n_samples)
        X_shuffled = X[indices]
        y_shuffled = y[indices]
        
        for i in range(0, n_samples, batch_size):
            X_batch = X_shuffled[i:i+batch_size]
            y_batch = y_shuffled[i:i+batch_size]
            
            gradient = grad_f(X_batch, y_batch, w)
            w -= alpha * gradient
        
        if epoch % 10 == 0:
            loss = f(X, y, w)
            print(f"Epoch {epoch}: Loss = {loss:.6f}")
    
    return w

# 线性回归示例
def linear_regression_loss(X, y, w):
    """最小二乘损失"""
    return np.mean((X @ w - y)**2)

def linear_regression_grad(X, y, w):
    """梯度"""
    return 2 * X.T @ (X @ w - y) / len(y)

# 生成数据
np.random.seed(42)
X = np.random.randn(1000, 5)
true_w = np.array([1, 2, -1, 0.5, -2])
y = X @ true_w + np.random.randn(1000) * 0.1

w0 = np.zeros(5)
w_sgd = sgd(linear_regression_loss, linear_regression_grad, X, y, w0, 
            batch_size=32, alpha=0.01, n_epochs=100)

print(f"真实参数: {true_w}")
print(f"SGD估计: {w_sgd}")
```

```python
def simple_neural_network(X, y, hidden_size=10, max_iter=1000):
    """
    简单神经网络（单隐层）的梯度下降训练
    """
    n_samples, n_features = X.shape
    n_outputs = y.shape[1] if len(y.shape) > 1 else 1
    
    # 初始化权重
    W1 = np.random.randn(n_features, hidden_size) * 0.01
    b1 = np.zeros(hidden_size)
    W2 = np.random.randn(hidden_size, n_outputs) * 0.01
    b2 = np.zeros(n_outputs)
    
    def relu(x):
        return np.maximum(0, x)
    
    def relu_deriv(x):
        return (x > 0).astype(float)
    
    alpha = 0.1
    
    for i in range(max_iter):
        # 前向传播
        z1 = X @ W1 + b1
        a1 = relu(z1)
        z2 = a1 @ W2 + b2
        
        # 损失 (MSE)
        loss = np.mean((z2 - y)**2)
        
        # 反向传播
        dz2 = 2 * (z2 - y) / n_samples
        dW2 = a1.T @ dz2
        db2 = np.sum(dz2, axis=0)
        
        da1 = dz2 @ W2.T
        dz1 = da1 * relu_deriv(z1)
        dW1 = X.T @ dz1
        db1 = np.sum(dz1, axis=0)
        
        # 更新
        W1 -= alpha * dW1
        b1 -= alpha * db1
        W2 -= alpha * dW2
        b2 -= alpha * db2
        
        if i % 100 == 0:
            print(f"Iter {i}: Loss = {loss:.6f}")
    
    return W1, b1, W2, b2

# 测试XOR问题
```

### 6.2 组合优化：旅行商问题

```python
def traveling_salesman_ga(distances, n_cities, pop_size=100, max_generations=500):
    """
    用遗传算法求解TSP
    """
    class TSPGA(GeneticAlgorithm):
        def __init__(self, distances, **kwargs):
            self.distances = distances
            self.n_cities = len(distances)
            super().__init__(pop_size=pop_size, n_genes=self.n_cities, 
                           bounds=(0, self.n_cities-1), **kwargs)
        
        def initialize(self):
            """初始化：随机排列"""
            pop = []
            for _ in range(self.pop_size):
                perm = np.random.permutation(self.n_cities)
                pop.append(perm)
            return np.array(pop)
        
        def evaluate(self, population, fitness_func):
            """评估路径长度"""
            fitness = []
            for tour in population:
                length = sum(self.distances[tour[i], tour[(i+1)%self.n_cities]] 
                           for i in range(self.n_cities))
                fitness.append(-length)  # 最小化转最大化
            return np.array(fitness)
        
        def crossover(self, parent1, parent2):
            """顺序交叉 (OX)"""
            if np.random.random() >= self.crossover_rate:
                return parent1.copy(), parent2.copy()
            
            # 选择交叉区间
            start, end = sorted(np.random.choice(self.n_cities, 2, replace=False))
            
            def ox(p1, p2):
                child = -np.ones(self.n_cities, dtype=int)
                child[start:end] = p1[start:end]
                
                # 从p2填充剩余位置
                p2_filtered = [x for x in p2 if x not in child[start:end]]
                idx = 0
                for i in list(range(start)) + list(range(end, self.n_cities)):
                    child[i] = p2_filtered[idx]
                    idx += 1
                return child
            
            return ox(parent1, parent2), ox(parent2, parent1)
        
        def mutate(self, individual):
            """交换变异"""
            if np.random.random() < self.mutation_rate:
                i, j = np.random.choice(self.n_cities, 2, replace=False)
                individual[i], individual[j] = individual[j], individual[i]
            return individual
    
    ga = TSPGA(distances, pop_size=pop_size)
    best_tour, history = ga.optimize(lambda x: 0, max_generations=max_generations)
    
    return best_tour, history

# 测试（随机距离矩阵）
```
