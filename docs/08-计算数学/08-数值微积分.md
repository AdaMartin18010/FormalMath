## 1. 概述

### 1.1 问题定义

数值微积分研究如何用**离散计算**逼近**连续运算**：
- **数值微分**：$f'(x) \approx \frac{f(x+h) - f(x)}{h}$
- **数值积分**：$\int_a^b f(x) dx \approx \sum_{i=0}^n w_i f(x_i)$
- **ODE求解**：$y' = f(t, y)$，$y(t_0) = y_0$

### 1.2 核心挑战

| 问题 | 核心挑战 | 解决策略 |
|-----|---------|---------|
| 数值微分 | 舍入误差与截断误差的平衡 | 高阶差分、复步长 |
| 数值积分 | 光滑性与计算效率 | Gauss求积、自适应 |
| ODE求解 | 稳定性与精度 | 隐式方法、步长控制 |

## 2. 数值微分

### 2.1 有限差分公式

#### 前向差分
$$f'(x) = \frac{f(x+h) - f(x)}{h} - \frac{h}{2}f''(\xi)$$

**误差**：$O(h)$（一阶精度）

#### 后向差分
$$f'(x) = \frac{f(x) - f(x-h)}{h} + \frac{h}{2}f''(\xi)$$

**误差**：$O(h)$

#### 中心差分
$$f'(x) = \frac{f(x+h) - f(x-h)}{2h} - \frac{h^2}{6}f'''(\xi)$$

**误差**：$O(h^2)$（二阶精度）

```python
import numpy as np
import matplotlib.pyplot as plt

def forward_diff(f, x, h):
    """前向差分"""
    return (f(x + h) - f(x)) / h

def backward_diff(f, x, h):
    """后向差分"""
    return (f(x) - f(x - h)) / h

def central_diff(f, x, h):
    """中心差分"""
    return (f(x + h) - f(x - h)) / (2 * h)

# 数值实验：比较不同方法的精度
def test_derivative_accuracy():
    f = lambda x: np.sin(x)
    df_exact = lambda x: np.cos(x)
    
    x0 = 1.0
    h_values = np.logspace(-15, 0, 100)
    
    errors_forward = []
    errors_backward = []
    errors_central = []
    
    for h in h_values:
        err_f = abs(forward_diff(f, x0, h) - df_exact(x0))
        err_b = abs(backward_diff(f, x0, h) - df_exact(x0))
        err_c = abs(central_diff(f, x0, h) - df_exact(x0))
        
        errors_forward.append(err_f)
        errors_backward.append(err_b)
        errors_central.append(err_c)
    
    plt.figure(figsize=(10, 6))
    plt.loglog(h_values, errors_forward, 'b-', label='前向差分 O(h)')
    plt.loglog(h_values, errors_backward, 'g-', label='后向差分 O(h)')
    plt.loglog(h_values, errors_central, 'r-', label='中心差分 O(h²)')
    plt.loglog(h_values, h_values, 'b--', alpha=0.5, label='O(h)')
    plt.loglog(h_values, h_values**2, 'r--', alpha=0.5, label='O(h²)')
    plt.xlabel('步长 h')
    plt.ylabel('绝对误差')
    plt.legend()
    plt.title('数值微分精度比较')
    plt.grid(True, which="both", ls="--")
    plt.show()
    
    # 找到最优步长
    h_opt_forward = h_values[np.argmin(errors_forward)]
    h_opt_central = h_values[np.argmin(errors_central)]
    print(f"前向差分最优步长: {h_opt_forward:.2e}")
    print(f"中心差分最优步长: {h_opt_central:.2e}")

test_derivative_accuracy()
```

#### 高阶差分公式

**五点中心差分**（四阶精度）：
$$f'(x) = \frac{-f(x+2h) + 8f(x+h) - 8f(x-h) + f(x-2h)}{12h} + O(h^4)$$

**二阶导数**：
$$f''(x) = \frac{f(x+h) - 2f(x) + f(x-h)}{h^2} - \frac{h^2}{12}f^{(4)}(\xi)$$

```python
def five_point_diff(f, x, h):
    """五点中心差分（四阶精度）"""
    return (-f(x + 2*h) + 8*f(x + h) - 8*f(x - h) + f(x - 2*h)) / (12 * h)

def second_derivative(f, x, h):
    """二阶导数中心差分"""
    return (f(x + h) - 2*f(x) + f(x - h)) / h**2
```

### 2.2 Richardson外推

#### 算法思想

利用不同步长的结果外推消除低阶误差项。

**定理 2.1**（Richardson外推）：若$D(h) = f'(x) + c_1 h^p + c_2 h^{2p} + \cdots$，则
$$D_{\text{improved}} = \frac{2^p D(h/2) - D(h)}{2^p - 1} = f'(x) + O(h^{2p})$$

```python
def richardson_extrapolation(f, x, h, p):
    """Richardson外推"""
    D_h = central_diff(f, x, h)
    D_h2 = central_diff(f, x, h/2)
    return (2**p * D_h2 - D_h) / (2**p - 1)

# 对中心差分(p=2)进行外推
def richardson_demo():
    f = lambda x: np.sin(x)
    df_exact = lambda x: np.cos(x)
    
    x0 = 1.0
    h = 0.1
    
    # 普通中心差分
    df_central = central_diff(f, x0, h)
    
    # Richardson外推
    df_richardson = richardson_extrapolation(f, x0, h, 2)
    
    print(f"精确值: {df_exact(x0):.10f}")
    print(f"中心差分: {df_central:.10f}, 误差: {abs(df_central - df_exact(x0)):.2e}")
    print(f"Richardson: {df_richardson:.10f}, 误差: {abs(df_richardson - df_exact(x0)):.2e}")

richardson_demo()
```

### 2.3 自动微分

#### 复步长微分

利用复数避免相消误差：
$$f'(x) = \frac{\text{Im}(f(x + ih))}{h} + O(h^2)$$

```python
def complex_step_diff(f, x, h=1e-100):
    """复步长微分（避免相消误差）"""
    return np.imag(f(x + 1j * h)) / h

# 比较精度
def compare_diff_methods():
    f = lambda x: np.exp(x)
    df_exact = lambda x: np.exp(x)
    
    x0 = 1.0
    
    # 各种方法
    h = 1e-5
    err_forward = abs(forward_diff(f, x0, h) - df_exact(x0))
    err_central = abs(central_diff(f, x0, h) - df_exact(x0))
    err_complex = abs(complex_step_diff(f, x0) - df_exact(x0))
    
    print(f"前向差分误差: {err_forward:.2e}")
    print(f"中心差分误差: {err_central:.2e}")
    print(f"复步长误差: {err_complex:.2e}")

compare_diff_methods()
```

## 3. 数值积分

### 3.1 Newton-Cotes公式

#### 梯形法则

$$\int_a^b f(x) dx \approx \frac{b-a}{2}[f(a) + f(b)]$$

**误差**：$-\frac{(b-a)^3}{12}f''(\xi)$

#### Simpson法则

$$\int_a^b f(x) dx \approx \frac{b-a}{6}\left[f(a) + 4f\left(\frac{a+b}{2}\right) + f(b)\right]$$

**误差**：$-\frac{(b-a)^5}{2880}f^{(4)}(\xi)$

#### Cotes系数

```python
def trapezoidal_rule(f, a, b):
    """梯形法则"""
    return (b - a) / 2 * (f(a) + f(b))

def simpson_rule(f, a, b):
    """Simpson法则"""
    c = (a + b) / 2
    return (b - a) / 6 * (f(a) + 4*f(c) + f(b))

def simpson_3_8_rule(f, a, b):
    """Simpson 3/8法则"""
    h = (b - a) / 3
    return 3 * h / 8 * (f(a) + 3*f(a+h) + 3*f(a+2*h) + f(b))

def boole_rule(f, a, b):
    """Boole法则"""
    h = (b - a) / 4
    x0, x1, x2, x3, x4 = a, a+h, a+2*h, a+3*h, b
    return 2 * h / 45 * (7*f(x0) + 32*f(x1) + 12*f(x2) + 32*f(x3) + 7*f(x4))

# 测试
f = lambda x: np.sin(x)
a, b = 0, np.pi
exact = 2.0

print(f"精确值: {exact}")
print(f"梯形法则: {trapezoidal_rule(f, a, b):.6f}, 误差: {abs(trapezoidal_rule(f, a, b) - exact):.2e}")
print(f"Simpson法则: {simpson_rule(f, a, b):.6f}, 误差: {abs(simpson_rule(f, a, b) - exact):.2e}")
```

### 3.2 复合求积公式

#### 复合梯形法则

将区间分成$n$等份：
$$\int_a^b f(x) dx \approx \frac{h}{2}\left[f(a) + 2\sum_{i=1}^{n-1}f(x_i) + f(b)\right]$$

其中$h = (b-a)/n$，$x_i = a + ih$

**误差**：$O(h^2)$

#### 复合Simpson法则

$$\int_a^b f(x) dx \approx \frac{h}{3}\left[f(a) + 4\sum_{i=1,3,5}^{n-1}f(x_i) + 2\sum_{i=2,4,6}^{n-2}f(x_i) + f(b)\right]$$

**误差**：$O(h^4)$

```python
def composite_trapezoidal(f, a, b, n):
    """复合梯形法则"""
    h = (b - a) / n
    x = np.linspace(a, b, n + 1)
    y = f(x)
    return h * (0.5 * y[0] + np.sum(y[1:-1]) + 0.5 * y[-1])

def composite_simpson(f, a, b, n):
    """复合Simpson法则（n为偶数）"""
    if n % 2 == 1:
        n += 1
    h = (b - a) / n
    x = np.linspace(a, b, n + 1)
    y = f(x)
    
    return h / 3 * (y[0] + 4*np.sum(y[1:n:2]) + 2*np.sum(y[2:n-1:2]) + y[n])

# 收敛性测试
def convergence_test():
    f = lambda x: np.exp(x)
    a, b = 0, 1
    exact = np.exp(1) - 1
    
    n_values = [4, 8, 16, 32, 64, 128, 256]
    
    errors_trap = []
    errors_simp = []
    
    for n in n_values:
        err_t = abs(composite_trapezoidal(f, a, b, n) - exact)
        err_s = abs(composite_simpson(f, a, b, n) - exact)
        errors_trap.append(err_t)
        errors_simp.append(err_s)
    
    h_values = [(b - a) / n for n in n_values]
    
    plt.figure(figsize=(10, 6))
    plt.loglog(h_values, errors_trap, 'bo-', label='复合梯形 O(h²)')
    plt.loglog(h_values, errors_simp, 'ro-', label='复合Simpson O(h⁴)')
    plt.loglog(h_values, np.array(h_values)**2, 'b--', alpha=0.5, label='O(h²)')
    plt.loglog(h_values, np.array(h_values)**4, 'r--', alpha=0.5, label='O(h⁴)')
    plt.xlabel('步长 h')
    plt.ylabel('绝对误差')
    plt.legend()
    plt.title('复合求积公式收敛性')
    plt.grid(True, which="both", ls="--")
    plt.show()

convergence_test()
```

#### Romberg积分

Richardson外推+复合梯形 = Romberg积分：

```python
def romberg_integration(f, a, b, max_level=5):
    """Romberg积分"""
    R = np.zeros((max_level, max_level))
    
    # 第一列：复合梯形
    for i in range(max_level):
        n = 2**i
        R[i, 0] = composite_trapezoidal(f, a, b, n)
    
    # Richardson外推
    for j in range(1, max_level):
        for i in range(j, max_level):
            R[i, j] = (4**j * R[i, j-1] - R[i-1, j-1]) / (4**j - 1)
    
    return R[max_level-1, max_level-1], R

# 测试
f = lambda x: 1 / (1 + x**2)
a, b = 0, 1
exact = np.pi / 4

result, table = romberg_integration(f, a, b, max_level=5)
print(f"Romberg结果: {result:.10f}")
print(f"精确值: {exact:.10f}")
print(f"误差: {abs(result - exact):.2e}")
print("\nRomberg表:")
print(table)
```

### 3.3 Gauss求积

#### 基本思想

选择**最优节点**（非等距）和权重，使得对$2n+1$次多项式精确。

#### Gauss-Legendre求积

$$\int_{-1}^1 f(x) dx \approx \sum_{i=0}^n w_i f(x_i)$$

其中$x_i$是Legendre多项式$P_{n+1}(x)$的根。

```python
from numpy.polynomial.legendre import leggauss

def gauss_legendre_quad(f, a, b, n):
    """Gauss-Legendre求积"""
    x, w = leggauss(n)
    # 区间变换 [-1,1] -> [a,b]
    x_mapped = (b - a) / 2 * x + (a + b) / 2
    w_mapped = (b - a) / 2 * w
    return np.sum(w_mapped * f(x_mapped))

# 比较Gauss求积与Simpson
f = lambda x: x**8  # 8次多项式
a, b = -1, 1
exact = 2 / 9

for n in range(1, 6):
    gl_result = gauss_legendre_quad(f, a, b, n)
    simp_result = composite_simpson(f, a, b, n)
    print(f"n={n}: Gauss误差={abs(gl_result-exact):.2e}, Simpson误差={abs(simp_result-exact):.2e}")
```

```python
from numpy.polynomial.hermite import hermgauss
from numpy.polynomial.laguerre import laggauss

def gauss_hermite_quad(f, n):
    """Gauss-Hermite求积：∫_{-∞}^{∞} e^{-x²} f(x) dx"""
```

### 3.4 自适应积分

#### 算法思想

根据局部误差估计自动细分区间。

```python
def adaptive_simpson(f, a, b, tol=1e-6, max_level=20):
    """自适应Simpson积分"""
    def adaptive_simpson_recursive(f, a, b, eps, whole, fa, fb, fm, depth):
        c = (a + b) / 2
        d = (a + c) / 2
        e = (c + b) / 2
        fd = f(d)
        fe = f(e)
        
        left = (c - a) / 6 * (fa + 4*fd + fm)
        right = (b - c) / 6 * (fm + 4*fe + fb)
        delta = left + right - whole
        
        if depth <= 0 or abs(delta) <= 15 * eps:
            return left + right + delta / 15
        
        return (adaptive_simpson_recursive(f, a, c, eps/2, left, fa, fm, fd, depth-1) +
                adaptive_simpson_recursive(f, c, b, eps/2, right, fm, fb, fe, depth-1))
    
    c = (a + b) / 2
    fa, fb, fm = f(a), f(b), f(c)
    whole = (b - a) / 6 * (fa + 4*fm + fb)
    
    return adaptive_simpson_recursive(f, a, b, tol, whole, fa, fb, fm, max_level)

# 测试：在x=0附近有奇点的函数
f = lambda x: 1 / np.sqrt(abs(x))

# 普通Simpson会失败
# result_normal = composite_simpson(f, a, b, 1000)

result_adaptive = adaptive_simpson(f, a + 1e-10, b, tol=1e-6)
print(f"自适应积分结果: {result_adaptive:.6f}")
print(f"精确值: {4:.6f}")
```

## 4. 常微分方程数值解

### 4.1 Euler方法

#### 显式Euler

$$y_{n+1} = y_n + hf(t_n, y_n)$$

**局部截断误差**：$O(h^2)$  
**全局误差**：$O(h)$

#### 隐式Euler

$$y_{n+1} = y_n + hf(t_{n+1}, y_{n+1})$$

更稳定，但需要求解非线性方程。

#### 梯形法则（改进Euler）

$$y_{n+1} = y_n + \frac{h}{2}[f(t_n, y_n) + f(t_{n+1}, y_{n+1})]$$

```python
def explicit_euler(f, y0, t_span, h):
    """显式Euler方法"""
    t0, tf = t_span
    t = np.arange(t0, tf + h, h)
    n = len(t)
    y = np.zeros(n)
    y[0] = y0
    
    for i in range(n - 1):
        y[i+1] = y[i] + h * f(t[i], y[i])
    
    return t, y

def improved_euler(f, y0, t_span, h):
    """改进Euler方法（Heun方法）"""
    t0, tf = t_span
    t = np.arange(t0, tf + h, h)
    n = len(t)
    y = np.zeros(n)
    y[0] = y0
    
    for i in range(n - 1):
        k1 = f(t[i], y[i])
        k2 = f(t[i+1], y[i] + h * k1)
        y[i+1] = y[i] + h / 2 * (k1 + k2)
    
    return t, y
```

### 4.2 Runge-Kutta方法

#### 经典四阶Runge-Kutta (RK4)

$$\begin{aligned}
k_1 &= f(t_n, y_n) \\
k_2 &= f(t_n + \frac{h}{2}, y_n + \frac{h}{2}k_1) \\
k_3 &= f(t_n + \frac{h}{2}, y_n + \frac{h}{2}k_2) \\
k_4 &= f(t_n + h, y_n + hk_3) \\
y_{n+1} &= y_n + \frac{h}{6}(k_1 + 2k_2 + 2k_3 + k_4)
\end{aligned}$$

**局部截断误差**：$O(h^5)$  
**全局误差**：$O(h^4)$

```python
def rk4(f, y0, t_span, h):
    """经典四阶Runge-Kutta"""
    t0, tf = t_span
    t = np.arange(t0, tf + h, h)
    n = len(t)
    y = np.zeros((n, len(np.atleast_1d(y0))))
    y[0] = y0
    
    for i in range(n - 1):
        k1 = f(t[i], y[i])
        k2 = f(t[i] + h/2, y[i] + h/2 * k1)
        k3 = f(t[i] + h/2, y[i] + h/2 * k2)
        k4 = f(t[i] + h, y[i] + h * k3)
        y[i+1] = y[i] + h / 6 * (k1 + 2*k2 + 2*k3 + k4)
    
    return t, y

# 测试：Van der Pol振荡器
def van_der_pol(t, y, mu=1.0):
    """Van der Pol方程"""
    return np.array([y[1], mu * (1 - y[0]**2) * y[1] - y[0]])

# 数值求解
y0 = np.array([2.0, 0.0])
t_span = (0, 20)
h = 0.01

t, y = rk4(lambda t, y: van_der_pol(t, y, mu=1.0), y0, t_span, h)

plt.figure(figsize=(12, 4))
plt.subplot(1, 2, 1)
plt.plot(t, y[:, 0], 'b-', label='y₁(t)')
plt.plot(t, y[:, 1], 'r--', label='y₂(t)')
plt.xlabel('t')
plt.legend()
plt.title('Van der Pol振荡器')

plt.subplot(1, 2, 2)
```

#### Runge-Kutta-Fehlberg (RKF45)

变步长方法，同时使用4阶和5阶公式估计误差：

```python
def rkf45(f, y0, t_span, tol=1e-6, h_init=0.1):
    """Runge-Kutta-Fehlberg自适应方法"""
    # Butcher表
    a = np.array([0, 1/4, 3/8, 12/13, 1, 1/2])
    b = np.array([
        [0, 0, 0, 0, 0],
        [1/4, 0, 0, 0, 0],
        [3/32, 9/32, 0, 0, 0],
        [1932/2197, -7200/2197, 7296/2197, 0, 0],
        [439/216, -8, 3680/513, -845/4104, 0],
        [-8/27, 2, -3544/2565, 1859/4104, -11/40]
    ])
    c4 = np.array([25/216, 0, 1408/2565, 2197/4104, -1/5, 0])
    c5 = np.array([16/135, 0, 6656/12825, 28561/56430, -9/50, 2/55])
    
    t0, tf = t_span
    t = [t0]
    y = [y0]
    h = h_init
    
    while t[-1] < tf:
        y_current = y[-1]
        t_current = t[-1]
        
        k = []
        for i in range(6):
            ti = t_current + a[i] * h
            yi = y_current + h * sum(b[i][j] * k[j] for j in range(i)) if i > 0 else y_current
            k.append(f(ti, yi))
        
        y4 = y_current + h * sum(c4[i] * k[i] for i in range(6))
        y5 = y_current + h * sum(c5[i] * k[i] for i in range(6))
        
        error = np.linalg.norm(y5 - y4)
        
        if error < tol:
            t.append(min(t_current + h, tf))
            y.append(y5)
        
        # 调整步长
        h = h * min(2, max(0.1, 0.84 * (tol / error)**0.25))
        
        if t_current + h > tf:
            h = tf - t_current
    
    return np.array(t), np.array(y)
```

### 4.3 线性多步法

#### Adams-Bashforth方法（显式）

**二阶**（AB2）：
$$y_{n+2} = y_{n+1} + \frac{h}{2}(3f_{n+1} - f_n)$$

**四阶**（AB4）：
$$y_{n+4} = y_{n+3} + \frac{h}{24}(55f_{n+3} - 59f_{n+2} + 37f_{n+1} - 9f_n)$$

#### Adams-Moulton方法（隐式）

**四阶**（AM4）：
$$y_{n+3} = y_{n+2} + \frac{h}{24}(9f_{n+3} + 19f_{n+2} - 5f_{n+1} + f_n)$$

#### 预测-校正法

```python
def adams_bashforth_4(f, y0, t_span, h):
    """Adams-Bashforth四阶方法"""
    t0, tf = t_span
    t = np.arange(t0, tf + h, h)
    n = len(t)
    y = np.zeros(n)
    
    # 用RK4启动
    y[0] = y0
    for i in range(3):
        k1 = f(t[i], y[i])
        k2 = f(t[i] + h/2, y[i] + h/2 * k1)
        k3 = f(t[i] + h/2, y[i] + h/2 * k2)
        k4 = f(t[i] + h, y[i] + h * k3)
        y[i+1] = y[i] + h / 6 * (k1 + 2*k2 + 2*k3 + k4)
    
    # AB4主循环
    for i in range(3, n - 1):
        f_vals = [f(t[i-j], y[i-j]) for j in range(4)]
        y[i+1] = y[i] + h / 24 * (55*f_vals[0] - 59*f_vals[1] + 37*f_vals[2] - 9*f_vals[3])
    
    return t, y

def predictor_corrector(f, y0, t_span, h):
    """Adams预测-校正法 (PECE模式)"""
    t0, tf = t_span
    t = np.arange(t0, tf + h, h)
    n = len(t)
    y = np.zeros(n)
    
    # 启动
    y[0] = y0
    for i in range(3):
        k1 = f(t[i], y[i])
        k2 = f(t[i] + h/2, y[i] + h/2 * k1)
        k3 = f(t[i] + h/2, y[i] + h/2 * k2)
        k4 = f(t[i] + h, y[i] + h * k3)
        y[i+1] = y[i] + h / 6 * (k1 + 2*k2 + 2*k3 + k4)
    
    # 预测-校正
    for i in range(3, n - 1):
        # 预测 (AB4)
        f_vals = [f(t[i-j], y[i-j]) for j in range(4)]
        y_pred = y[i] + h / 24 * (55*f_vals[0] - 59*f_vals[1] + 37*f_vals[2] - 9*f_vals[3])
        
        # 校正 (AM4)
        f_pred = f(t[i+1], y_pred)
        y[i+1] = y[i] + h / 24 * (9*f_pred + 19*f_vals[0] - 5*f_vals[1] + f_vals[2])
    
    return t, y
```

### 4.4 稳定性分析

#### 绝对稳定性

应用于测试方程$y' = \lambda y$：

**显式Euler**：稳定当$|1 + h\lambda| < 1$，即$h\lambda \in (-2, 0)$

**隐式Euler**：稳定当$\text{Re}(h\lambda) < 0$

**RK4**：稳定区域更大，$\text{Re}(h\lambda) \in (-2.78, 0)$近似

#### 刚性方程

刚性方程的特征值差异很大，需要隐式方法：

```python
def backward_euler(f, y0, t_span, h, tol=1e-10, max_iter=100):
    """隐式Euler方法（用于刚性方程）"""
    from scipy.optimize import fsolve
    
    t0, tf = t_span
    t = np.arange(t0, tf + h, h)
    n = len(t)
    y = np.zeros(n)
    y[0] = y0
    
    for i in range(n - 1):
        # 求解 y_{i+1} = y_i + h*f(t_{i+1}, y_{i+1})
        def equation(y_next):
            return y_next - y[i] - h * f(t[i+1], y_next)
        
        y[i+1] = fsolve(equation, y[i])[0]
    
    return t, y
```

```python
def lotka_volterra(t, y, alpha=1.0, beta=0.1, delta=0.075, gamma=1.5):
```

### 5.3 金融：Black-Scholes期权定价

```python
def black_scholes_pde(S_max, T, r, sigma, K, N_S=100, N_t=1000):
    """用有限差分求解Black-Scholes PDE"""
    # 网格
    S = np.linspace(0, S_max, N_S)
    t = np.linspace(0, T, N_t)
    dt = t[1] - t[0]
    dS = S[1] - S[0]
    
    # 终值条件（看跌期权）
    V = np.maximum(K - S, 0)
    
    # 反向时间迭代
    for n in range(N_t - 1, 0, -1):
        V_new = V.copy()
        for i in range(1, N_S - 1):
            # 中心差分系数
            alpha = 0.5 * dt * (sigma**2 * S[i]**2 / dS**2 - r * S[i] / dS)
            beta = 1 - dt * (sigma**2 * S[i]**2 / dS**2 + r)
            gamma = 0.5 * dt * (sigma**2 * S[i]**2 / dS**2 + r * S[i] / dS)
            
            V_new[i] = alpha * V[i-1] + beta * V[i] + gamma * V[i+1]
        
        # 边界条件
        V_new[0] = K * np.exp(-r * (T - t[n-1]))  # S=0
        V_new[-1] = 0  # S=S_max
        
        V = V_new
    
    return S, V

# 计算
S, V = black_scholes_pde(S_max=200, T=1.0, r=0.05, sigma=0.2, K=100)

plt.figure(figsize=(10, 5))

plt.ylabel('期权价值 V')

plt.title('Black-Scholes看跌期权定价')
```
