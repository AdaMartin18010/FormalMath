### 1.2 每日运维检查清单

```bash
#!/bin/bash
# daily_check.sh

LOG_FILE="/var/log/ops/daily_check_$(date +%Y%m%d).log"

exec > >(tee $LOG_FILE)
exec 2>&1

echo "=== FormalMath 每日检查 ==="
echo "执行时间: $(date)"
echo "执行人: $(whoami)"
echo ""

# 1. 系统资源
echo "[1/8] 系统资源检查..."
echo "CPU负载: $(uptime | awk -F'load average:' '{print $2}')"
echo "内存使用:"
free -h | grep -E "Mem|Swap"
echo "磁盘使用:"
df -h | grep -E "Filesystem|/dev/"

# 2. 服务状态
echo -e "\n[2/8] 服务状态检查..."
for service in formalmath-cognitive formalmath-assessment formalmath-adaptive nginx; do
    status=$(systemctl is-active $service)
    if [ "$status" == "active" ]; then
        echo "  ✓ $service: 运行中"
    else
        echo "  ✗ $service: $status"
    fi
done

# 3. API健康检查
echo -e "\n[3/8] API健康检查..."
for port in 8001 8002 8003; do
    response=$(curl -s -o /dev/null -w "%{http_code}" --max-time 5 http://localhost:$port/health)
    if [ "$response" == "200" ]; then
        echo "  ✓ 端口 $port: 正常"
    else
        echo "  ✗ 端口 $port: 异常 (HTTP $response)"
    fi
done

# 4. 错误日志检查
echo -e "\n[4/8] 错误日志检查 (最近24小时)..."
ERROR_COUNT=$(find /var/log/formalmath -name "*.log" -mtime -1 -exec grep -l "ERROR\|FATAL\|Exception" {} \; | wc -l)
if [ $ERROR_COUNT -eq 0 ]; then
    echo "  ✓ 无错误日志"
else
    echo "  ⚠ 发现 $ERROR_COUNT 个日志文件有错误"
    find /var/log/formalmath -name "*.log" -mtime -1 -exec grep -l "ERROR\|FATAL\|Exception" {} \;
fi

# 5. 备份状态
echo -e "\n[5/8] 备份状态检查..."
TODAY=$(date +%Y%m%d)
if [ -d "/backup/database/$TODAY" ]; then
    BACKUP_COUNT=$(ls /backup/database/$TODAY/*.gz 2>/dev/null | wc -l)
    echo "  ✓ 今日数据库备份: $BACKUP_COUNT 个文件"
else
    echo "  ✗ 今日数据库备份缺失"
fi

# 6. 磁盘空间预警
echo -e "\n[6/8] 磁盘空间检查..."
DISK_USAGE=$(df / | tail -1 | awk '{print $5}' | sed 's/%//')
if [ $DISK_USAGE -gt 80 ]; then
    echo "  ⚠ 磁盘使用率过高: ${DISK_USAGE}%"
elif [ $DISK_USAGE -gt 90 ]; then
    echo "  ✗ 磁盘使用率危急: ${DISK_USAGE}%"
else
    echo "  ✓ 磁盘使用率正常: ${DISK_USAGE}%"
fi

# 7. 内存检查
echo -e "\n[7/8] 内存检查..."
MEM_USAGE=$(free | grep Mem | awk '{printf "%.0f", $3/$2 * 100.0}')
if [ $MEM_USAGE -gt 85 ]; then
    echo "  ⚠ 内存使用率过高: ${MEM_USAGE}%"
else
    echo "  ✓ 内存使用率正常: ${MEM_USAGE}%"
fi

# 8. 网络连接
echo -e "\n[8/8] 网络连接检查..."
ESTABLISHED=$(netstat -an | grep ESTABLISHED | wc -l)
TIME_WAIT=$(netstat -an | grep TIME_WAIT | wc -l)
echo "  ESTABLISHED: $ESTABLISHED"
echo "  TIME_WAIT: $TIME_WAIT"

echo -e "\n=== 检查完成 ==="
echo "日志已保存: $LOG_FILE"
```

### 1.3 服务启停操作

```bash
# 启动所有服务
start_all() {
    echo "启动 FormalMath 服务..."
    sudo systemctl start formalmath-cognitive
    sudo systemctl start formalmath-assessment
    sudo systemctl start formalmath-adaptive
    sudo systemctl start nginx
    echo "服务启动完成"
}

# 停止所有服务
stop_all() {
    echo "停止 FormalMath 服务..."
    sudo systemctl stop nginx
    sudo systemctl stop formalmath-adaptive
    sudo systemctl stop formalmath-assessment
    sudo systemctl stop formalmath-cognitive
    echo "服务停止完成"
}

# 重启服务
restart_all() {
    stop_all
    sleep 2
    start_all
}

# 平滑重启 (零停机)
graceful_restart() {
    # 使用systemd的reload或API的优雅重启
    sudo systemctl reload nginx
    
    # 对每个服务，启动新实例，切换流量，停止旧实例
    for service in formalmath-cognitive formalmath-assessment formalmath-adaptive; do
        echo "平滑重启 $service..."
        # 具体实现取决于负载均衡配置
    done
}
```

### 2.2 月度维护

```bash
#!/bin/bash
# monthly_maintenance.sh

echo "=== 月度维护任务 ==="

# 1. 系统更新
echo "[1/6] 系统更新..."
sudo apt update
sudo apt upgrade -y

# 2. 日志归档
echo "[2/6] 日志归档..."
ARCHIVE_DIR="/var/log/archive/$(date -d 'last month' +%Y%m)"
mkdir -p $ARCHIVE_DIR

# 压缩上月日志
find /var/log/formalmath -name "*.log" -mtime +30 -exec gzip {} \;
find /var/log/formalmath -name "*.gz" -mtime +30 -exec mv {} $ARCHIVE_DIR/ \;

# 上传到冷存储
aws s3 sync $ARCHIVE_DIR s3://formalmath-logs/archive/

# 3. 备份验证
echo "[3/6] 备份验证..."
/opt/formalmath/tools/backup/backup_drill.sh

# 4. 证书检查
echo "[4/6] SSL证书检查..."
openssl x509 -in /etc/nginx/ssl/formalmath.crt -noout -dates
EXPIRY=$(openssl x509 -in /etc/nginx/ssl/formalmath.crt -noout -enddate | cut -d= -f2)
EXPIRY_EPOCH=$(date -d "$EXPIRY" +%s)
NOW_EPOCH=$(date +%s)
DAYS_LEFT=$(( (EXPIRY_EPOCH - NOW_EPOCH) / 86400 ))

if [ $DAYS_LEFT -lt 30 ]; then
    echo "  ⚠ SSL证书将在 $DAYS_LEFT 天后过期"
else
    echo "  ✓ SSL证书还有 $DAYS_LEFT 天过期"
fi

# 5. 安全扫描
echo "[5/6] 安全扫描..."
sudo rkhunter --check --sk
sudo clamscan -r /opt/formalmath --infected

# 6. 容量评估
echo "[6/6] 容量评估..."
CURRENT_SIZE=$(du -sb /opt/formalmath | cut -f1)
GROWTH_RATE=0.1  # 假设月增长率10%
PREDICTED_6M=$(( CURRENT_SIZE * (1 + GROWTH_RATE * 6) ))

echo "当前数据量: $(numfmt --to=iec-i $CURRENT_SIZE)"
echo "预计6个月后: $(numfmt --to=iec-i $PREDICTED_6M)"

echo "月度维护完成"
```

## 三、性能优化

### 3.1 性能监控

```python
#!/usr/bin/env python3
# performance_monitor.py

import psutil
import sqlite3
import json
from datetime import datetime
import time

def collect_metrics():
    """收集性能指标"""
    metrics = {
        "timestamp": datetime.now().isoformat(),
        "cpu": {
            "percent": psutil.cpu_percent(interval=1),
            "count": psutil.cpu_count(),
            "freq": psutil.cpu_freq()._asdict() if psutil.cpu_freq() else None
        },
        "memory": psutil.virtual_memory()._asdict(),
        "disk": {
            "/": psutil.disk_usage("/")._asdict()
        },
        "network": psutil.net_io_counters()._asdict()
    }
    
    # 添加数据库指标
    try:
        conn = sqlite3.connect('/opt/formalmath/data/cognitive.db')
        cursor = conn.cursor()
        
        # 数据库大小
        cursor.execute("PRAGMA page_count;")
        page_count = cursor.fetchone()[0]
        cursor.execute("PRAGMA page_size;")
        page_size = cursor.fetchone()[0]
        metrics["database"] = {
            "size_mb": (page_count * page_size) / 1024 / 1024
        }
        conn.close()
    except Exception as e:
        metrics["database"] = {"error": str(e)}
    
    return metrics

def store_metrics(metrics):
    """存储指标到数据库"""
    conn = sqlite3.connect('/opt/formalmath/data/metrics.db')
    cursor = conn.cursor()
    
    cursor.execute('''
        CREATE TABLE IF NOT EXISTS system_metrics (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            timestamp TEXT,
            cpu_percent REAL,
            memory_percent REAL,
            disk_percent REAL,
            raw_json TEXT
        )
    ''')
    
    cursor.execute('''
        INSERT INTO system_metrics (timestamp, cpu_percent, memory_percent, disk_percent, raw_json)
        VALUES (?, ?, ?, ?, ?)
    ''', (
        metrics["timestamp"],
        metrics["cpu"]["percent"],
        metrics["memory"]["percent"],
        metrics["disk"]["/"]["percent"],
        json.dumps(metrics)
    ))
    
    conn.commit()
    conn.close()

if __name__ == '__main__':
    while True:
        metrics = collect_metrics()
        store_metrics(metrics)
        time.sleep(60)  # 每分钟收集一次
```

### 3.2 优化建议

| 问题 | 检查命令 | 优化措施 |
|------|----------|----------|
| **CPU过高** | `htop`, `ps aux --sort=-%cpu` | 优化算法、增加缓存、水平扩展 |
| **内存不足** | `free -h`, `ps aux --sort=-%mem` | 增加内存、优化缓存策略、内存泄漏排查 |
| **磁盘IO高** | `iostat -x 1` | 使用SSD、优化查询、增加缓存 |
| **网络延迟** | `ping`, `mtr` | CDN加速、连接池优化、压缩传输 |

## 四、容量管理

### 4.1 容量规划

```python
#!/usr/bin/env python3
# capacity_planning.py

import sqlite3
import json
from datetime import datetime, timedelta
import numpy as np

def predict_growth():
    """预测增长趋势"""
    conn = sqlite3.connect('/opt/formalmath/data/metrics.db')
    cursor = conn.cursor()
    
    # 获取历史数据
    cursor.execute('''
        SELECT timestamp, disk_percent 
        FROM system_metrics 
        WHERE timestamp > datetime('now', '-30 days')
        ORDER BY timestamp
    ''')
    
    data = cursor.fetchall()
    conn.close()
    
    if len(data) < 7:
        return {"error": "数据不足"}
    
    # 简单线性预测
    days = list(range(len(data)))
    disk_usage = [d[1] for d in data]
    
    # 计算增长率
    z = np.polyfit(days, disk_usage, 1)
    growth_rate = z[0]  # 每日增长率
    
    current = disk_usage[-1]
    
    # 预测未来
    predictions = {}
    for days_ahead in [30, 60, 90, 180, 365]:
        predicted = current + growth_rate * days_ahead
        predictions[f'{days_ahead}d'] = min(predicted, 100)
    
    return {
        "current_disk_usage": current,
        "daily_growth_rate": growth_rate,
        "predictions": predictions,
        "alert_threshold": 80,
        "critical_threshold": 90
    }

if __name__ == '__main__':
    result = predict_growth()
    print(json.dumps(result, indent=2))
```
