## 一、备份策略

### 1.1 备份范围

| 数据类型 | 位置 | 备份频率 | 保留周期 | 存储位置 |
|----------|------|----------|----------|----------|
| **文档内容** | docs/, 数学家理念体系/ | 实时(Git) | 永久 | GitHub + 异地 |
| **数据库** | data/*.db | 每小时 | 30天 | 本地 + 云存储 |
| **配置** | .env, *.conf | 每日 | 90天 | 版本控制 + 加密存储 |
| **Lean项目** | docs/09-形式化证明/ | 实时(Git) | 永久 | GitHub + 异地 |
| **用户上传** | uploads/ | 每日 | 90天 | 云对象存储 |
| **日志** | /var/log/formalmath/ | 每日 | 365天 | 本地 + 冷存储 |

## 二、备份操作

### 2.1 Git 仓库备份

```bash
#!/bin/bash
# backup_git.sh

REPO_URL="https://github.com/username/FormalMath.git"
BACKUP_DIR="/backup/git/$(date +%Y%m%d)"

mkdir -p $BACKUP_DIR

echo "备份 Git 仓库..."

# 方法1: 裸克隆 (推荐)
git clone --bare $REPO_URL $BACKUP_DIR/FormalMath.git

cd $BACKUP_DIR/FormalMath.git

# 获取所有分支
git fetch --all

# 获取所有标签
git fetch --tags

# 打包
cd ..
tar czf FormalMath.git.$(date +%Y%m%d).tar.gz FormalMath.git

# 上传到云存储
aws s3 cp FormalMath.git.$(date +%Y%m%d).tar.gz s3://formalmath-backup/git/

echo "Git备份完成: $BACKUP_DIR"
```

### 2.2 数据库备份

```bash
#!/bin/bash
# backup_database.sh

BACKUP_DIR="/backup/database/$(date +%Y%m%d)"
RETENTION_DAYS=30

mkdir -p $BACKUP_DIR

echo "开始数据库备份..."

# 备份所有数据库
for db in /opt/formalmath/data/*.db; do
    if [ -f "$db" ]; then
        db_name=$(basename $db)
        backup_file="$BACKUP_DIR/${db_name}.$(date +%H%M%S).gz"
        
        echo "备份: $db_name"
        
        # SQLite备份 (使用.dump保证一致性)
        sqlite3 $db ".backup '/tmp/temp_backup.db'"
        gzip -c /tmp/temp_backup.db > $backup_file
        
        # 计算校验和
        sha256sum $backup_file > ${backup_file}.sha256
        
        rm /tmp/temp_backup.db
    fi
done

# 上传到云存储
echo "上传到云存储..."
aws s3 sync $BACKUP_DIR s3://formalmath-backup/database/$(date +%Y%m%d)/

# 清理旧备份
echo "清理旧备份..."
find /backup/database -type d -mtime +$RETENTION_DAYS -exec rm -rf {} \; 2>/dev/null

echo "数据库备份完成"
```

### 2.3 配置文件备份

```bash
#!/bin/bash
# backup_config.sh

BACKUP_DIR="/backup/config/$(date +%Y%m%d)"

mkdir -p $BACKUP_DIR

echo "备份配置文件..."

# 系统配置
configs=(
    "/opt/formalmath/.env"
    "/opt/formalmath/docs/09-形式化证明/Lean4/lakefile.toml"
    "/etc/nginx/nginx.conf"
    "/etc/systemd/system/formalmath-*.service"
    "/etc/fail2ban/jail.local"
    "/etc/ssh/sshd_config"
)

for config in "${configs[@]}"; do
    if [ -f "$config" ]; then
        cp --parents $config $BACKUP_DIR/
        echo "已备份: $config"
    fi
done

# 加密敏感配置
echo "加密敏感配置..."
tar czf - $BACKUP_DIR | gpg --encrypt --recipient admin@formalmath.org \
    > $BACKUP_DIR.configs.tar.gz.gpg

# 上传到云存储
aws s3 cp $BACKUP_DIR.configs.tar.gz.gpg s3://formalmath-backup/config/

echo "配置备份完成"
```

### 2.4 完整系统备份

```bash
#!/bin/bash
# backup_full.sh

BACKUP_DIR="/backup/full/$(date +%Y%m%d)"
EXCLUDE_FILE="/tmp/backup_exclude.txt"

mkdir -p $BACKUP_DIR

# 排除列表
cat > $EXCLUDE_FILE << EOF
/proc
/sys
/dev
/run
/tmp
/var/tmp
/var/cache
/var/log
/backup
/home/*/.cache
*.log
.lake/build
__pycache__
*.pyc
node_modules
.git
EOF

echo "开始完整系统备份..."

# 创建系统快照
tar czpf $BACKUP_DIR/system_backup.tar.gz \
    --exclude-from=$EXCLUDE_FILE \
    /

# 备份引导扇区 (物理机)
if [ -d /sys/firmware/efi ]; then
    # UEFI系统
    dd if=/dev/sda1 of=$BACKUP_DIR/efi_partition.img bs=1M
else
    # BIOS系统
    dd if=/dev/sda of=$BACKUP_DIR/mbr.img bs=512 count=1
fi

# 生成备份清单
cat > $BACKUP_DIR/MANIFEST.txt << EOF
备份时间: $(date)
主机名: $(hostname)
系统版本: $(lsb_release -d 2>/dev/null || cat /etc/os-release | grep PRETTY_NAME)
内核版本: $(uname -r)
备份大小: $(du -sh $BACKUP_DIR | cut -f1)
EOF

# 上传到云存储
echo "上传到云存储..."
aws s3 sync $BACKUP_DIR s3://formalmath-backup/full/$(date +%Y%m%d)/

# 清理临时文件
rm $EXCLUDE_FILE

echo "完整备份完成: $BACKUP_DIR"
```

## 三、恢复操作

### 3.1 文档恢复

```bash
#!/bin/bash
# restore_docs.sh

BACKUP_DATE=${1:-$(date +%Y%m%d)}
RESTORE_DIR="/tmp/restore_$$"

echo "从备份恢复文档..."

# 从云存储下载
mkdir -p $RESTORE_DIR
aws s3 cp s3://formalmath-backup/git/FormalMath.git.${BACKUP_DATE}.tar.gz $RESTORE_DIR/

# 解压
cd $RESTORE_DIR
tar xzf FormalMath.git.${BACKUP_DATE}.tar.gz

# 恢复 (选择方式)
echo "恢复方式:"
echo "1) 完整替换 (删除现有，恢复备份)"
echo "2) 合并恢复 (保留现有，添加缺失)"
read -p "选择 [1/2]: " choice

case $choice in
    1)
        echo "完整替换..."
        rm -rf /opt/formalmath/docs
        rm -rf /opt/formalmath/数学家理念体系
        git clone $RESTORE_DIR/FormalMath.git /opt/formalmath_restored
        ;;
    2)
        echo "合并恢复..."
        cd /opt/formalmath
        git remote add backup $RESTORE_DIR/FormalMath.git
        git fetch backup
        git merge backup/main --allow-unrelated-histories
        ;;
    *)
        echo "无效选择"
        exit 1
        ;;
esac

# 清理
rm -rf $RESTORE_DIR

echo "文档恢复完成"
```

### 3.2 数据库恢复

```bash
#!/bin/bash
# restore_database.sh

DB_NAME=$1
BACKUP_DATE=${2:-$(date +%Y%m%d)}
BACKUP_TIME=${3:-"latest"}

echo "恢复数据库: $DB_NAME"

# 停止服务
echo "停止相关服务..."
sudo systemctl stop formalmath-*

# 查找备份文件
if [ "$BACKUP_TIME" == "latest" ]; then
    BACKUP_FILE=$(aws s3 ls s3://formalmath-backup/database/$BACKUP_DATE/ \
        | grep "$DB_NAME" | sort | tail -1 | awk '{print $4}')
else
    BACKUP_FILE="${DB_NAME}.${BACKUP_TIME}.gz"
fi

if [ -z "$BACKUP_FILE" ]; then
    echo "错误: 未找到备份文件"
    exit 1
fi

# 下载备份
DOWNLOAD_PATH="/tmp/${BACKUP_FILE}"
echo "下载备份: $BACKUP_FILE"
aws s3 cp s3://formalmath-backup/database/$BACKUP_DATE/$BACKUP_FILE $DOWNLOAD_PATH

# 验证校验和
echo "验证备份完整性..."
if [ -f "${DOWNLOAD_PATH}.sha256" ]; then
    aws s3 cp s3://formalmath-backup/database/$BACKUP_DATE/${BACKUP_FILE}.sha256 \
        ${DOWNLOAD_PATH}.sha256
    sha256sum -c ${DOWNLOAD_PATH}.sha256 || {
        echo "错误: 校验和验证失败"
        exit 1
    }
fi

# 备份当前数据库 (以防万一)
CURRENT_BACKUP="/opt/formalmath/data/${DB_NAME}.bak.$(date +%Y%m%d_%H%M%S)"
cp /opt/formalmath/data/$DB_NAME $CURRENT_BACKUP
echo "当前数据库已备份到: $CURRENT_BACKUP"

# 恢复数据库
echo "恢复数据库..."
gunzip -c $DOWNLOAD_PATH > /tmp/restore_temp.db

# 验证数据库完整性
if sqlite3 /tmp/restore_temp.db "PRAGMA integrity_check;" | grep -q "ok"; then
    mv /tmp/restore_temp.db /opt/formalmath/data/$DB_NAME
    chown formalmath:formalmath /opt/formalmath/data/$DB_NAME
    chmod 644 /opt/formalmath/data/$DB_NAME
    echo "数据库恢复成功"
else
    echo "错误: 备份数据库损坏"
    exit 1
fi

# 启动服务
echo "启动服务..."
sudo systemctl start formalmath-*

# 清理
rm -f $DOWNLOAD_PATH ${DOWNLOAD_PATH}.sha256

echo "数据库恢复完成"
```

### 3.3 配置恢复

```bash
#!/bin/bash
# restore_config.sh

BACKUP_DATE=${1:-$(date +%Y%m%d)}

echo "恢复配置文件..."

# 下载并解密
aws s3 cp s3://formalmath-backup/config/$BACKUP_DATE.configs.tar.gz.gpg /tmp/

gpg --decrypt /tmp/$BACKUP_DATE.configs.tar.gz.gpg > /tmp/configs.tar.gz
tar xzf /tmp/configs.tar.gz -C /tmp/

# 显示可恢复的配置
echo "可恢复的配置文件:"
find /tmp/backup/config/$BACKUP_DATE -type f

read -p "确认恢复这些配置文件? [y/N]: " confirm

if [ "$confirm" == "y" ] || [ "$confirm" == "Y" ]; then
    # 备份当前配置
    sudo tar czf /etc/backup_config.$(date +%Y%m%d).tar.gz /etc/nginx /etc/systemd/system/formalmath-*
    
    # 恢复配置
    sudo cp -r /tmp/backup/config/$BACKUP_DATE/* /
    
    # 重启服务
    sudo systemctl daemon-reload
    sudo systemctl restart nginx
    sudo systemctl restart formalmath-*
    
    echo "配置恢复完成"
else
    echo "已取消"
fi

# 清理
rm -rf /tmp/configs.tar.gz /tmp/backup
```

```bash
#!/bin/bash
# point_in_time_recovery.sh

# 恢复到指定时间点

    echo "用法: $0 'YYYY-MM-DD HH:MM:SS'"
    exit 1

# 1. 找到目标时间点之前的最后一个全量备份

    BACKUP_DATE=$(echo $line | awk '{print $1}')
    BACKUP_EPOCH=$(date -d $BACKUP_DATE +%s)
    
    if [ $BACKUP_EPOCH -le $TARGET_EPOCH ]; then
        echo "候选备份: $BACKUP_DATE"
    fi
done

# 2. 恢复全量备份
# 3. 应用增量备份/WAL到目标时间点
# 4. 验证数据一致性

echo "时间点恢复完成"
```

## 四、灾难恢复

### 4.1 灾难恢复计划

#### 场景1: 服务器完全损坏

```
恢复步骤:
1. 准备新服务器
   - 安装操作系统
   - 配置网络
   - 安装基础软件

2. 恢复系统
   - 从完整备份恢复
   - 恢复引导扇区
   - 重启系统

3. 恢复应用
   - 恢复Git仓库
   - 恢复数据库
   - 恢复配置

4. 验证
   - 服务健康检查
   - 数据一致性检查
   - 功能测试

预计恢复时间: 4-6小时
```

#### 场景2: 数据中心故障

```
恢复步骤:
1. 启动异地容灾站点
2. 切换DNS到备用站点
3. 从异地备份恢复数据
4. 验证服务

预计恢复时间: 1-2小时 (RTO)
```

#### 场景3: 数据损坏

```
恢复步骤:
1. 停止写入操作
2. 评估损坏范围
3. 从备份恢复
4. 应用日志恢复 (如可能)
5. 验证数据完整性

预计恢复时间: 30分钟-2小时
```

### 4.2 灾难恢复脚本

```bash
#!/bin/bash
# disaster_recovery.sh

DISASTER_TYPE=$1

case $DISASTER_TYPE in
    "server_failure")
        echo "执行服务器故障恢复..."
        
        # 1. 系统恢复
        echo "[1/5] 恢复系统..."
        # 从完整备份恢复
        
        # 2. Git仓库
        echo "[2/5] 恢复Git仓库..."
        ./restore_docs.sh
        
        # 3. 数据库
        echo "[3/5] 恢复数据库..."
        for db in cognitive assessment adaptive; do
            ./restore_database.sh "${db}.db"
        done
        
        # 4. 配置
        echo "[4/5] 恢复配置..."
        ./restore_config.sh
        
        # 5. 验证
        echo "[5/5] 验证恢复..."
        ./verify_recovery.sh
        ;;
        
    "data_corruption")
        echo "执行数据损坏恢复..."
        
        # 停止服务
        sudo systemctl stop formalmath-*
        
        # 确定损坏的数据库
        for db in /opt/formalmath/data/*.db; do
            echo "检查: $db"
            if ! sqlite3 $db "PRAGMA integrity_check;" | grep -q "ok"; then
                echo "发现损坏: $db"
                db_name=$(basename $db)
                ./restore_database.sh $db_name
            fi
        done
        
        # 重新启动
        sudo systemctl start formalmath-*
        ;;
        
    "ransomware")
        echo "执行勒索软件恢复..."
        
        # 1. 隔离系统
        sudo iptables -P INPUT DROP
        sudo iptables -P OUTPUT DROP
        sudo iptables -P FORWARD DROP
        
        # 2. 从干净备份恢复
        # 不恢复任何可能感染的文件
        
        # 3. 重建系统
        # ...
        
        # 4. 重新开放网络
        sudo iptables -P INPUT ACCEPT
        sudo iptables -P OUTPUT ACCEPT
        sudo iptables -P FORWARD ACCEPT
        ;;
        
    *)
        echo "未知的灾难类型: $DISASTER_TYPE"
        echo "可用类型: server_failure, data_corruption, ransomware"
        exit 1
        ;;
esac
```

### 4.3 恢复验证

```bash
#!/bin/bash
# verify_recovery.sh

echo "=== 恢复验证 ==="

ERRORS=0

# 1. 检查服务状态
echo "[1/6] 检查服务状态..."
for service in formalmath-cognitive formalmath-assessment formalmath-adaptive; do
    if systemctl is-active --quiet $service; then
        echo "  ✓ $service 运行中"
    else
        echo "  ✗ $service 未运行"
        ERRORS=$((ERRORS + 1))
    fi
done

# 2. 检查API响应
echo "[2/6] 检查API响应..."
for port in 8001 8002 8003; do
    if curl -sf http://localhost:$port/health > /dev/null; then
        echo "  ✓ 端口 $port 响应正常"
    else
        echo "  ✗ 端口 $port 无响应"
        ERRORS=$((ERRORS + 1))
    fi
done

# 3. 检查数据库完整性
echo "[3/6] 检查数据库完整性..."
for db in /opt/formalmath/data/*.db; do
    db_name=$(basename $db)
    if sqlite3 $db "PRAGMA integrity_check;" | grep -q "ok"; then
        echo "  ✓ $db_name 完整性正常"
    else
        echo "  ✗ $db_name 损坏"
        ERRORS=$((ERRORS + 1))
    fi
done

# 4. 检查Lean构建
echo "[4/6] 检查Lean构建..."
cd /opt/formalmath/docs/09-形式化证明/Lean4
if timeout 300 lake build > /dev/null 2>&1; then
    echo "  ✓ Lean构建成功"
else
    echo "  ✗ Lean构建失败"
    ERRORS=$((ERRORS + 1))
fi

# 5. 检查关键文件
echo "[5/6] 检查关键文件..."
CRITICAL_FILES=(
    "/opt/formalmath/README.md"
    "/opt/formalmath/docs/01-基础数学"
    "/opt/formalmath/数学家理念体系"
)

for file in "${CRITICAL_FILES[@]}"; do
    if [ -e "$file" ]; then
        echo "  ✓ $file 存在"
    else
        echo "  ✗ $file 缺失"
        ERRORS=$((ERRORS + 1))
    fi
done

# 6. 检查日志
echo "[6/6] 检查错误日志..."
RECENT_ERRORS=$(grep -i "error\|fatal\|exception" /var/log/formalmath/app/*.log 2>/dev/null | wc -l)
if [ $RECENT_ERRORS -eq 0 ]; then
    echo "  ✓ 无近期错误"
else
    echo "  ⚠ 发现 $RECENT_ERRORS 条错误日志"
fi

# 总结
echo ""
if [ $ERRORS -eq 0 ]; then
    echo "✅ 恢复验证通过"
    exit 0
else
    echo "❌ 恢复验证失败，发现 $ERRORS 个问题"
    exit 1
fi
```

```python
        "status": "ok" if git_ok else "failed",
        "path": git_backup
    })

    db_backup_dir = f"/backup/database/{datetime.now():%Y%m%d}"
    db_files = []
    if os.path.exists(db_backup_dir):
        db_files = [f for f in os.listdir(db_backup_dir) if f.endswith('.gz')]
        # 检查最新文件的时间
        if db_files:
            latest = max(db_files, key=lambda x: os.path.getmtime(os.path.join(db_backup_dir, x)))
            latest_time = datetime.fromtimestamp(os.path.getmtime(os.path.join(db_backup_dir, latest)))
            db_ok = (datetime.now() - latest_time) < timedelta(hours=2)
        else:
            db_ok = False
    else:
        db_ok = False
    
    status["checks"].append({
        "name": "数据库备份",
        "status": "ok" if db_ok else "failed",
        "files_count": len(db_files)
    })
    
    # 检查云存储同步
    # (需要实现具体的云存储API检查)
    
    # 汇总状态
    all_ok = all(c["status"] == "ok" for c in status["checks"])
    status["overall"] = "ok" if all_ok else "failed"
    
    return status

if __name__ == '__main__':
```

### 6.2 备份质量报告

```python
#!/usr/bin/env python3
# backup_report.py

import os
import json
from datetime import datetime, timedelta
import subprocess

def generate_backup_report():
    """生成备份质量报告"""
    report = {
        "generated_at": datetime.now().isoformat(),
        "period": f"{(datetime.now() - timedelta(days=7)).strftime('%Y-%m-%d')} to {datetime.now().strftime('%Y-%m-%d')}",
        "summary": {},
        "details": []
    }
    
    # 备份统计
    backup_types = ["git", "database", "config", "full", "lean"]
    
    for btype in backup_types:
        backup_dir = f"/backup/{btype}"
        if os.path.exists(backup_dir):
            # 计算备份大小
            result = subprocess.run(
                ["du", "-sb", backup_dir],
                capture_output=True, text=True
            )
            size = int(result.stdout.split()[0]) if result.returncode == 0 else 0
            
            # 计算备份数量
            count = len([d for d in os.listdir(backup_dir) 
                        if os.path.isdir(os.path.join(backup_dir, d))])
            
            report["details"].append({
                "type": btype,
                "total_size_bytes": size,
                "backup_count": count,
                "avg_size_mb": round(size / count / 1024 / 1024, 2) if count > 0 else 0
            })
    
    # 云存储同步状态
    result = subprocess.run(
        ["aws", "s3", "ls", "s3://formalmath-backup/", "--recursive"],
        capture_output=True, text=True
    )
    if result.returncode == 0:
        cloud_files = len(result.stdout.strip().split('\n'))
        report["summary"]["cloud_files_count"] = cloud_files
    
    # 生成Markdown报告
    markdown = f"""# 备份质量报告

**生成时间**: {report['generated_at']}
**统计周期**: {report['period']}

## 摘要

| 指标 | 数值 |
|------|------|
| 云存储文件数 | {report['summary'].get('cloud_files_count', 'N/A')} |

## 详细统计

| 备份类型 | 数量 | 总大小 | 平均大小 |
|----------|------|--------|----------|
"""
    
    for detail in report["details"]:
        size_gb = detail["total_size_bytes"] / 1024 / 1024 / 1024
        markdown += f"| {detail['type']} | {detail['backup_count']} | {size_gb:.2f} GB | {detail['avg_size_mb']:.2f} MB |\n"
    
    return markdown

if __name__ == '__main__':
    report = generate_backup_report()
    print(report)
    
    # 保存报告
    report_file = f"/var/log/backup/report_{datetime.now():%Y%m%d}.md"
    with open(report_file, 'w') as f:
        f.write(report)
    
    print(f"\n报告已保存: {report_file}")
```
