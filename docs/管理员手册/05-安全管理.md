```

│  ┌──────────────┐  ┌──────────────┐  ┌──────────────────────┐ │

│  └──────────────┘  └──────────────┘  └──────────────────────┘ │
└─────────────────────────────────────────────────────────────────┘
```

### 1.2 安全原则

1. **最小权限原则**: 用户和进程只拥有必要的最小权限
2. **纵深防御**: 多层安全控制，不依赖单一防护
3. **默认拒绝**: 未明确允许的访问默认拒绝
4. **安全审计**: 所有关键操作记录审计日志
5. **加密传输**: 所有数据传输使用TLS加密

### 2.3 API 认证配置

```python
# FastAPI 安全中间件
from fastapi import FastAPI, Depends, HTTPException, status
from fastapi.security import OAuth2PasswordBearer, HTTPBearer
from jose import JWTError, jwt
from passlib.context import CryptContext

app = FastAPI()

# 安全配置
SECRET_KEY = "your-secret-key-change-in-production"
ALGORITHM = "HS256"
ACCESS_TOKEN_EXPIRE_MINUTES = 30

pwd_context = CryptContext(schemes=["bcrypt"], deprecated="auto")
oauth2_scheme = OAuth2PasswordBearer(tokenUrl="token")
security = HTTPBearer()

async def verify_token(token: str = Depends(oauth2_scheme)):
    """验证JWT令牌"""
    credentials_exception = HTTPException(
        status_code=status.HTTP_401_UNAUTHORIZED,
        detail="Could not validate credentials",
        headers={"WWW-Authenticate": "Bearer"},
    )
    try:
        payload = jwt.decode(token, SECRET_KEY, algorithms=[ALGORITHM])
        username: str = payload.get("sub")
        if username is None:
            raise credentials_exception
    except JWTError:
        raise credentials_exception
    return username

@app.get("/admin/")
async def admin_endpoint(current_user: str = Depends(verify_token)):
    return {"message": f"Hello {current_user}"}
```

### 2.4 基于角色的访问控制 (RBAC)

```python
# RBAC 配置
ROLES = {
    "admin": ["*"],  # 所有权限
    "editor": ["read", "write", "update"],
    "viewer": ["read"],
    "auditor": ["read", "audit"]
}

PERMISSIONS = {
    "document": ["create", "read", "update", "delete"],
    "user": ["create", "read", "update", "delete"],
    "system": ["restart", "configure", "backup"],
    "audit": ["view_logs", "export_logs"]
}

def check_permission(user_role: str, resource: str, action: str) -> bool:
    """检查用户是否有权限执行操作"""
    if user_role not in ROLES:
        return False
    
    user_perms = ROLES[user_role]
    if "*" in user_perms:
        return True
    
    required_perm = f"{resource}:{action}"
    return required_perm in user_perms or action in user_perms

# 装饰器
from functools import wraps

def require_permission(resource: str, action: str):
    def decorator(func):
        @wraps(func)
        async def wrapper(*args, current_user: dict, **kwargs):
            if not check_permission(current_user["role"], resource, action):
                raise HTTPException(
                    status_code=status.HTTP_403_FORBIDDEN,
                    detail="Permission denied"
                )
            return await func(*args, **kwargs)
        return wrapper
    return decorator
```

### 3.2 数据加密

#### 静态数据加密

```python
# 数据库加密辅助函数
from cryptography.fernet import Fernet
from cryptography.hazmat.primitives import hashes
from cryptography.hazmat.primitives.kdf.pbkdf2 import PBKDF2HMAC
import base64
import os

class DataEncryption:
    def __init__(self, master_key: str):
        """初始化加密器
        
        Args:
            master_key: 主密钥，从环境变量获取
        """
        kdf = PBKDF2HMAC(
            algorithm=hashes.SHA256(),
            length=32,
            salt=os.environ.get('ENCRYPTION_SALT', 'default-salt').encode(),
            iterations=100000,
        )
        key = base64.urlsafe_b64encode(kdf.derive(master_key.encode()))
        self.cipher = Fernet(key)
    
    def encrypt(self, data: str) -> str:
        """加密字符串数据"""
        return self.cipher.encrypt(data.encode()).decode()
    
    def decrypt(self, encrypted: str) -> str:
        """解密数据"""
        return self.cipher.decrypt(encrypted.encode()).decode()

# 使用示例
encryption = DataEncryption(os.environ['MASTER_KEY'])
encrypted_user_data = encryption.encrypt(sensitive_data)
```

### 3.3 密钥管理

```bash
# 密钥管理脚本
#!/bin/bash
# key_management.sh

KEYS_DIR="/etc/formalmath/keys"
BACKUP_DIR="/backup/keys"

# 生成新密钥
generate_key() {
    local key_name=$1
    openssl rand -base64 32 > "$KEYS_DIR/$key_name.key"
    chmod 600 "$KEYS_DIR/$key_name.key"
    echo "生成密钥: $key_name"
}

# 轮换密钥
rotate_key() {
    local key_name=$1
    
    # 备份旧密钥
    cp "$KEYS_DIR/$key_name.key" "$BACKUP_DIR/$key_name.key.$(date +%Y%m%d)"
    
    # 生成新密钥
    generate_key $key_name
    
    # 更新服务配置 (需要重启服务)
    echo "密钥已轮换，请重启服务以生效"
}

# 定期轮换
case "$1" in
    rotate)
        rotate_key "$2"
        ;;
    generate)
        generate_key "$2"
        ;;
    *)
        echo "用法: $0 {generate|rotate} <key_name>"
        ;;
esac
```

## 五、审计与合规

### 5.1 审计日志

```python
# 审计日志中间件
import logging
import json
from datetime import datetime
from fastapi import Request
import hashlib

class AuditLogger:
    def __init__(self, log_file: str = "/var/log/formalmath/audit.log"):
        self.logger = logging.getLogger("audit")
        handler = logging.FileHandler(log_file)
        handler.setFormatter(logging.Formatter('%(message)s'))
        self.logger.addHandler(handler)
        self.logger.setLevel(logging.INFO)
    
    async def log_request(self, request: Request, user: str, action: str):
        """记录请求审计日志"""
        log_entry = {
            "timestamp": datetime.utcnow().isoformat(),
            "user": user,
            "action": action,
            "method": request.method,
            "path": request.url.path,
            "client_ip": request.client.host,
            "user_agent": request.headers.get("user-agent"),
            "request_id": request.headers.get("x-request-id", "unknown")
        }
        
        self.logger.info(json.dumps(log_entry))
    
    def log_data_access(self, user: str, resource: str, operation: str):
        """记录数据访问"""
        log_entry = {
            "timestamp": datetime.utcnow().isoformat(),
            "event": "data_access",
            "user": user,
            "resource": resource,
            "operation": operation,
            "hash": hashlib.sha256(f"{user}:{resource}:{operation}".encode()).hexdigest()[:16]
        }
        self.logger.info(json.dumps(log_entry))

# FastAPI中间件
@app.middleware("http")
async def audit_middleware(request: Request, call_next):
    response = await call_next(request)
    
    # 记录敏感操作
    if request.method in ["POST", "PUT", "DELETE"]:
        audit_logger = AuditLogger()
        user = request.headers.get("x-user-id", "anonymous")
        await audit_logger.log_request(request, user, f"{request.method}_{request.url.path}")
    
    return response
```

## 六、应急响应

### 6.1 安全事件分级

| 级别 | 描述 | 示例 | 响应时间 |
|------|------|------|----------|
| **Critical** | 系统被攻破 | 服务器被入侵、数据泄露 | 15分钟 |
| **High** | 严重漏洞 | 远程代码执行、权限提升 | 1小时 |
| **Medium** | 一般漏洞 | XSS、CSRF、信息泄露 | 24小时 |
| **Low** | 轻微问题 | 配置不当、弱密码 | 1周 |

### 6.2 应急响应流程

```
检测 -> 遏制 -> 消除 -> 恢复 -> 总结

1. 检测 (Detection)
   - 监控告警触发
   - 用户报告
   - 安全扫描发现

2. 遏制 (Containment)
   - 隔离受影响系统
   - 阻断攻击向量
   - 保留证据

3. 消除 (Eradication)
   - 清除恶意代码
   - 修复漏洞
   - 重置凭证

4. 恢复 (Recovery)
   - 从干净备份恢复
   - 逐步恢复服务
   - 持续监控

5. 总结 (Lessons)
   - 事件分析
   - 改进措施
   - 更新预案
```

### 6.3 应急响应脚本

```bash
#!/bin/bash
# security_incident_response.sh

INCIDENT_TYPE=$1
AFFECTED_HOST=$2

log_incident() {
    echo "[$(date -Iseconds)] $1" >> /var/log/formalmath/security_incidents.log
}

case $INCIDENT_TYPE in
    "breach")
        log_incident "SECURITY BREACH detected on $AFFECTED_HOST"
        
        # 1. 隔离主机
        echo "正在隔离主机 $AFFECTED_HOST..."
        sudo iptables -A INPUT -s $AFFECTED_HOST -j DROP
        sudo iptables -A OUTPUT -d $AFFECTED_HOST -j DROP
        
        # 2. 停止服务
        echo "停止相关服务..."
        ssh $AFFECTED_HOST "sudo systemctl stop formalmath-*"
        
        # 3. 创建取证镜像
        echo "创建取证镜像..."
        sudo dd if=/dev/sda1 of=/forensics/$(date +%Y%m%d)_${AFFECTED_HOST}.img bs=1M
        
        # 4. 通知
        echo "发送紧急通知..."
        mail -s "SECURITY BREACH: $AFFECTED_HOST" security@formalmath.org < /dev/null
        ;;
        
    "ddos")
        log_incident "DDoS attack detected"
        
        # 启用Cloudflare
        echo "启用DDoS防护..."
        # 调用API启用更严格的规则
        
        # 限流
        sudo ufw limit proto tcp from any to any port 80
        sudo ufw limit proto tcp from any to any port 443
        ;;
        
    "malware")
        log_incident "Malware detected on $AFFECTED_HOST"
        
        # 隔离
        ssh $AFFECTED_HOST "sudo systemctl isolate rescue.target"
        
        # 扫描
        ssh $AFFECTED_HOST "sudo clamscan -r / --infected --remove=no"
        ;;
        
    *)
        echo "Unknown incident type: $INCIDENT_TYPE"
        exit 1
        ;;
esac
```
