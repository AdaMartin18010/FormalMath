% ============= 底部: Kähler流形 =============
\node[anchor=north, font=\small, draw=green!30, fill=green!5, rounded corners, inner sep=10pt] 
    at (4, -4.5) {
    \begin{tabular}{ll}
        \textbf{Kähler流形:} & \\
        同时具有: & 1. 复结构 $J$ \\
                 & 2. Riemann度量 $g$ \\
                 & 3. 辛结构 $\omega$ \\[5pt]
        相容条件: & $\omega(u, v) = g(Ju, v)$\\
                  & $g(Ju, Jv) = g(u, v)$ \\[5pt]
        例子: & $\mathbb{C}P^n$, 代数簇, Calabi-Yau
    \end{tabular}
};
