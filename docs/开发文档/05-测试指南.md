## 测试类型

### 1. 功能测试

验证功能是否符合需求：

```python
# 示例：评估系统功能测试
def test_formative_assessment():
    """测试形成性评价功能"""
    system = FormalMathAssessmentSystem()
    
    # 注册学习者
    profile = system.register_learner("test_001", "测试用户")
    assert profile.learner_id == "test_001"
    
    # 进行形成性评价
    result = system.conduct_formative_assessment("test_001")
    
    # 验证结果
    assert result.overall_score >= 0
    assert result.overall_score <= 100
    assert result.level in ['BEGINNER', 'DEVELOPING', 'PROFICIENT', 'ADVANCED', 'EXPERT']
```

### 2. 内容测试

验证文档内容的正确性：

```python
# 示例：元数据验证
def test_document_metadata():
    """测试文档元数据完整性"""
    extractor = MetadataExtractor('.')
    records = extractor.scan_directory('docs/**/*.md')
    
    for record in records:
        # 必需字段
        assert 'msc_primary' in record.metadata, f"{record.file_path} 缺少 msc_primary"
        assert 'title' in record.metadata, f"{record.file_path} 缺少 title"
        
        # MSC 格式验证
        msc = record.metadata['msc_primary']
        assert re.match(r'^\d{2}-XX$', msc), f"{record.file_path} MSC 格式错误"
```

### 3. 数学正确性测试

验证数学内容的正确性：

```python
# 示例：公式语法测试
def test_math_formula_syntax():
    """测试数学公式语法"""
    import katex
    
    formulas = [
        r"x^2 + y^2 = z^2",
        r"\\forall x \\in \\mathbb{R}",
        r"\\int_{a}^{b} f(x) dx",
    ]
    
    for formula in formulas:
        try:
            katex.renderToString(formula)
        except Exception as e:
            pytest.fail(f"公式渲染失败: {formula}, 错误: {e}")
```

### 4. 链接测试

验证内部和外部链接：

```python
# 示例：链接检查
def test_internal_links():
```

## 单元测试

### Python 单元测试

```python
# tests/unit/tools/test_assessment_system.py
import pytest
from tools.assessment_system.assessment_system import FormalMathAssessmentSystem
from tools.assessment_system.evaluation_criteria import MathematicalAbilityProfile

class TestAssessmentSystem:
    """评估系统单元测试"""
    
    @pytest.fixture
    def system(self):
        """创建评估系统实例"""
        return FormalMathAssessmentSystem()
    
    @pytest.fixture
    def sample_learner(self, system):
        """创建测试学习者"""
        return system.register_learner("test_001", "测试用户")
    
    def test_register_learner(self, system):
        """测试学习者注册"""
        profile = system.register_learner("test_002", "新用户")
        
        assert profile.learner_id == "test_002"
        assert profile.name == "新用户"
        assert profile.ability_profile is not None
    
    def test_conduct_formative_assessment(self, system, sample_learner):
        """测试形成性评价"""
        result = system.conduct_formative_assessment("test_001")
        
        assert result.learner_id == "test_001"
        assert 0 <= result.overall_score <= 100
        assert result.level in ['BEGINNER', 'DEVELOPING', 'PROFICIENT', 'ADVANCED', 'EXPERT']
        assert len(result.recommendations) >= 0
    
    def test_update_learner_ability(self, system, sample_learner):
        """测试更新能力档案"""
        ability = MathematicalAbilityProfile(
            conceptual_understanding=80,
            procedural_fluency=75,
            strategic_competence=70,
            adaptive_reasoning=85,
            productive_disposition=90
        )
        
        system.update_learner_ability("test_001", ability)
        
        updated = system.get_learner_profile("test_001")
        assert updated.ability_profile.conceptual_understanding == 80
    
    @pytest.mark.parametrize("score,expected_level", [
        (30, "BEGINNER"),
        (50, "DEVELOPING"),
        (70, "PROFICIENT"),
        (85, "ADVANCED"),
        (95, "EXPERT"),
    ])
    def test_score_to_level_conversion(self, system, score, expected_level):
        """测试分数到等级的转换"""
        level = system._score_to_level(score)
        assert level == expected_level
```

```python
import pytest
from fastapi.testclient import TestClient

class TestAPIIntegration:
    """API 集成测试"""
    
    @pytest.fixture
    def client(self):
        """创建测试客户端"""

    def test_create_and_get_assessment(self, client):
        """测试创建和获取评估"""
        # 创建评估
        response = client.post('/api/v1/assessments', json={
            'learner_id': 'test_001',

        assert response.json()['data']['learner_id'] == 'test_001'
```

## CI/CD 集成

### GitHub Actions 配置

```yaml
name: Tests

on:
  push:
    branches: [main, develop]
  pull_request:
    branches: [main]

  unit-tests:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v3

      - name: Run unit tests

      - name: Upload coverage
        uses: codecov/codecov-action@v3
```
