# 控制论实例与代码实现

## 实例1：倒立摆控制

### 问题描述
控制小车上倒立摆的角度 $\theta$ 保持在竖直位置。

### 状态空间模型
$$\dot{x} = Ax + Bu$$
其中 $x = [\theta, \dot{\theta}, x, \dot{x}]^T$

### Python实现（LQR控制）

```python
import numpy as np
from scipy.linalg import solve_continuous_are
import matplotlib.pyplot as plt

# 倒立摆参数
M = 1.0  # 小车质量
m = 0.1  # 摆质量
l = 0.5  # 摆长
g = 9.81  # 重力加速度

# 线性化后的状态矩阵（在竖直位置）
A = np.array([[0, 1, 0, 0],
              [(M+m)*g/(M*l), 0, 0, 0],
              [0, 0, 0, 1],
              [-m*g/M, 0, 0, 0]])

B = np.array([[0],
              [-1/(M*l)],
              [0],
              [1/M]])

# LQR设计
Q = np.diag([10, 1, 1, 1])  # 状态权重
R = np.array([[1]])  # 控制权重

# 求解Riccati方程
P = solve_continuous_are(A, B, Q, R)
K = np.linalg.inv(R) @ B.T @ P

print("LQR增益矩阵 K:", K)

# 仿真
def pendulum_dynamics(x, u):
    """非线性倒立摆动力学"""
    theta, theta_dot, pos, vel = x
    
    # 非线性方程
    sin_theta = np.sin(theta)
    cos_theta = np.cos(theta)
    
    denom = M + m * sin_theta**2
    
    theta_ddot = ((M+m)*g*sin_theta - m*l*theta_dot**2*sin_theta*cos_theta + cos_theta*u) / (l*denom)
    pos_ddot = (m*l*theta_dot**2*sin_theta - m*g*sin_theta*cos_theta + u) / denom
    
    return np.array([theta_dot, theta_ddot, vel, pos_ddot])

# 仿真
dt = 0.01
t = np.arange(0, 10, dt)
x = np.array([0.1, 0, 0, 0])  # 初始状态（偏离10度）
x_history = [x.copy()]

for _ in t[1:]:
    u = -K @ x  # LQR控制
    x_dot = pendulum_dynamics(x, u)
    x = x + x_dot * dt
    x_history.append(x.copy())

x_history = np.array(x_history)

# 绘图
fig, axes = plt.subplots(2, 2, figsize=(12, 8))
axes[0, 0].plot(t, x_history[:, 0] * 180/np.pi)
axes[0, 0].set_ylabel('Angle (deg)')
axes[0, 0].grid(True)

axes[0, 1].plot(t, x_history[:, 1])
axes[0, 1].set_ylabel('Angular Velocity')
axes[0, 1].grid(True)

axes[1, 0].plot(t, x_history[:, 2])
axes[1, 0].set_ylabel('Position')
axes[1, 0].set_xlabel('Time')
axes[1, 0].grid(True)

axes[1, 1].plot(t, [-K @ x for x in x_history])
axes[1, 1].set_ylabel('Control Input')
axes[1, 1].set_xlabel('Time')
axes[1, 1].grid(True)

plt.tight_layout()
plt.savefig('inverted_pendulum_lqr.png')
plt.show()
```

## 实例2：PID控制器设计

```python
class PIDController:
    def __init__(self, Kp, Ki, Kd):
        self.Kp = Kp
        self.Ki = Ki
        self.Kd = Kd
        self.integral = 0
        self.prev_error = 0
    
    def update(self, error, dt):
        self.integral += error * dt
        derivative = (error - self.prev_error) / dt
        self.prev_error = error
        return self.Kp * error + self.Ki * self.integral + self.Kd * derivative

# 温度控制系统仿真
Kp, Ki, Kd = 2.0, 0.5, 0.1
pid = PIDController(Kp, Ki, Kd)

# 系统参数（一阶惯性）
tau = 2.0  # 时间常数
K = 1.0    # 增益

# 仿真
dt = 0.1
t = np.arange(0, 50, dt)
setpoint = 100  # 目标温度
temperature = 20  # 初始温度
temp_history = [temperature]

for _ in t[1:]:
    error = setpoint - temperature
    control = pid.update(error, dt)
    # 一阶惯性系统
    temperature += (K * control - temperature) / tau * dt
    temp_history.append(temperature)

plt.figure(figsize=(10, 5))
plt.plot(t, temp_history, label='Temperature')
plt.axhline(y=setpoint, color='r', linestyle='--', label='Setpoint')
plt.xlabel('Time')
plt.ylabel('Temperature')
plt.legend()
plt.grid(True)
plt.savefig('pid_control.png')
plt.show()
```

## 习题

### 习题1
判断系统的可控性：$A = \begin{pmatrix} 1 & 1 \\ 0 & 1 \end{pmatrix}$，$B = \begin{pmatrix} 1 \\ 0 \end{pmatrix}$

**解答**：可控性矩阵 $\mathcal{C} = [B \ AB] = \begin{pmatrix} 1 & 1 \\ 0 & 0 \end{pmatrix}$，秩为1 < 2，系统不可控。

### 习题2
设计状态反馈使闭环极点在 $-1, -2$。

**解答**：设 $K = [k_1 \ k_2]$，$A - BK$ 的特征值为 $-1, -2$。
解得 $k_1, k_2$ 使 $\det(sI - (A-BK)) = (s+1)(s+2)$。
