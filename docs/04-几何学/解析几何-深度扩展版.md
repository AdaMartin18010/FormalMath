        surface = NURBSSurface(control_points, weights, knot_vectors, degrees)
        self.nurbs_surfaces.append(surface)
        return surface

    def boolean_operations(self, solid1, solid2, operation):
        """布尔运算"""
        if operation == "union":
            return self.union(solid1, solid2)
        elif operation == "intersection":
            return self.intersection(solid1, solid2)
        elif operation == "difference":
            return self.difference(solid1, solid2)

    def surface_fairing(self, surface, fairness_weight):
        """曲面光顺"""
        return self.optimize_surface(surface, fairness_weight)
