## 1. 素性测试算法

### 试除法

```python
def is_prime_trial(n):
    """试除法素性测试"""
    if n < 2:
        return False
    if n == 2:
        return True
    if n % 2 == 0:
        return False

    for i in range(3, int(n**0.5) + 1, 2):
        if n % i == 0:
            return False
    return True
```

### Miller-Rabin素性测试

```python
def miller_rabin(n, k=10):
    """Miller-Rabin素性测试"""
    if n < 2: return False
    if n == 2 or n == 3: return True
    if n % 2 == 0: return False

    # 写成 n-1 = 2^r * d
    r, d = 0, n - 1
    while d % 2 == 0:
        r += 1
        d //= 2

    for _ in range(k):
        a = random.randrange(2, n - 1)
        x = pow(a, d, n)

        if x == 1 or x == n - 1:
            continue

        for _ in range(r - 1):
            x = pow(x, 2, n)
            if x == n - 1:
                break
        else:
            return False

    return True
```

## 2. 因数分解算法

### 试除法

```python
def trial_factorization(n):
    """试除法因数分解"""
    factors = []
    d = 2
    while d * d <= n:
        while n % d == 0:
            factors.append(d)
            n //= d
        d += 1
    if n > 1:
        factors.append(n)
    return factors
```

### Pollard's Rho算法

```python
def pollards_rho(n):
    """Pollard's Rho因数分解算法"""
    if n % 2 == 0:
        return 2

    x = random.randrange(2, n)
    y = x
    c = random.randrange(1, n)
    d = 1

    f = lambda x: (pow(x, 2, n) + c) % n

    while d == 1:
        x = f(x)
        y = f(f(y))
        d = math.gcd(abs(x - y), n)

    if d == n:
        return None  # 失败，需要重试
    return d
```

### 费马分解法

```python
def fermat_factorization(n):
    """费马分解法（适用于两因子接近的情况）"""
    a = math.ceil(math.sqrt(n))
    b2 = a*a - n

    while not int(math.sqrt(b2))**2 == b2:
        a += 1
        b2 = a*a - n

    b = int(math.sqrt(b2))
    return (a - b, a + b)
```

## 3. 模运算算法

### 快速幂算法

```python
def mod_pow(base, exp, mod):
    """快速模幂算法（平方乘算法）"""
    result = 1
    base = base % mod

    while exp > 0:
        if exp % 2 == 1:
            result = (result * base) % mod
        exp = exp >> 1
        base = (base * base) % mod

    return result
```

### 扩展欧几里得算法

```python
def extended_gcd(a, b):
    """扩展欧几里得算法"""
    if a == 0:
        return b, 0, 1

    gcd, x1, y1 = extended_gcd(b % a, a)
    x = y1 - (b // a) * x1
    y = x1

    return gcd, x, y

def mod_inverse(a, m):
    """计算模逆元"""
    gcd, x, _ = extended_gcd(a % m, m)
    if gcd != 1:
        return None  # 逆元不存在
    return (x % m + m) % m
```

## 4. 素数分布计算

### 素数计数函数 π(x)

```python
def prime_pi(x):
    """计算不超过x的素数个数（Meissel-Lehmer算法简化版）"""
    if x < 2:
        return 0

    # 简单版本：埃拉托斯特尼筛
    sieve = [True] * (x + 1)
    sieve[0] = sieve[1] = False

    for i in range(2, int(x**0.5) + 1):
        if sieve[i]:
            for j in range(i*i, x + 1, i):
                sieve[j] = False

    return sum(sieve)

## 5. 连分数计算

def continued_fraction_sqrt(n, terms=20):
    """计算√n的连分数展开"""
    a0 = int(n**0.5)
    if a0 * a0 == n:
        return [a0]

    cf = [a0]
    m, d, a = 0, 1, a0

    seen = {}
    while terms > 0:
        m = d * a - m
        d = (n - m * m) // d
        if d == 0:
            break
        a = (a0 + m) // d
        cf.append(a)
        terms -= 1

    return cf

def verify_goldbach(limit):
    """验证偶数可以表示为两个素数之和"""
    primes = [i for i in range(2, limit) if is_prime_trial(i)]
    prime_set = set(primes)

    for n in range(4, limit, 2):
        found = False
        for p in primes:
            if p > n:
                break
            if n - p in prime_set:
                found = True
                break
        if not found:
            return False, n
    return True, None

result, counterexample = verify_goldbach(1000)
print(f"验证结果: {result}")
```
