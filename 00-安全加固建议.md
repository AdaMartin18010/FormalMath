### 3.2 日志脱敏处理

```python
# FormalMath-Enhanced/api/app/core/logging.py
import re
import logging

class SensitiveDataFilter(logging.Filter):
    """敏感数据过滤器"""
    
    PATTERNS = [
        (r'password["\']?\s*[:=]\s*["\'][^"\']+', 'password=***'),
        (r'token["\']?\s*[:=]\s*["\'][^"\']+', 'token=***'),
        (r'authorization["\']?\s*[:=]\s*["\'][^"\']+', 'authorization=***'),
    ]
    
    def filter(self, record):
        message = record.getMessage()
        for pattern, replacement in self.PATTERNS:
            message = re.sub(pattern, replacement, message, flags=re.I)
        record.msg = message
        return True

# 配置日志
logging.basicConfig(
    level=logging.INFO,
    format='%(asctime)s - %(name)s - %(levelname)s - %(message)s',
    handlers=[
        logging.StreamHandler()
    ]
)

# 添加过滤器
for handler in logging.root.handlers:
    handler.addFilter(SensitiveDataFilter())
```

### 3.3 请求限流

```python
# FormalMath-Enhanced/api/app/core/rate_limit.py
from fastapi import Request, HTTPException
import redis
import time

class RateLimiter:
    def __init__(self, redis_client: redis.Redis):
        self.redis = redis_client
        
    async def check_rate_limit(
        self, 
        request: Request, 
        max_requests: int = 100,
        window: int = 60
    ):
        """检查请求频率限制"""
        client_ip = request.client.host
        key = f"rate_limit:{client_ip}"
        
        current = self.redis.get(key)
        if current is None:
            self.redis.setex(key, window, 1)
            return True
            
        if int(current) >= max_requests:
            raise HTTPException(
                status_code=429,
                detail="Too many requests"
            )
            
        self.redis.incr(key)
        return True
```

#### 4.1.2 安全事件告警

```python
# FormalMath-Enhanced/api/app/core/security_alerts.py
import requests
from datetime import datetime

class SecurityAlertManager:
    def __init__(self, webhook_url: str):
        self.webhook_url = webhook_url
        
    async def send_alert(self, event_type: str, details: dict):
        """发送安全告警"""
        payload = {
            "timestamp": datetime.utcnow().isoformat(),
            "event_type": event_type,
            "severity": details.get("severity", "medium"),
            "description": details.get("description"),
            "source_ip": details.get("source_ip"),
            "user_id": details.get("user_id"),
        }
        
        requests.post(self.webhook_url, json=payload)
```

### 4.2 容器安全加固

```dockerfile
# FormalMath-Interactive/Dockerfile - 加固版本

# 阶段1: 构建
FROM node:18-alpine AS builder

# 创建非root用户
RUN addgroup -g 1001 -S nodejs && \
    adduser -S nextjs -u 1001

WORKDIR /app
COPY package*.json ./
RUN npm ci --only=production

COPY . .
RUN npm run build

# 阶段2: 生产运行
FROM nginx:1.25-alpine

# 安全更新
RUN apk upgrade --no-cache && \
    apk add --no-cache dumb-init

# 创建非root用户
RUN addgroup -g 1001 -S nodejs && \
    adduser -S nextjs -u 1001 -G nodejs

# 配置nginx
COPY nginx.conf /etc/nginx/conf.d/default.conf

# 复制构建产物
COPY --from=builder --chown=nextjs:nodejs /app/dist /usr/share/nginx/html

# 设置权限
RUN chown -R nextjs:nodejs /var/cache/nginx && \
    chown -R nextjs:nodejs /var/log/nginx && \
    chown -R nextjs:nodejs /etc/nginx/conf.d && \
    touch /var/run/nginx.pid && \
    chown -R nextjs:nodejs /var/run/nginx.pid

# 健康检查
HEALTHCHECK --interval=30s --timeout=3s --start-period=5s --retries=3 \
    CMD wget --quiet --tries=1 --spider http://localhost:80/ || exit 1

# 切换到非root用户
USER nextjs

EXPOSE 80

ENTRYPOINT ["dumb-init", "--"]
CMD ["nginx", "-g", "daemon off;"]
```
