- 建立过程性评价体系

- 《深化新时代教育评价改革总体方案》

```python
def calculate_participation_score(learning_history):
    """计算学习参与度"""
    time_score = calculate_time_score(learning_history)
    frequency_score = calculate_frequency_score(learning_history)
    depth_score = calculate_depth_score(learning_history)
    breadth_score = calculate_breadth_score(learning_history)

    participation_score = (
        time_score * 0.3 +
        frequency_score * 0.3 +
        depth_score * 0.2 +
        breadth_score * 0.2
    )

    return participation_score
```

## 📈 三、增值评价设计 (编号: DESIGN.003.03)

### 3.1 增值评价模型

#### 模型1: 能力增值模型

**模型原理**: 评估学习者在不同能力维度上的增值

**模型设计**:

```python
class ValueAddedModel:
    def calculate_ability_value_added(self, initial_ability, current_ability):
        """计算能力增值"""
        value_added = {}
        for dimension in ability_dimensions:
            initial = initial_ability[dimension]
            current = current_ability[dimension]
            value_added[dimension] = current - initial

        return value_added

    def calculate_relative_value_added(self, value_added, initial_ability):
        """计算相对增值"""
        relative_value_added = {}
        for dimension in ability_dimensions:
            if initial_ability[dimension] > 0:
                relative = value_added[dimension] / initial_ability[dimension]
            else:
                relative = value_added[dimension]
            relative_value_added[dimension] = relative

        return relative_value_added
```

```python
def calculate_knowledge_value_added(initial_knowledge, current_knowledge):
```

```python
def calculate_ability_value_added_by_level(
    initial_ability, current_ability
):
    """按层次计算能力增值"""
    value_added_by_level = {}
    for level in [0, 1, 2, 3]:
```

#### 指标2: 知识增值指标

```python
def calculate_knowledge_value_added_indicators(
    initial_knowledge, current_knowledge
):
    """计算知识增值指标"""
    # 1. 新掌握概念数量
    new_concepts = set(current_knowledge.keys()) - set(initial_knowledge.keys())
    new_concepts_count = len(new_concepts)

    # 2. 概念掌握度提升
    mastery_improvement = {}
    for concept in set(initial_knowledge.keys()) & set(current_knowledge.keys()):
        initial = initial_knowledge[concept]
        current = current_knowledge[concept]
        mastery_improvement[concept] = current - initial

    # 3. 知识关联度提升
    initial_connections = count_knowledge_connections(initial_knowledge)
    current_connections = count_knowledge_connections(current_knowledge)
    connection_improvement = current_connections - initial_connections

    return {
        'new_concepts': new_concepts_count,
        'mastery_improvement': mastery_improvement,
        'connection_improvement': connection_improvement
    }
```
