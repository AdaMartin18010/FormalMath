# 归档函数
function Archive-File($RelativePath) {
    $src = Join-Path $BaseDir $RelativePath
    $dst = Join-Path $ArchiveDir $RelativePath
    if (Test-Path $src) {
        $dstParent = Split-Path $dst -Parent
        if (-not (Test-Path $dstParent)) { New-Item -ItemType Directory -Path $dstParent -Force | Out-Null }
        Move-Item -Path $src -Destination $dst -Force
        Write-Host "ARCHIVED: $RelativePath"
    }
}

# 遍历 $BaseDir 下所有 .md 文件
Get-ChildItem -Path $BaseDir -Recurse -Filter "*.md" | ForEach-Object {
    $rel = $_.FullName.Substring((Resolve-Path $BaseDir).Path.Length + 1)
    if (-not $KeepList.Contains($rel)) {
        Archive-File $rel
    }
}
