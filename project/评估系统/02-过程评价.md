## 1. 过程性评价概述

### 1.1 设计理念

过程性评价关注学习过程中的表现，而非仅看最终结果。其核心思想：

> **"评价不是为了甄别，而是为了促进发展"**

### 2.2 采集技术方案

```python
# 前端埋点SDK
class LearningTracker {
    constructor(config) {
        this.userId = config.userId;
        this.sessionId = this.generateSessionId();
        this.buffer = [];
        this.flushInterval = 5000; // 5秒上报一次
        
        this.init();
    }
    
    init() {
        // 自动跟踪页面浏览
        this.trackPageView();
        
        // 自动跟踪点击
        document.addEventListener('click', (e) => {
            this.trackEvent('click', {
                target: e.target.tagName,
                id: e.target.id,
                class: e.target.className
            });
        });
        
        // 定时上报
        setInterval(() => this.flush(), this.flushInterval);
        
        // 页面关闭前上报
        window.addEventListener('beforeunload', () => this.flush());
    }
    
    trackEvent(eventType, data) {
        const event = {
            user_id: this.userId,
            session_id: this.sessionId,
            event_type: eventType,
            timestamp: new Date().toISOString(),
            page_url: window.location.href,
            data: data
        };
        
        this.buffer.push(event);
        
        // 紧急事件立即上报
        if (['exercise_submit', 'video_complete'].includes(eventType)) {
            this.flush();
        }
    }
    
    trackPageView() {
        this.trackEvent('page_view', {
            title: document.title,
            referrer: document.referrer
        });
    }
    
    trackContentView(contentId, contentType, duration) {
        this.trackEvent('content_view', {
            content_id: contentId,
            content_type: contentType,
            duration: duration
        });
    }
    
    trackExercise(exerciseId, action, data) {
        this.trackEvent(`exercise_${action}`, {
            exercise_id: exerciseId,
            ...data
        });
    }
    
    async flush() {
        if (this.buffer.length === 0) return;
        
        const events = [...this.buffer];
        this.buffer = [];
        
        try {
            await fetch('/api/v1/tracking/events', {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify({ events }),
                keepalive: true
            });
        } catch (err) {
            // 上报失败，数据回滚
            this.buffer.unshift(...events);
        }
    }
}

// 使用示例
const tracker = new LearningTracker({ userId: 'student_001' });

// 跟踪视频学习

// ... 播放结束后

    video_id: 'vid_001', 
    watch_time: 300,
    completion_rate: 1.0 
});

// 跟踪练习

    is_correct: true,

});
```

### 3.1 学习参与度

```python
class ParticipationEvaluator:
    """学习参与度评估器"""
    
    def evaluate(self, learner_id: str, period: str) -> ParticipationScore:
        """
        评估学习参与度
        
        维度:
        1. 时间投入 (40%)
        2. 内容覆盖 (30%)
        3. 活动参与 (30%)
        """
        data = self.fetch_data(learner_id, period)
        
        # 1. 时间投入评分
        time_score = self._evaluate_time_investment(data)
        
        # 2. 内容覆盖评分
        coverage_score = self._evaluate_content_coverage(data)
        
        # 3. 活动参与评分
        activity_score = self._evaluate_activity_participation(data)
        
        # 综合评分
        overall = (
            0.40 * time_score.overall +
            0.30 * coverage_score.overall +
            0.30 * activity_score.overall
        )
        
        return ParticipationScore(
            time=time_score,
            coverage=coverage_score,
            activity=activity_score,
            overall=overall
        )
    
    def _evaluate_time_investment(self, data: BehaviorData) -> TimeScore:
        """评估时间投入"""
        
        # 有效学习时长 (去除空闲时间)
        effective_time = data.session_duration - data.idle_time
        
        # 每日平均学习时长
        daily_avg = effective_time / data.days_count
        
        # 评分标准
        time_score = min(daily_avg / 3600, 1.0)  # 每天1小时满分
        
        # 学习频次
        frequency_score = min(data.weekly_frequency / 5, 1.0)  # 每周5次满分
        
        # 学习规律度 (变异系数倒数)
        regularity_score = self._calculate_regularity(data.daily_durations)
        
        return TimeScore(
            duration=time_score,
            frequency=frequency_score,
            regularity=regularity_score,
            overall=0.5*time_score + 0.3*frequency_score + 0.2*regularity_score
        )
    
    def _evaluate_content_coverage(self, data: BehaviorData) -> CoverageScore:
        """评估内容覆盖"""
        
        # 内容浏览覆盖度
        content_coverage = data.contents_viewed / data.total_contents
        
        # 视频学习覆盖度
        video_coverage = data.videos_completed / data.total_videos
        
        # 练习完成覆盖度
        exercise_coverage = data.exercises_completed / data.total_exercises
        
        return CoverageScore(
            content=min(content_coverage, 1.0),
            video=min(video_coverage, 1.0),
            exercise=min(exercise_coverage, 1.0),
            overall=0.4*content_coverage + 0.3*video_coverage + 0.3*exercise_coverage
        )
    
    def _evaluate_activity_participation(self, data: BehaviorData) -> ActivityScore:
        """评估活动参与"""
        
        # 讨论参与度
        discussion_score = min(data.discussions_joined / 3, 1.0)  # 每周3次满分
        
        # 提问参与度
        questioning_score = min(data.questions_asked / 2, 1.0)  # 每周2次满分
        
        # 互动参与度 (回答问题、点赞等)
        interaction_score = min(data.interactions / 10, 1.0)
        
        return ActivityScore(
            discussion=discussion_score,
            questioning=questioning_score,
            interaction=interaction_score,
            overall=0.4*discussion_score + 0.3*questioning_score + 0.3*interaction_score
        )
```

### 3.2 学习主动性

```python
class InitiativeEvaluator:
    """学习主动性评估器"""
    
    def evaluate(self, learner_id: str, period: str) -> InitiativeScore:
        """
        评估学习主动性
        
        维度:
        1. 主动探索 (35%)
        2. 主动提问 (25%)
        3. 主动反思 (20%)
        4. 拓展学习 (20%)
        """
        data = self.fetch_data(learner_id, period)
        
        exploration = self._evaluate_exploration(data)
        questioning = self._evaluate_questioning(data)
        reflection = self._evaluate_reflection(data)
        extension = self._evaluate_extension(data)
        
        overall = (
            0.35 * exploration +
            0.25 * questioning +
            0.20 * reflection +
            0.20 * extension
        )
        
        return InitiativeScore(
            exploration=exploration,
            questioning=questioning,
            reflection=reflection,
            extension=extension,
            overall=overall
        )
    
    def _evaluate_exploration(self, data: BehaviorData) -> float:
        """评估主动探索"""
        
        # 主动搜索行为
        search_score = min(data.searches / 5, 1.0)
        
        # 超出推荐内容的学习
        beyond_recommended = data.beyond_recommended_views / max(data.total_views, 1)
        
        # 深度阅读 (单页面停留时间)
        deep_reading_ratio = data.deep_reading_count / max(data.pages_viewed, 1)
        
        return 0.3*search_score + 0.4*beyond_recommended + 0.3*deep_reading_ratio
    
    def _evaluate_reflection(self, data: BehaviorData) -> float:
        """评估主动反思"""
        
        # 笔记记录
        note_score = min(data.notes_created / 3, 1.0)  # 每周3条笔记满分
        
        # 错题回顾
        error_review_score = data.error_reviews / max(data.errors_made, 1)
        
        # 反思日志
        reflection_score = min(data.reflection_logs / 1, 1.0)  # 每周1次反思满分
        
        return 0.4*note_score + 0.35*error_review_score + 0.25*reflection_score
```

### 3.3 学习坚持性

```python
class PersistenceEvaluator:
    """学习坚持性评估器"""
    
    def evaluate(self, learner_id: str, period: str) -> PersistenceScore:
        """
        评估学习坚持性
        
        维度:
        1. 持续学习 (40%)
        2. 规律学习 (30%)
        3. 目标完成 (30%)
        """
        data = self.fetch_data(learner_id, period)
        
        continuity = self._evaluate_continuity(data)
        regularity = self._evaluate_regularity(data)
        completion = self._evaluate_completion(data)
        
        overall = 0.40 * continuity + 0.30 * regularity + 0.30 * completion
        
        return PersistenceScore(
            continuity=continuity,
            regularity=regularity,
            completion=completion,
            overall=overall
        )
    
    def _evaluate_continuity(self, data: BehaviorData) -> float:
        """评估持续学习"""
        
        # 最长连续学习天数
        max_streak = data.max_consecutive_days
        streak_score = min(max_streak / 7, 1.0)  # 连续7天满分
        
        # 学习中断次数
        interruptions = data.interruption_count
        interruption_score = max(0, 1 - interruptions / 3)  # 3次以上中断扣分
        
        # 总学习天数占比
        study_days_ratio = data.study_days / data.total_days
        
        return 0.4*streak_score + 0.3*interruption_score + 0.3*study_days_ratio
```

```python
    def analyze_and_intervene(self, learner_id: str):
        """分析学习情况并提供干预"""
        
        # 获取过程性评价结果

        interventions = []

        if evaluation.participation.overall < 0.5:

                'actions': [

                'type': 'initiative_boost',
                'priority': 'medium',

                'actions': [
                    '推荐相关拓展内容',
                    '鼓励做学习笔记',
                    '推荐讨论话题'
                ]
```
