## 1. 报告体系概述

### 1.1 报告类型矩阵

| 报告类型 | 频率 | 受众 | 核心内容 | 格式 |
|----------|------|------|----------|------|
| **即时报告** | 实时 | 学生 | 答题结果、即时反馈 | 弹窗/Toast |
| **日度报告** | 每日 | 学生 | 日学习总结 | 页面/消息 |
| **周度报告** | 每周 | 学生/家长 | 周学习分析 | PDF/网页 |
| **月度报告** | 每月 | 学生/教师 | 月度成长分析 | PDF/网页 |
| **诊断报告** | 诊断后 | 学生/教师 | 全面能力评估 | PDF/网页 |
| **学期报告** | 每学期 | 学生/家长/教师 | 学期总结规划 | PDF |
| **班级报告** | 每周/月 | 教师 | 班级整体分析 | 网页/Excel |
| **对比报告** | 按需 | 教研团队 | 多维度对比 | 网页/PPT |

## 4. 报告生成代码

### 4.1 Python报告生成器

```python
"""
FormalMath 报告生成系统
"""

from typing import Dict, List
from dataclasses import dataclass
from datetime import datetime
import json


@dataclass
class ReportData:
    """报告数据"""
    student_info: Dict
    evaluation_result: Dict
    learning_data: Dict
    historical_data: Dict
    peer_data: Dict


class ReportGenerator:
    """报告生成器"""
    
    def __init__(self, template_dir: str = 'templates'):
        self.template_dir = template_dir
        self.templates = self._load_templates()
    
    def generate_weekly_report(self, data: ReportData) -> Dict:
        """生成周报"""
        template = self.templates['weekly']
        
        # 准备数据
        context = self._prepare_weekly_context(data)
        
        # 渲染模板
        content = self._render_template(template, context)
        
        # 生成图表
        charts = self._generate_weekly_charts(data)
        
        return {
            'type': 'weekly',
            'title': f"学习周报 - {data.student_info['name']}",
            'content': content,
            'charts': charts,
            'generated_at': datetime.now().isoformat()
        }
    
    def generate_diagnosis_report(self, data: ReportData) -> Dict:
        """生成诊断报告"""
        template = self.templates['diagnosis']
        
        context = self._prepare_diagnosis_context(data)
        content = self._render_template(template, context)
        charts = self._generate_diagnosis_charts(data)
        
        return {
            'type': 'diagnosis',
            'title': f"认知诊断报告 - {data.student_info['name']}",
            'content': content,
            'charts': charts,
            'pdf_url': self._generate_pdf(content),
            'generated_at': datetime.now().isoformat()
        }
    
    def generate_class_report(self, class_data: Dict) -> Dict:
        """生成班级报告"""
        template = self.templates['class']
        
        context = self._prepare_class_context(class_data)
        content = self._render_template(template, context)
        
        return {
            'type': 'class',
            'title': f"班级学习周报 - {class_data['class_name']}",
            'content': content,
            'generated_at': datetime.now().isoformat()
        }
    
    def _prepare_weekly_context(self, data: ReportData) -> Dict:
        """准备周报上下文数据"""
        return {
            'student_name': data.student_info['name'],
            'start_date': data.learning_data['start_date'],
            'end_date': data.learning_data['end_date'],
            'study_days': data.learning_data['study_days'],
            'total_duration': self._format_duration(
                data.learning_data['total_duration']
            ),
            'cei_score': round(data.evaluation_result['cei'], 1),
            'knowledge_score': round(
                data.evaluation_result['knowledge_mastery'] * 100
            ),
            # ... 更多数据
        }
    
    def _prepare_diagnosis_context(self, data: ReportData) -> Dict:
        """准备诊断报告上下文"""
        result = data.evaluation_result
        
        return {
            'student_name': data.student_info['name'],
            'student_id': data.student_info['id'],
            'diagnosis_date': result['timestamp'],
            'cei_score': round(result['cei'], 1),
            'cei_level': self._get_cei_level(result['cei']),
            'current_level': result['ability_level']['current'],
            'recommended_level': result['ability_level']['recommended'],
            'confidence': round(result['confidence'] * 100),
            'knowledge_score': round(result['knowledge_mastery'] * 100),
            'skill_score': round(result['skill_proficiency'] * 100),
            'problem_score': round(result['problem_solving'] * 100),
            'creativity_score': round(result['creative_thinking'] * 100),
            'l0_score': round(result['level_scores']['L0'] * 100),
            'l1_score': round(result['level_scores']['L1'] * 100),
            'l2_score': round(result['level_scores']['L2'] * 100),

    def _render_template(self, template: str, context: Dict) -> str:
        """渲染模板"""
        # 使用简单的字符串替换
        # 生产环境建议使用Jinja2等模板引擎
        result = template
        for key, value in context.items():
            placeholder = f'{{{key}}}'
            result = result.replace(placeholder, str(value))
        return result

    def _get_cei_level(self, cei: float) -> str:
        """获取CEI等级"""
        if cei >= 95:
            return '卓越'
        elif cei >= 85:
            return '优秀'
        elif cei >= 70:
            return '良好'
        elif cei >= 55:
            return '中等'
        elif cei >= 40:
            return '待提高'
        else:

    def _format_duration(self, seconds: int) -> str:
        """格式化时长"""
        hours = seconds // 3600
        minutes = (seconds % 3600) // 60
        return f"{hours}小时{minutes}分"
    
    def _load_templates(self) -> Dict[str, str]:
        """加载模板"""
        # 从文件加载模板

    def _load_template_file(self, filename: str) -> str:
        """加载模板文件"""
        try:
            with open(f"{self.template_dir}/{filename}", 'r', encoding='utf-8') as f:
```

```python
import matplotlib.pyplot as plt
import numpy as np
from typing import Dict, List
import base64
from io import BytesIO


class ChartGenerator:
    """图表生成器"""
    
    def generate_radar_chart(self, data: Dict[str, float]) -> str:
        """生成雷达图 (返回base64)"""
        categories = list(data.keys())
        values = list(data.values())
        
        # 闭合图形
        values += values[:1]
        angles = np.linspace(0, 2 * np.pi, len(categories), endpoint=False).tolist()
        angles += angles[:1]
        
        fig, ax = plt.subplots(figsize=(8, 8), subplot_kw=dict(projection='polar'))
        ax.plot(angles, values, 'o-', linewidth=2, color='#1890ff')
        ax.fill(angles, values, alpha=0.25, color='#1890ff')
        ax.set_xticks(angles[:-1])
        ax.set_xticklabels(categories)
        ax.set_ylim(0, 100)
        
        # 保存为base64
        buffer = BytesIO()
        plt.savefig(buffer, format='png', bbox_inches='tight')
        buffer.seek(0)
        image_png = buffer.getvalue()
        buffer.close()
        
        return base64.b64encode(image_png).decode()
    
    def generate_progress_bar(self, value: float, label: str) -> str:
        """生成进度条SVG"""
        filled = int(value / 5)  # 20个格子
        empty = 20 - filled
        
        bar = '█' * filled + '░' * empty
        return f"{bar} {value}%"
```

```python
class ReportDistributor:
    """报告分发器"""
    
    def __init__(self):
        self.channels = {
            'in_app': InAppChannel(),
            'email': EmailChannel(),
            'push': PushNotificationChannel(),
            'wechat': WechatChannel()
        }
    
    def distribute(self, report: Dict, recipients: List[Dict]):
        """分发报告"""
        for recipient in recipients:
            preferred_channel = recipient.get('preferred_channel', 'in_app')
            channel = self.channels.get(preferred_channel)
            
            if channel:
                channel.send(report, recipient)
    
    def schedule_reports(self):
        """定时报告调度"""
        from apscheduler.schedulers.background import BackgroundScheduler
        
        scheduler = BackgroundScheduler()
        
        # 每周日晚8点发送周报
        scheduler.add_job(
            self._send_weekly_reports,
            'cron',
            day_of_week='sun',
            hour=20,
            minute=0
        )
        
        # 每月1日发送月报
        scheduler.add_job(
            self._send_monthly_reports,
            'cron',
            day=1,
            hour=9,
            minute=0
        )
        
        scheduler.start()
```
