```python
class PathAdjuster:
    """路径调整器 - 根据学习表现动态调整路径"""
    
    def adjust_based_on_performance(
        self,
        path: LearningPath,
        activities: List[LearningActivity]
    ) -> Optional[PathAdjustment]:
        """基于表现调整路径"""
        
        # 1. 分析最近的表现
        recent_activities = activities[-10:]  # 最近10个活动
        
        # 2. 检测困难领域
        struggles = self.profiler.detect_struggle_areas(
            self.user_profile.user_id,
            recent_activities
        )
        
        # 3. 识别学习缺口
        current_concept = self._get_current_concept(path)
        if current_concept:
            gaps = self.profiler.identify_gaps(
                self.user_profile.user_id,
                [current_concept.id],
                self.user_profile.mastered_concepts
            )
            
            if gaps:
                # 需要补充先修知识
                return self._insert_prerequisite_nodes(path, gaps)
        
        # 4. 根据表现调整难度
        avg_score = np.mean([a.score for a in recent_activities if a.score])
        
        if avg_score > 85:
            # 表现优秀，可以尝试更高难度
            return self._suggest_advanced_topics(path)
        elif avg_score < 60:
            # 表现不佳，需要复习或降低难度
            return self._add_review_nodes(path, struggles)
        
        return None
    
    def _insert_prerequisite_nodes(
        self,
        path: LearningPath,
        gaps: List[str]
    ) -> PathAdjustment:
        """在路径中插入先修节点"""
        # 找到当前位置
        current_idx = self._get_current_index(path)
        
        # 插入先修节点
        new_nodes = []
        for gap_id in gaps[:3]:  # 最多插入3个
            concept = self.kg.get_concept(gap_id)
            if concept:
                node = PathNode(
                    node_id=f"gap_{gap_id}_{int(time.time())}",
                    concept=concept,
                    order=current_idx,
                    status=PathStatus.PENDING,
                    priority_score=1.5  # 高优先级
                )
                new_nodes.append(node)
        
        # 调整后续节点顺序
        for node in path.nodes[current_idx:]:
            node.order += len(new_nodes)
        
        # 插入新节点
        path.nodes = path.nodes[:current_idx] + new_nodes + path.nodes[current_idx:]
        
        return PathAdjustment(
            adjustment_id=f"adj_{uuid.uuid4().hex[:8]}",
            path_id=path.path_id,
            user_id=self.user_profile.user_id,
            trigger_reason="识别到知识缺口",
            adjustment_type="add",
            new_nodes=new_nodes
        )
```

```python
            # 更新缓存

                f"path:{path_id}",
```

```python
class PathScorer:
    def __init__(self, user_profile: UserProfile):
        self.user_profile = user_profile
        self.kg = get_knowledge_graph()
        self._precompute_distances()
    
    def _precompute_distances(self):
        """预计算概念间距离"""
        self.distance_cache = {}
        
        # 使用Floyd-Warshall或Johnson算法预计算
        import networkx as nx
        
        G = self.kg.to_networkx()
        
        # 为每个节点计算到所有其他节点的最短路径
        for source in G.nodes():
            lengths = nx.single_source_shortest_path_length(G, source)
            for target, length in lengths.items():
                self.distance_cache[(source, target)] = length
    
    def estimate_remaining_cost(self, current: str, targets: Set[str], visited: Set[str]) -> float:
        """使用预计算的距离"""
        if not targets:
            return 0
        
        estimates = []
        for target in targets:
            if target in visited:
                continue
            distance = self.distance_cache.get((current, target), float('inf'))
            estimates.append(distance)
        
        return min(estimates) * 5 if estimates else 0
```

```python
from functools import lru_cache
import hashlib

class PathGenerator:
    def __init__(self, user_profile: UserProfile):
        self.user_profile = user_profile
        self.kg = get_knowledge_graph()
    
    def _get_cache_key(self, target_concepts: List[str], algorithm: str) -> str:
        """生成缓存键"""
        # 基于用户特征和目标概念生成唯一键
        key_data = {
            "user_level": self.user_profile.current_level.value,
            "mastered": sorted(self.user_profile.mastered_concepts.keys()),
            "targets": sorted(target_concepts),
            "algorithm": algorithm
        }
        return hashlib.md5(
            json.dumps(key_data, sort_keys=True).encode()
        ).hexdigest()
    
    def generate(self, target_concepts: List[str], algorithm: str = "astar") -> Optional[LearningPath]:
        """生成路径，支持缓存"""
        cache_key = self._get_cache_key(target_concepts, algorithm)
        
        # 尝试从缓存获取
        cached = self._get_from_cache(cache_key)
        if cached:
            return cached
        
        # 生成新路径
        if algorithm == "astar":
            path = self._generate_astar(target_concepts)
        elif algorithm == "dp":
            path = self._generate_dp(target_concepts)
        else:
            raise ValueError(f"Unknown algorithm: {algorithm}")
        
        # 缓存结果
        if path:
            self._save_to_cache(cache_key, path)
        
        return path
```

```python
from neo4j import GraphDatabase

class Neo4jKnowledgeGraph:
    """Neo4j知识图谱实现"""
    
    def __init__(self, uri: str, user: str, password: str):
        self.driver = GraphDatabase.driver(uri, auth=(user, password))
    
    def get_prerequisites(self, concept_id: str, recursive: bool = False) -> List[str]:
        """使用Cypher查询先修概念"""
        with self.driver.session() as session:
            if recursive:
                query = """
                MATCH path = (c:Concept {id: $id})-[:PREREQUISITE*]->(prereq:Concept)
                RETURN DISTINCT prereq.id as id
                """
            else:
                query = """
                MATCH (c:Concept {id: $id})-[:PREREQUISITE]->(prereq:Concept)
                RETURN prereq.id as id
                """
            
            result = session.run(query, id=concept_id)
            return [record["id"] for record in result]
    
    def get_shortest_path(self, start_id: str, end_id: str) -> List[str]:
        """获取最短学习路径"""
        with self.driver.session() as session:
            query = """
            MATCH path = shortestPath(
                (start:Concept {id: $start})-[:PREREQUISITE|SUCCESSOR*]->(end:Concept {id: $end})
            )
            RETURN [node in nodes(path) | node.id] as path_ids
            """
            result = session.run(query, start=start_id, end=end_id)
            record = result.single()
            return record["path_ids"] if record else []
```

```python
            if rule["condition"](user_profile):
                reminder_time = self._parse_time(rule["time"], user_profile.timezone)

                            count=self._get_review_count(user_profile),
                            days=user_profile.streak_days
                        ),
                        "send_at": reminder_time,
                        "priority": "high" if rule_name == "streak_warning" else "normal"
                    })
        
        return reminders
    
    def _has_review_due(self, user_profile: UserProfile) -> bool:
        """检查是否有需要复习的概念"""
        for concept_id, mastery in user_profile.mastered_concepts.items():
            last_review = user_profile.last_review_dates.get(concept_id)
            if last_review:
                days_since_review = (datetime.now() - last_review).days
                # 根据掌握度确定复习间隔
                interval = 1 if mastery < 0.7 else (7 if mastery < 0.9 else 30)
                if days_since_review >= interval:
                    return True
```

```typescript
// components/PathVisualization.tsx
import React, { useEffect, useRef } from 'react';
import * as d3 from 'd3';

interface PathNode {
  id: string;
  name: string;
  status: 'pending' | 'in_progress' | 'completed';
  x: number;
  y: number;
}

interface PathVisualizationProps {
  nodes: PathNode[];
  onNodeClick: (node: PathNode) => void;
}

export const PathVisualization: React.FC<PathVisualizationProps> = ({
  nodes,
  onNodeClick
}) => {
  const svgRef = useRef<SVGSVGElement>(null);

  useEffect(() => {
    if (!svgRef.current) return;

    const svg = d3.select(svgRef.current);
    svg.selectAll('*').remove();

    const width = 800;
    const height = 400;

    // 绘制连接线
    svg.selectAll('line')
      .data(nodes.slice(0, -1))
      .enter()
      .append('line')
      .attr('x1', (d) => d.x)
      .attr('y1', (d) => d.y)
      .attr('x2', (_, i) => nodes[i + 1].x)
      .attr('y2', (_, i) => nodes[i + 1].y)
      .attr('stroke', '#ccc')
      .attr('stroke-width', 2);

    // 绘制节点
    const nodeGroups = svg.selectAll('g.node')
      .data(nodes)
      .enter()
      .append('g')
      .attr('class', 'node')
      .attr('transform', (d) => `translate(${d.x},${d.y})`)
      .on('click', (_, d) => onNodeClick(d));

    // 节点圆形
    nodeGroups.append('circle')
      .attr('r', 20)
      .attr('fill', (d) => {
        switch (d.status) {
          case 'completed': return '#52c41a';
          case 'in_progress': return '#1890ff';
          default: return '#d9d9d9';
        }
      })
      .attr('stroke', '#fff')
      .attr('stroke-width', 3);

    // 节点标签
    nodeGroups.append('text')
      .text((d) => d.name)
      .attr('text-anchor', 'middle')
      .attr('dy', 35)
      .attr('font-size', '12px');

  }, [nodes, onNodeClick]);

  return (
    <svg
      ref={svgRef}
      width={800}
      height={400}
      style={{ border: '1px solid #eee' }}
    />
  );
};
```

### 6.2 学习进度追踪

```typescript
// components/ProgressTracker.tsx
import React from 'react';
import { Progress, Tooltip } from 'antd';
import { CheckCircleOutlined, LoadingOutlined } from '@ant-design/icons';

interface LearningNode {
  id: string;
  name: string;
  status: 'pending' | 'in_progress' | 'completed';
  estimatedTime: number;
  spentTime?: number;
}

export const ProgressTracker: React.FC<{ nodes: LearningNode[] }> = ({ nodes }) => {
  const completedCount = nodes.filter(n => n.status === 'completed').length;
  const progress = Math.round((completedCount / nodes.length) * 100);

  const totalEstimatedTime = nodes.reduce((sum, n) => sum + n.estimatedTime, 0);
  const totalSpentTime = nodes.reduce((sum, n) => sum + (n.spentTime || 0), 0);

  return (
    <div className="progress-tracker">
      <div className="overall-progress">
        <h3>学习进度</h3>
        <Progress
          percent={progress}
          status={progress === 100 ? 'success' : 'active'}
          format={(percent) => `${percent}% (${completedCount}/${nodes.length})`}
        />
      </div>

      <div className="time-stats">
        <Tooltip title="预计学习时间">
          <span>⏱️ {Math.round(totalEstimatedTime / 60)}小时</span>
        </Tooltip>
        <Tooltip title="已学习时间">
          <span>✓ {Math.round(totalSpentTime / 60)}小时</span>
        </Tooltip>
      </div>

      <div className="node-list">
        {nodes.map((node, index) => (
          <div
            key={node.id}
            className={`node-item ${node.status}`}
          >
            <span className="node-number">{index + 1}</span>
            {node.status === 'completed' && <CheckCircleOutlined className="status-icon completed" />}
            {node.status === 'in_progress' && <LoadingOutlined className="status-icon in-progress" />}
            <span className="node-name">{node.name}</span>
            <span className="node-time">{node.estimatedTime}分钟</span>
          </div>
        ))}
      </div>
    </div>
  );
};
```
