```python
from concurrent.futures import ThreadPoolExecutor

def _update_knowledge_state_parallel(self, ...):
    """并行更新知识状态"""
    with ThreadPoolExecutor(max_workers=4) as executor:
        futures = []
        for node_id in self.knowledge_graph.nodes:
            future = executor.submit(
                self._update_single_node, node_id, ...
            )
            futures.append((node_id, future))
        
        for node_id, future in futures:
            new_state[node_id] = future.result()
```

#### 建议2: 缓存IRT计算结果
```python
from functools import lru_cache

@lru_cache(maxsize=1024)
def _calculate_irt_probability(self, theta: float, a: float, b: float) -> float:
    """缓存IRT概率计算"""
    return 1 / (1 + np.exp(-a * (theta - b)))
```

#### 建议3: 提前收敛检测
```python
def _check_early_convergence(self, prev_state, curr_state, iteration):
    """提前检测收敛，避免不必要的迭代"""
    if iteration < 5:  # 最少迭代次数
        return False
    
    # 检查最大变化是否小于阈值
    max_diff = max(
        abs(prev_state[k] - curr_state[k])
        for k in prev_state
    )
    return max_diff < self.convergence_threshold * 0.5
```

## 4. 功能增强建议

### 4.1 自适应测试(CAT)

实现基于IRT的计算机自适应测试：

```python
class AdaptiveTestSelector:
    """自适应题目选择器"""
    
    def select_next_question(self, session: DiagnosisSession) -> Question:
        """基于当前能力估计选择最优题目"""
        # 获取未作答题目
        unanswered = self.get_unanswered_questions(session)
        
        # 计算每道题的信息量
        best_question = None
        max_info = 0
        
        for question in unanswered:
            info = self.calculate_information(
                theta=session.current_ability_estimate,
                a=question.discrimination,
                b=question.difficulty
            )
            if info > max_info:
                max_info = info
                best_question = question
        
        return best_question
    
    def calculate_information(self, theta: float, a: float, b: float) -> float:
        """计算Fisher信息量"""
        p = 1 / (1 + np.exp(-a * (theta - b)))
        return a ** 2 * p * (1 - p)
```

### 5.1 单元测试

```python
# tests/test_hcd_engine.py
import pytest
import numpy as np
from app.diagnosis.hcd_engine import HCDEngine, HCDResult
from app.models.knowledge_graph import KnowledgeGraph

class TestHCDEngine:
    """HCD引擎测试"""
    
    @pytest.fixture
    def engine(self):
        kg = KnowledgeGraph()  # 使用测试图谱
        return HCDEngine(kg)
    
    def test_initialization(self, engine):
        """测试引擎初始化"""
        assert engine.max_iterations == 100
        assert engine.convergence_threshold == 0.001
    
    def test_irt_probability_calculation(self, engine):
        """测试IRT概率计算"""
        # P(theta=0, a=1, b=0) 应该约等于 0.5
        prob = engine._calculate_irt_probability(0, 1, 0)
        assert 0.49 < prob < 0.51
    
    def test_hierarchy_constraint_soft(self, engine):
        """测试软层次约束"""
        state = {"L0_A": 0.3, "L1_A": 0.9}
        engine.constraint_policy = ConstraintViolationPolicy.SOFT
        
        new_state = engine._apply_hierarchy_constraints(state)
        
        # L1应该被惩罚降低
        assert new_state["L1_A"] < state["L1_A"]
    
    def test_hierarchy_constraint_hard(self, engine):
        """测试硬层次约束"""
        state = {"L0_A": 0.3, "L1_A": 0.9}
        engine.constraint_policy = ConstraintViolationPolicy.HARD
        
        new_state = engine._apply_hierarchy_constraints(state)
        
        # L1不应超过L0
        assert new_state["L1_A"] <= new_state["L0_A"]
    
    def test_convergence_detection(self, engine):
        """测试收敛检测"""
        prev = {"A": 0.5, "B": 0.6}
        curr = {"A": 0.5001, "B": 0.6001}
        
        assert engine._check_convergence(prev, curr)
    
    def test_diagnose_with_empty_responses(self, engine):
        """测试空响应处理"""
        result = engine.diagnose([], {})
        
        assert isinstance(result, HCDResult)
        assert result.iterations == 1
        assert not result.converged
```
