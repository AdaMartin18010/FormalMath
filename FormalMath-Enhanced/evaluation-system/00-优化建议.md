### 2.2 架构级别问题

#### 问题4: 缺少缓存机制 ⚠️
**描述**: 频繁查询的数据没有缓存

**影响**: 数据库压力大，响应慢

**建议方案**:
```python
import redis
from functools import wraps

cache = redis.Redis(host='localhost', port=6379, db=0)

def cache_result(key_prefix, ttl=300):
    def decorator(func):
        @wraps(func)
        async def wrapper(*args, **kwargs):
            cache_key = f"{key_prefix}:{hash(str(args) + str(kwargs))}"
            
            # 尝试从缓存获取
            cached = cache.get(cache_key)
            if cached:
                return json.loads(cached)
            
            # 执行函数
            result = await func(*args, **kwargs)
            
            # 缓存结果
            cache.setex(cache_key, ttl, json.dumps(result))
            return result
        return wrapper
    return decorator

# 使用示例

    # 查询逻辑...
```

```python
@router.get("/records/{user_id}")
async def get_user_records(
```

```python
from sklearn.linear_model import LinearRegression
import numpy as np

class PerformancePredictor:
    """成绩预测器"""
    
    def predict_future_performance(
        self,
        historical_data: List[Dict],
        periods_ahead: int = 3
    ) -> Dict:
        """基于历史数据预测未来表现"""
        
        # 准备数据
        X = np.array([[i] for i in range(len(historical_data))])
        
        predictions = {}
        dimensions = ['knowledge_mastery', 'problem_solving', 
                     'proof_ability', 'application', 'innovation']
        
        for dim in dimensions:
            y = np.array([d[dim] for d in historical_data])
            
            # 训练模型
            model = LinearRegression()
            model.fit(X, y)
            
            # 预测未来
            future_X = np.array([[len(historical_data) + i] 
                                for i in range(periods_ahead)])
            future_scores = model.predict(future_X)
            
            predictions[dim] = {
                "trend": "up" if model.coef_[0] > 0 else "down",
                "slope": float(model.coef_[0]),
                "predicted_scores": future_scores.tolist(),
                "r2_score": float(model.score(X, y))
            }
        
        return predictions
```

### 4.3 高级对比分析

```python
class AdvancedComparisonAnalyzer:
    """高级对比分析器"""
    
    def compare_multiple_dimensions(
        self,
        user_scores: Dict[str, float],
        class_scores: Dict[str, List[float]],
        historical_percentiles: List[float]
    ) -> Dict:
        """多维度对比分析"""
        
        results = {}
        
        for dimension, score in user_scores.items():
            class_scores_dim = class_scores.get(dimension, [])
            
            if not class_scores_dim:
                continue
            
            # 计算统计指标
            mean = np.mean(class_scores_dim)
            std = np.std(class_scores_dim)
            
            # 计算z-score
            z_score = (score - mean) / std if std > 0 else 0
            
            # 计算百分位
            percentile = sum(1 for s in class_scores_dim if s < score) / len(class_scores_dim) * 100
            
            # 趋势分析
            trend = "improving" if len(historical_percentiles) > 1 and \
                    historical_percentiles[-1] > historical_percentiles[0] else "stable"
            
            results[dimension] = {
                "user_score": score,
                "class_mean": mean,
                "class_std": std,
                "z_score": z_score,
                "percentile": percentile,
                "trend": trend,
                "significance": "significant" if abs(z_score) > 1.96 else "not_significant"
            }
        
        return results
```

```python
from fastapi.responses import StreamingResponse
import csv
import io

@router.get("/export/{user_id}")
async def export_evaluation_data(
    user_id: str,
    format: str = Query("csv", enum=["csv", "json", "xlsx"]),
    start_date: Optional[str] = None,
    end_date: Optional[str] = None,
    db: Session = Depends(get_db)
):
    """导出评估数据"""
    
    # 查询数据
    query = db.query(EvaluationRecord).filter(
        EvaluationRecord.user_id == user_id
    )
    
    if start_date:
        query = query.filter(EvaluationRecord.created_at >= start_date)
    if end_date:
        query = query.filter(EvaluationRecord.created_at <= end_date)
    
    records = query.all()
    
    if format == "csv":
        output = io.StringIO()
        writer = csv.writer(output)
        writer.writerow(["Record ID", "Date", "Knowledge", "Problem Solving", 
                        "Proof", "Application", "Innovation", "Total"])
        
        for r in records:
            writer.writerow([
                r.record_id, r.created_at, r.knowledge_mastery,
                r.problem_solving, r.proof_ability, r.application,
                r.innovation, r.weighted_score
            ])
        
        output.seek(0)
        return StreamingResponse(
            io.BytesIO(output.getvalue().encode()),
            media_type="text/csv",
            headers={"Content-Disposition": f"attachment; filename={user_id}_evaluations.csv"}
        )
```

```python
            async with session.post(

                json=payload

                return await response.json()
        
        async def run_concurrent():
            payload = {

                "scores": {"knowledge_mastery": 80.0, ...}

            async with aiohttp.ClientSession() as session:

                return results
        
        results = asyncio.run(run_concurrent())
        assert all(r.get("user_id") == "concurrent_test" for r in results)
```
